\begin{frame}{Exercițiul 1: Log-Randamente}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        Fie o serie de prețuri $P_t$. Definim log-randamentele $r_t = \log(P_t / P_{t-1})$.
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Reprezentați grafic seria $P_t$ și seria de randamente $r_t$. Ce caracteristici vizuale observați?\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Calculați ACF-ul eșantion al $r_t$ și al $r_t^2$ pentru decalajele $h = 1, \ldots, 20$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Ce sugerează un ACF aproape nul al $r_t$ despre predictibilitate? Ce indică valorile semnificative ale ACF al $r_t^2$?
    \end{block}

    \vspace{0.3cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 1: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\small
        \textcolor{MainBlue}{\textbf{(A)}} $P_t$ prezintă de obicei un trend nestaţionar; $r_t$ oscilează în jurul valorii zero cu perioade de volatilitate ridicată şi scăzută (aglomerarea volatilității).\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} ACF($r_t$): majoritatea valorilor se încadrează în benzile de confidenţă $\pm 1{,}96/\sqrt{n}$ $\Rightarrow$ randamentele sunt aproximativ necorelate. ACF($r_t^2$): valori pozitive semnificative la multe decalaje $\Rightarrow$ randamentele la pătrat sunt corelate.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} ACF($r_t$) $\approx 0$ $\Rightarrow$ randamentele sunt aproape \textbf{zgomot alb}: informaţia din trecut nu ajută la prognoza randamentelor viitoare (ipoteza eficienţei slabe a pieţei). ACF($r_t^2$) semnificativ $\Rightarrow$ \textbf{efecte ARCH/GARCH}: volatilitatea este predictibilă chiar şi atunci când randamentele nu sunt.
        }
    \end{exampleblock}

\begin{frame}{Exercițiul 2: Periodograma Brută}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        Fie $n = 256$ observaţii. Periodograma la frecvenţa Fourier $\omega_j = 2\pi j/n$ este:
        \[
            I(\omega_j) = \frac{1}{2\pi n}\left|d(\omega_j)\right|^2, \qquad
            d(\omega_j) = \sum_{t=1}^{n} Y_t\, e^{-i\omega_j t}
        \]
    \end{alertblock}

    \vspace{0.15cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Reprezentaţi $I(\omega_j)$ în funcţie de $\omega_j \in [0, \pi]$. Cum identificaţi frecvenţa dominantă?\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Dacă seria are un tipar trimestrial (perioadă $= 4$), la ce frecvenţă unghiulară $\omega^*$ aşteptaţi vârful cel mai mare al periodogramei?\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Ce indică o periodogramă complet plată?
    \end{block}

    \vspace{0.15cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 2: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\small
        \textcolor{MainBlue}{\textbf{(A)}} Frecvenţa dominantă este $\omega_j$ la care $I(\omega_j)$ atinge maximul global. Perioada corespunzătoare este $T = 2\pi / \omega_j$ unităţi de timp.\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Un tipar trimestrial are perioada $T = 4$, deci:
            $\omega^* = 2\pi/4 = \pi/2 \approx 1{,}571$ rad.
            Aceasta corespunde celei de-a 64-a frecvenţe Fourier pentru $n = 256$ ($j^* = n/T = 64$).\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} O periodogramă plată $I(\omega_j) \approx c$ pentru orice $j$ indică \textbf{zgomot alb}: toate frecvenţele contribuie în mod egal la varianţa seriei ($\int_0^\pi f(\omega)\,d\omega = \gamma(0)$, unde $f(\omega) = \sigma^2/(2\pi) = $ const).
        }
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_periodogram}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_periodogram}
\end{frame}

\begin{frame}{Exercițiul 3: Inconsistența Periodogramei}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        Arătaţi că periodograma este un estimator \emph{inconsistent} al densităţii spectrale.
    \end{alertblock}

    \vspace{0.2cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Enunţaţi distribuţia asimptotică a $I(\omega_j)$ când $n \to \infty$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Derivaţi $\text{Var}[I(\omega_j)]$ în funcţie de $n$. Converge la zero?\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Propuneţi o soluţie care produce un estimator consistent şi explicaţi compromisul implicat.
    \end{block}

    \vspace{0.3cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 3: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\footnotesize
        \textcolor{MainBlue}{\textbf{(A)}} Pentru $\omega_j \neq 0, \pi$:
            $\dfrac{2\,I(\omega_j)}{f(\omega_j)} \xrightarrow{d} \chi^2_2 \;\Rightarrow\; I(\omega_j) \stackrel{d}{\approx} f(\omega_j)\cdot\chi^2_2/2$\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} $\text{Var}[I(\omega_j)] \approx f(\omega_j)^2$. Aceasta \textbf{nu scade} cu $n$ $\Rightarrow$ periodograma este \textbf{inconsistentă}.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} \textbf{Netezire} peste $m$ ordonate adiacente:
            $\hat{f}(\omega_j) = \frac{1}{2m+1}\sum_{k=-m}^{m} I(\omega_{j+k})$.
            Varianţă $\propto 1/m$, rezoluţie $\propto m/n$. Netezire mai mare $\Rightarrow$ varianţă mai mică, bias mai mare.
        }
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_periodogram}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_periodogram}
\end{frame}

    \begin{block}{Exercițiu}

        \textcolor{MainBlue}{\textbf{(B)}} Calculaţi raportul $f(0)/f(\pi)$ şi interpretaţi rezultatul.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Ce este un \emph{spectru roșu}? Daţi un exemplu economic.
    \end{block}

    \vspace{0.3cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 4: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\small
        \textcolor{MainBlue}{\textbf{(A)}} Pentru AR(1) cu $\phi = 0{,}8$:
            $f(\omega) = \frac{1}{2\pi}\cdot\frac{1}{1{,}64 - 1{,}6\cos\omega}$\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} $f(0) = \frac{25}{2\pi}$, $f(\pi) = \frac{1}{2\pi \cdot 3{,}24}$, deci
            $\frac{f(0)}{f(\pi)} = \left(\frac{1{,}8}{0{,}2}\right)^2 = 81$.
            Cea mai mare parte a varianţei este concentrată la frecvenţe joase.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Un \textbf{spectru roșu} este dominat de frecvenţe joase: cea mai mare parte a varianţei provine din oscilaţii lente (cicluri lungi). PIB-ul, consumul şi preţurile acţiunilor prezintă spectre roşii --- evoluează lent şi persistent.
        }
    \end{exampleblock}
    \end{cminipage}

\begin{frame}{Exercițiul 5: Welch vs.\ Multitaper}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        Comparaţi doi estimatori spectrali aplicaţi aceleiaşi serii cu $n = 512$ observaţii:
        \textbf{Welch}: $L = 4$ segmente suprapuse; \quad \textbf{Multitaper}: $NW = 4$, $K = 7$ tapere DPSS.
    \end{alertblock}

    \vspace{0.2cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Calculaţi rezoluţia în frecvenţă (lăţimea de bandă) pentru fiecare estimator.\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Care estimator are varianţă mai mică? De ce?\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Descrieţi compromisul bias-varianţă pentru fiecare metodă.
    \end{block}

    \vspace{0.3cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}

\begin{frame}{Exercițiul 5: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\footnotesize
        \textcolor{MainBlue}{\textbf{(A)}} Welch ($L=4$, $n_{seg}=128$): $\Delta\omega_{\text{Welch}} \approx 2/128 \approx 0{,}0156$ cicluri/obs.
            Multitaper ($NW=4$, $n=512$): $\Delta\omega_{\text{MT}} \approx 8/512 = 0{,}0156$ cicluri/obs.

        \textcolor{MainBlue}{\textbf{(C)}} Ambele metode sacrifică rezoluţia pentru varianţă mai mică. Multitaper suprimă în mod optim lobii laterali prin taperele DPSS --- este preferat în majoritatea aplicaţiilor.
        }

\begin{frame}{Exercițiul 7: Funcția de Transfer a Filtrului HP}
    \begin{cminipage}{0.95\textwidth}
    {\small
    \begin{alertblock}{Întrebare}
        Filtrul HP extrage trendul $\tau_t$ minimizând
        $\sum(Y_t-\tau_t)^2 + \lambda\sum(\Delta^2\tau_t)^2$.
        Câştigul la putere al componentei \emph{ciclu} este:
        $|A_{HP}(\omega)|^2 = \frac{\lambda\,[2(1-\cos\omega)]^2}{1 + \lambda\,[2(1-\cos\omega)]^2}$
    \end{alertblock}

    \vspace{0.1cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Evaluaţi $|A_{HP}(\omega)|^2$ la $\omega = 0$ şi $\omega = \pi$ pentru orice $\lambda$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Găsiţi frecvenţa de tăiere la jumătate din putere pentru $\lambda = 1600$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Rezumaţi critica lui Hamilton (2018) la adresa filtrului HP.
    \end{block}

    \vspace{0.1cm}
    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    }
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 7: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\footnotesize
        \textcolor{MainBlue}{\textbf{(A)}} La $\omega=0$: $2(1-\cos 0)=0 \Rightarrow |A_{HP}(0)|^2=0$ (trend eliminat).
            La $\omega=\pi$: $|A_{HP}(\pi)|^2 = 16\lambda/(1+16\lambda)\approx 1$ (frecv.\ înalte trec).\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Condiţia $|A_{HP}(\omega)|^2=0{,}5$: $\lambda[2(1-\cos\omega)]^2=1$, deci
            $2(1-\cos\omega)=0{,}025$, $\omega^*\approx 0{,}158$ rad $\approx 2\pi/40$ (perioadă $\approx 10$ ani).\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} \textbf{Hamilton (2018)}: filtrul HP este echivalent cu o regresie pe valori decalate. Ciclul obţinut este corelat spurios la toate decalajele şi impune structură ARMA(4,4) zgomotului alb --- generând \emph{cicluri artificiale}. Hamilton recomandă o regresie simplă în 2 paşi.
        }
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_filters}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_filters}

\begin{frame}{Exercițiul 8: Estimatorul GPH}
    \begin{cminipage}{0.95\textwidth}
    {\small
    \begin{alertblock}{Întrebare}
        Pentru ARFIMA(0,$d$,0), estimatorul GPH aplică MCO la:
        \[
            \log I(\omega_j) = c - d\log[4\sin^2(\omega_j/2)] + u_j, \quad j=1,\ldots,m
        \]
    \end{alertblock}

    \vspace{0.1cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Care este estimatorul MCO $\hat{d}$? Cum se calculează?\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Ce lăţime de bandă $m$ ar trebui utilizată? Recomandări teoretice?\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Care este eroarea standard asimptotică a lui $\hat{d}$?
    \end{block}

    \vspace{0.1cm}
    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    }
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 8: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\footnotesize
        \textcolor{MainBlue}{\textbf{(A)}} Fie $x_j = -\log[4\sin^2(\omega_j/2)]$ şi $y_j = \log I(\omega_j)$:
            $\hat{d} = \frac{\sum_{j=1}^m (x_j - \bar{x})(y_j - \bar{y})}{\sum_{j=1}^m (x_j - \bar{x})^2}$ \quad (panta MCO)\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} $m = O(n^\alpha)$, $\alpha \in (0,1)$; practică: $m \approx n^{0{,}5}$ până la $n^{0{,}6}$. $m$ mic $\Rightarrow$ varianţă mare; $m$ mare $\Rightarrow$ bias.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} $\text{ES}(\hat{d}) = \pi/\sqrt{6m}$. Pentru $n=1000$, $m=32$: ES $\approx 0{,}226$. Hurst: $H = d+0{,}5$; $d>0 \Rightarrow H>0{,}5$ (persistenţă).
        }
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_long\_memory}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_long_memory}
\end{frame}

\begin{frame}{Exercițiul 9: Estimatorul Whittle Local}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        Comparaţi estimatorii GPH şi Whittle Local pentru o serie cu $n = 1000$ observaţii şi $d = 0{,}3$ adevărat.
    \end{alertblock}

    \vspace{0.2cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Scrieţi funcţia obiectiv a Whittle Local şi precizaţi ce minimizează.\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Ce lăţime de bandă $m$ se foloseşte şi care este varianţa asimptotică a $\hat{d}_{WL}$?\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} De ce este Whittle Local mai eficient decât GPH?
    \end{block}

    \vspace{0.3cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 9: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\footnotesize
        \textcolor{MainBlue}{\textbf{(A)}} Whittle Local minimizează log-verosimilitatea restricţionată:
            \[
                Q(d) = \log G(d) - 2d\,\bar{\lambda}_m, \quad G(d) = \frac{1}{m}\sum_{j=1}^{m}\frac{I(\omega_j)}{\omega_j^{-2d}}, \quad \hat{d}_{WL} = \arg\min_d Q(d)
            \]
        \textcolor{MainBlue}{\textbf{(B)}} $m=O(n^{0{,}6\text{--}0{,}8})$; pentru $n=1000$ tipic $m\approx 100$--$250$. $\text{Var}(\hat{d}_{WL})=1/(4m)$; pentru $m=100$: ES $=0{,}05$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} GPH foloseşte MCO pe log-periodogramă (erori $\chi^2_2$ cu coadă grea); WL foloseşte o verosimilitate propriu-zisă. WL este asimptotic eficient: $\text{Var}=1/(4m)$ faţă de $\approx 1{,}645/m$ (GPH) pentru acelaşi $m$.
        }
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_long\_memory}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_long_memory}
\end{frame}

\begin{frame}{Exercițiul 10: Analiza Spectrală Încrucișată}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        Presupunem că $Y_t = X_{t-4} + \varepsilon_t$, unde $\varepsilon_t \perp X_s$ pentru orice $s, t$.
    \end{alertblock}

    \vspace{0.2cm}

    \begin{block}{Exercițiu}
        \textcolor{MainBlue}{\textbf{(A)}} Derivaţi spectrul încrucişat $S_{XY}(\omega) = e^{-4i\omega}S_{XX}(\omega)$. Care este funcţia de fază $\phi(\omega)$?\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Calculaţi coerenţa la pătrat $C^2_{XY}(\omega)$: (i) fără zgomot, (ii) cu zgomot $\varepsilon_t \sim (0, \sigma^2_\varepsilon)$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Ce indică panta fazei $-d\phi/d\omega$ şi care este valoarea sa în acest caz?
    \end{block}

    \vspace{0.2cm}

    \begin{center}
        \textit{Răspunsul pe slide-ul următor...}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Exercițiul 10: Soluție}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Soluție}
        {\small
        \textcolor{MainBlue}{\textbf{(A)}} Funcţia de transfer $H(\omega) = e^{-4i\omega}$, deci $S_{XY}(\omega) = e^{-4i\omega}\,S_{XX}(\omega)$.
            Faza: $\phi(\omega) = -4\omega$ (liniară în $\omega$).\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} $C^2_{XY}(\omega) = |S_{XY}|^2/(S_{XX} \cdot S_{YY})$:
            (i) Fără zgomot: $C^2_{XY}=1$. \quad (ii) Cu zgomot: $C^2_{XY} = S_{XX}/(S_{XX}+\sigma^2_\varepsilon/(2\pi)) < 1$.\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} \textbf{Întârzierea de grup} $= -d\phi/d\omega = 4$ perioade la toate frecvenţele --- confirmă o deplasare pură de 4 perioade, fără distorsiune de fază.
        }
    \end{exampleblock}
    \end{cminipage}

\end{frame}

\begin{frame}{Discuție: Alegerea Estimatorului Spectral}

    {\small
    \begin{alertblock}{Întrebare Cheie}
        Aveţi $n = 1000$ observaţii. Ce estimator spectral folosiţi şi de ce?

        \begin{enumerate}\setlength{\itemsep}{0pt}
            \item \textbf{Doriţi o privire rapidă asupra frecvenţelor dominante?}

            \item \textbf{Serie scurtă sau segmente nestaţionare?}

                    \item Welch (segmente suprapuse) --- reduce sensibilitatea la nestaţionaritate
                \end{itemize}
            \item \textbf{Seria urmează un model AR/ARMA cunoscut?}
                \begin{itemize}\setlength{\itemsep}{0pt}
                    \item Spectrul AR parametric --- neted şi consistent cu modelul
                \end{itemize}
        \end{enumerate}
    \end{block}
    }

\begin{frame}{Discuție: Interpretarea Spectrelor Economice}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\small
    \begin{alertblock}{Întrebare Cheie}
        Calculaţi densitatea spectrală a creşterii trimestriale a PIB şi găsiţi cea mai mare putere la frecvenţe sub $\pi/16$. Ce implică aceasta?
    \end{alertblock}
    \begin{block}{Ghid de Interpretare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Frecvenţă $\omega < \pi/16$ $\Rightarrow$ perioadă $> 32$ trimestre $= 8$ ani: \textbf{cicluri lungi}
            \item Spectrul este \textbf{roşu}: domină componenta persistentă, cu evoluţie lentă
            \item Banda ciclului de afaceri: $\omega \in [\pi/16, \pi/3]$ (perioade 6--32 trimestre)
        \end{itemize}
    \end{block}
    \begin{exampleblock}{Reguli Practice}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Trend}: putere la $\omega \approx 0$ (frecvenţă foarte joasă)
            \item \textbf{Ciclu de afaceri}: putere în banda Baxter-King (6--32 trimestre)
            \item \textbf{Sezonalitate}: vârfuri clare la $\omega = \pi/2$ (trimestrial), $2\pi/12$ (lunar)
            \item \textbf{Zgomot alb}: spectru plat la toate frecvenţele
        \end{itemize}
    \end{exampleblock}
    }
    \end{cminipage}
    \quantlet{TSA\_ch12\_gdp}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_gdp}
\end{frame}

\begin{frame}{Discuție: Filtrul HP în Practică}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    {\small
    \begin{alertblock}{Întrebare Cheie}
        Ar trebui economiştii să renunţe la filtrul HP? Care sunt alternativele?
    \end{alertblock}
    \begin{block}{Argumente Împotrivă (Hamilton 2018)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Impune structură ARMA(4,4) spurioasă zgomotului alb
            \item Generează cicluri artificiale absente din date
            \item Estimările la sfârşitul eşantionului sunt nesigure
            \item Rezultatele nu sunt invariante la alegerea lui $\lambda$
        \end{itemize}
    \end{block}
    \begin{exampleblock}{Alternative}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Filtrul Hamilton}: regresia $Y_{t+h}$ pe $Y_t, Y_{t-1}, Y_{t-2}, Y_{t-3}$; reziduul ca ciclu
            \item \textbf{Filtrul bandpass} (Baxter-King, Christiano-Fitzgerald): izolează direct frecvenţele ciclului de afaceri
            \item \textbf{Model cu componente neobservate}: modelare explicită trend + ciclu (metode state-space)
        \end{itemize}
    \end{exampleblock}
    }
    \end{cminipage}

\begin{frame}{Exerciții pentru Acasă}
    \begin{cminipage}{0.95\textwidth}
    {\footnotesize
    \begin{enumerate}\setlength{\itemsep}{2pt}
        \item \textbf{Teoretic}: Arătaţi că pentru zgomot alb $\varepsilon_t \sim \mathrm{WN}(0,\sigma^2)$, densitatea spectrală este $f(\omega) = \sigma^2/(2\pi)$ pentru orice $\omega \in [0,\pi]$.

        \item \textbf{Calcul}: Pentru procesul AR(2) $Y_t = 0{,}6Y_{t-1} - 0{,}3Y_{t-2} + \varepsilon_t$:
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Derivaţi analitic densitatea spectrală $f(\omega)$
                \item Simulaţi $n = 512$ observaţii şi comparaţi periodograma cu spectrul teoretic
                \item Aplicaţi estimarea Welch şi multitaper; comparaţi toate trei metode
            \end{itemize}

        \item \textbf{Aplicat}: Descărcaţi datele lunare privind numărul de pete solare (\texttt{sidc.be}):
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Estimaţi densitatea spectrală şi identificaţi ciclul de $\approx 11$ ani
                \item Testaţi memoria lungă; raportaţi $\hat{d}$ prin GPH şi Whittle Local
            \end{itemize}

        \item \textbf{Gândire Critică}: De ce aplicarea filtrului HP cu $\lambda = 1600$ pe date lunare (în loc de trimestriale) produce rezultate eronate?
    \end{enumerate}
    }
    \end{cminipage}
    \quantlet{TSA\_ch12\_sunspot}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_sunspot}
\end{frame}

\begin{frame}{Indicii pentru Soluții}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\footnotesize
    \begin{block}{Indicii}
        \begin{enumerate}\setlength{\itemsep}{1pt}
            \item Prin teorema Wiener-Khinchin: $f(\omega) = \frac{1}{2\pi}\sum_{h}\gamma(h)e^{-i\omega h}$. Pentru zgomot alb: $\gamma(0)=\sigma^2$, $\gamma(h)=0$ pentru $h\neq 0$, deci $f(\omega) = \sigma^2/(2\pi)$ --- plat.

            \item Spectrul AR(2): $f(\omega) = \frac{\sigma^2}{2\pi|1 - 0{,}6e^{-i\omega} + 0{,}3e^{-2i\omega}|^2}$. Procesul are un vârf spectral (rădăcini complexe). Verificaţi numeric: vârful ar trebui să se afle între 0 şi $\pi/2$.

            \item Ciclul petelor solare: aşteptaţi un vârf clar la $\omega = 2\pi/(11 \times 12) \approx 0{,}048$ rad/lună. Testele de memorie lungă dau de obicei $\hat{d} \approx 0{,}3$--$0{,}5$ pentru petele solare.

            \item Calibrarea filtrului HP: $\lambda = 1600$ este proiectat pentru date trimestriale. Pentru date lunare, valoarea corectă este $\lambda \approx 1600 \times 3^4 = 129{.}600$. Folosind $\lambda=1600$ pe date lunare, frecvenţa de tăiere este greşită, distorsionând ciclul extras.
        \end{enumerate}
    \end{block}
    }
    \end{cminipage}
    \quantlet{TSA\_ch12\_filters}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_filters}
\end{frame}
