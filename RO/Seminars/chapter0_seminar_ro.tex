        \textbf{Întrebare:} Care NU este o caracteristică a datelor de tip serie de timp?

        \vspace{0.3cm}

        \begin{enumerate}[A.]
            \item Observațiile sunt ordonate în timp \incorrect
            \item Observațiile consecutive sunt de obicei corelate \incorrect
            \item \textbf{\textcolor{Forest}{Observațiile sunt independente și identic distribuite}} \correct

        \end{enumerate}

            \item Observațiile seriilor de timp sunt \textbf{dependente} (autocorelate), nu independente

            \item Această dependență temporală necesită \textbf{metode specializate}
        \end{itemize}

\begin{frame}{Quiz 3: Răspuns}
    \begin{exampleblock}{Răspuns: C -- Prognozele reacționează rapid la schimbările recente}
        Cu $\alpha = 0.9$: $\hat{X}_{t+1} = 0.9 X_t + 0.1 \hat{X}_t$
        \begin{itemize}
            \item \textbf{$\alpha$ mare} (ex.\ 0.9): 90\% pondere pe ultima observație
                \begin{itemize}
                    \item Prognoze foarte receptive la date noi
                \end{itemize}
            \item \textbf{$\alpha$ mic} (ex.\ 0.1): prognoze mai netede, mai stabile
                \begin{itemize}
                    \item Ia în calcul mai multe observații din trecut
                \end{itemize}
        \end{itemize}

\begin{frame}{Quiz 5: Răspuns}
    \begin{exampleblock}{Răspuns: C -- Eroarea Absolută Medie Procentuală (MAPE)}
        MAPE $= \frac{100}{n}\sum\left|\frac{e_t}{X_t}\right|$ exprimă erorile ca \textbf{procente}.

        \begin{itemize}
            \item MAE, RMSE, MSE sunt \textbf{dependente de scală} (unități ale lui $X_t$)
            \item MAPE este \textbf{independentă de scală} (întotdeauna în \%)
            \item Atenție: MAPE devine instabilă când $X_t \approx 0$
        \end{itemize}
    \end{exampleblock}

\begin{frame}{Quiz 6: Validare încrucișată}
    \begin{alertblock}{Întrebare}
        De ce nu putem folosi validarea încrucișată standard k-fold pentru seriile de timp?
    \end{alertblock}

    \vspace{0.4cm}

        \item Datele seriilor de timp sunt prea mici
        \item Ar încălca ordonarea temporală (viitorul prezicând trecutul)
        \item Validarea încrucișată este întotdeauna invalidă
        \item Seriile de timp nu au nevoie de validare
    \end{enumerate}

        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Date}: $X = [10, 12, 11, 14, 13]$ cu $\alpha = 0.3$, $\hat{X}_1 = 10$
            \item \textbf{Calculați}: a) Prognozele $\hat{X}_2$ până la $\hat{X}_6$; b) MAE și RMSE
            \item \textbf{Formula}: $\hat{X}_{t+1} = \alpha X_t + (1-\alpha)\hat{X}_t$
        \end{itemize}
    \end{alertblock}

    \vspace{0.2cm}
    \begin{exampleblock}{Soluție}
        \begin{center}
        \small
        \begin{tabular}{c|ccccc|c}
            $t$ & 1 & 2 & 3 & 4 & 5 & 6\\
            \hline
            $X_t$ & 10 & 12 & 11 & 14 & 13 & ?\\
            $\hat{X}_t$ & 10 & 10 & 10.6 & 10.72 & 11.70 & \textbf{12.09}\\
        \end{tabular}
        \end{center}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{MAE} $= 1.745$ \quad \textbf{RMSE} $= 2.04$
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{Exercițiu 2: Metrici de eroare}
    \begin{alertblock}{Problemă}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Date}: $X = [100, 110, 105, 120]$, $\hat{X} = [95, 108, 110, 115]$
            \item \textbf{Calculați}: MAE, MSE, RMSE, MAPE
        \end{itemize}
    \end{alertblock}

    \vspace{0.2cm}
    \begin{exampleblock}{Soluție}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Erori}: $e = [5, 2, -5, 5]$
            \item \textbf{MAE} $= (|5|+|2|+|-5|+|5|)/4 = 4.25$
            \item \textbf{MSE} $= (25+4+25+25)/4 = 19.75$
            \item \textbf{RMSE} $= \sqrt{19.75} = 4.44$
            \item \textbf{MAPE} $= 25 \times (0.05+0.018+0.048+0.042) = 3.95\%$
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{Exercițiu 3: Indici sezonieri}
    \begin{alertblock}{Problemă}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Date}: Indici sezonieri: $S = [0.85, 1.05, 0.90, 1.20]$, Trend T4: $T = 1000$
            \item \textbf{Calculați}: a) Verificați normalizarea. b) Prognoza T4. c) Desezonalizați $X_{T4} = 1150$
        \end{itemize}
    \end{alertblock}

    \vspace{0.2cm}
    \begin{exampleblock}{Soluție}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{a) Normalizare}: $\sum S_i = 0.85+1.05+0.90+1.20 = 4.00$ \checkmark
            \item \textbf{b) Prognoză}: $\hat{X}_{T4} = 1000 \times 1.20 = \textbf{1200}$
            \item \textbf{c) Desezonalizare}: $X_{desezonalizat} = 1150/1.20 = \textbf{958.33}$ (sub trend)
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{AI în analiza seriilor de timp}
    \begin{block}{De ce folosim instrumente AI în acest curs?}
        {\small

            \item Asistenții AI (Claude, ChatGPT, GitHub Copilot) pot genera cod și analiză
            \item Rolul vostru este să \textbf{evaluați, interpretați și criticați} --- competențe pe care AI nu le poate înlocui
        \end{itemize}
        }
    \end{block}

    \vspace{0.2cm}

    \textbf{Obiective de învățare:}
    \begin{itemize}\setlength{\itemsep}{1pt}
        \item Scrieți prompt-uri precise pentru sarcini econometrice
        \item Identificați erori în analiza statistică generată de AI
        \item Interpretați rezultatele critic folosind conceptele din curs
        \item Comparați soluțiile AI cu metodologia manuală
    \end{itemize}

    \vspace{0.2cm}

    \begin{alertblock}{Important}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item AI este un \textbf{instrument}, nu un înlocuitor al înțelegerii
            \item Trebuie să puteți explica \textit{de ce} fiecare pas este corect
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Exercițiu AI 1: Audit descompunere}
    \begin{block}{Rezultat AI (vânzări lunare retail SUA, 2015--2023)}
        {\small
        \textbf{Prompt}: ``Am un fișier cu vânzări lunare retail din SUA, din 2015 până în 2023. Ce tendințe și pattern-uri apar?''\\[2pt]
        \textbf{AI}: ``Am aplicat STL aditiv, perioadă=12. Trendul surprinde creșterea de la \$420 mld la \$580 mld. Componenta sezonieră este stabilă pe întreaga perioadă.''
        }
    \end{block}
    \vspace{-1mm}
    {\small
    \textbf{Datele observate}:
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Oscilația sezonieră dec--ian: $\pm\$18$ mld în 2015, $\pm\$45$ mld în 2023
        \item Deviația std a reziduurilor: \$3{,}2 mld (2015--2017) vs \$8{,}7 mld (2021--2023)
    \end{itemize}

    \vspace{1mm}
    \textbf{Cerințe}:
    \begin{enumerate}\setlength{\itemsep}{1pt}
        \item Ce tip de descompunere este adecvat pentru aceste date? Argumentați pe baza cifrelor de mai sus.
        \item Ce transformare aplicați seriei înainte de a folosi STL aditiv?
        \item Cum verificați dacă reziduurile au varianță constantă?
        \item Ce consecințe are alegerea greșită a tipului de descompunere asupra prognozei?
    \end{enumerate}
    }
\end{frame}
