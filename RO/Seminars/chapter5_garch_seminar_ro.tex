\begin{frame}{Întrebarea 2}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{În modelul GARCH(1,1): $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$}
        Ce reprezintă parametrul $\alpha$?
        \textcolor{MainBlue}{\textbf{(A)}} Persistența volatilității\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} Nivelul de bază al volatilității\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Reacția la șocuri recente (news coefficient)\\[3pt]
        \textcolor{MainBlue}{\textbf{(D)}} Varianța necondiționată
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Răspuns Întrebarea 2}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspuns Corect: (C)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Reacția la șocuri recente (news coefficient)}
        \end{itemize}
    \end{exampleblock}

    \pause
    \begin{block}{Interpretarea Parametrilor GARCH(1,1)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\omega$ = nivelul de bază (floor) al volatilității
            \item $\alpha$ = \textbf{reacția} la pătratele inovațiilor (``news'')
            \item $\beta$ = \textbf{persistența} volatilității (memory)
            \item $\alpha + \beta$ = persistența totală
        \end{itemize}

        \vspace{0.3cm}
        Un $\alpha$ mare înseamnă că volatilitatea reacționează puternic la șocuri recente.
    \end{block}
    \end{cminipage}
    \quantlet{TSA\_ch5\_garch\_sim}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_garch_sim}
\end{frame}

\begin{frame}{Întrebarea 3}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Care este condiția de stationaritate pentru GARCH(1,1)?}
        \textcolor{MainBlue}{\textbf{(A)}} $\omega > 0$\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} $\alpha + \beta = 1$\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} $\alpha + \beta < 1$\\[3pt]
        \textcolor{MainBlue}{\textbf{(D)}} $\alpha > \beta$
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Răspuns Întrebarea 3}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspuns Corect: (C)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{$\alpha + \beta < 1$}
        \end{itemize}
    \end{exampleblock}

    \pause
    \begin{block}{Condiții Complete}
        Pentru stationaritatea GARCH(1,1):
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\omega > 0$ (asigură varianță pozitivă)
            \item $\alpha \geq 0$, $\beta \geq 0$ (non-negativitate)
            \item $\alpha + \beta < 1$ (\textbf{stationaritate în covarianță})
        \end{itemize}

        \vspace{0.3cm}
        Dacă $\alpha + \beta = 1$ $\Rightarrow$ IGARCH (șocurile au efect permanent)
    \end{block}
    \end{cminipage}
    \quantlet{TSA\_ch5\_garch\_sim}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_garch_sim}
\end{frame}

\begin{frame}{Întrebarea 4}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Care este formula varianței necondiționate în GARCH(1,1)?}
        \textcolor{MainBlue}{\textbf{(A)}} $\bar{\sigma}^2 = \omega$\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} $\bar{\sigma}^2 = \dfrac{\omega}{1 - \alpha}$\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} $\bar{\sigma}^2 = \dfrac{\omega}{1 - \alpha - \beta}$\\[3pt]
        \textcolor{MainBlue}{\textbf{(D)}} $\bar{\sigma}^2 = \dfrac{\omega}{\alpha + \beta}$
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Răspuns Întrebarea 4}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspuns Corect: (C)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\bar{\sigma}^2 = \dfrac{\omega}{1 - \alpha - \beta}$
        \end{itemize}
    \end{exampleblock}

    \pause
    \begin{block}{Demonstrație}
        Luând așteptarea necondiționată a GARCH(1,1):
        \begin{align*}
            \E[\sigma_t^2] &= \omega + \alpha \E[\varepsilon_{t-1}^2] + \beta \E[\sigma_{t-1}^2] \\
            \bar{\sigma}^2 &= \omega + \alpha \bar{\sigma}^2 + \beta \bar{\sigma}^2 \\
            \bar{\sigma}^2 (1 - \alpha - \beta) &= \omega \\
            \bar{\sigma}^2 &= \frac{\omega}{1 - \alpha - \beta}
        \end{align*}
    \end{block}
    \end{cminipage}
    \quantlet{TSA\_ch5\_garch\_sim}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_garch_sim}

        \textcolor{MainBlue}{\textbf{(B)}} Nu necesită restricții de non-negativitate\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} Are mai puțini parametri\\[3pt]

    \end{block}

\begin{frame}{Răspuns Întrebarea 7}

\begin{frame}{Răspuns Întrebarea 8}

            \item \textbf{Testul Engle (ARCH-LM)}
        \end{itemize}
    \end{exampleblock}

    \pause
    \begin{block}{Procedura Testului ARCH-LM}
        \begin{enumerate}\setlength{\itemsep}{0pt}
            \item Estimează modelul pentru medie, obține reziduurile $\hat{\varepsilon}_t$
            \item Calculează $\hat{\varepsilon}_t^2$
            \item Regresează: $\hat{\varepsilon}_t^2 = \beta_0 + \beta_1 \hat{\varepsilon}_{t-1}^2 + \cdots + \beta_q \hat{\varepsilon}_{t-q}^2 + u_t$
            \item Statistica: $LM = T \cdot R^2 \sim \chi^2(q)$ sub $H_0$
        \end{enumerate}

        $H_0$: Nu există efecte ARCH $\quad$ $H_1$: Există efecte ARCH
    \end{block}
    \end{cminipage}

\begin{frame}{Întrebarea 13}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Modelul APARCH generalizează familia GARCH prin parametrul $\delta$. Ce obținem pentru $\delta = 2$, $\gamma_1 = 0$?}
        \textcolor{MainBlue}{\textbf{(A)}} EGARCH(1,1)\\[3pt]
        \textcolor{MainBlue}{\textbf{(B)}} TGARCH(1,1)\\[3pt]
        \textcolor{MainBlue}{\textbf{(C)}} GARCH(1,1) standard\\[3pt]
        \textcolor{MainBlue}{\textbf{(D)}} IGARCH(1,1)
    \end{block}

    \vspace{0.3cm}

    \begin{center}
        \textit{APARCH: $\sigma_t^\delta = \omega + \alpha_1(|\varepsilon_{t-1}| - \gamma_1 \varepsilon_{t-1})^\delta + \beta_1 \sigma_{t-1}^\delta$}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Răspuns Întrebarea 13}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspuns Corect: (C)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{GARCH(1,1) standard}
        \end{itemize}
    \end{exampleblock}

    \pause
    \begin{block}{Cazuri Particulare ale APARCH(1,1)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\delta = 2$, $\gamma_1 = 0$: \textbf{GARCH(1,1)} standard
            \item $\delta = 2$, $\gamma_1 \neq 0$: \textbf{GJR-GARCH}
            \item $\delta = 1$, $\gamma_1 \neq 0$: \textbf{TGARCH} (Zakoian, 1994)
            \item $\delta = 1$, $\gamma_1 = 0$: \textbf{AVGARCH} (Taylor, 1986)
            \item Studii empirice: $\hat{\delta} \approx 1.0$--$1.5$ pentru acțiuni (Ding et al., 1993)
        \end{itemize}
    \end{block}
    \end{cminipage}

\section{Întrebări Adevărat/Fals}

\begin{frame}{Adevărat sau Fals?}
    \begin{cminipage}{0.95\textwidth}
    \begin{enumerate}\setlength{\itemsep}{0pt}
        \item Modelele ARIMA pot captura volatility clustering.

        \vspace{0.3cm}

        \item În GARCH(1,1), dacă $\alpha + \beta = 1$, modelul se numește IGARCH.

        \vspace{0.3cm}

        \item GJR-GARCH folosește o variabilă indicator pentru șocuri negative.

        \vspace{0.3cm}

        \item Prognoza volatilității GARCH converge către zero pe termen lung.

        \vspace{0.3cm}

        \item EGARCH poate avea parametri negativi fără a genera varianță negativă.

        \vspace{0.3cm}

        \item Value at Risk (VaR) poate fi calculat folosind prognoza volatilității GARCH.
    \end{enumerate}
    \end{cminipage}
\end{frame}

\begin{frame}{Răspunsuri Adevărat/Fals}
    \begin{cminipage}{0.95\textwidth}
    \begin{enumerate}\setlength{\itemsep}{0pt}
        \item \textbf{FALS} --- ARIMA presupune varianță constantă; GARCH modelează volatilitatea.

        \vspace{0.2cm}

        \item \textbf{ADEVĂRAT} --- IGARCH = Integrated GARCH, volatilitatea are rădăcină unitară.

        \vspace{0.2cm}

        \item \textbf{ADEVĂRAT} --- $I_{t-1} = 1$ dacă $\varepsilon_{t-1} < 0$, altfel 0.

        \vspace{0.2cm}

        \item \textbf{FALS} --- Converge către varianța necondiționată $\bar{\sigma}^2$, nu zero.

        \vspace{0.2cm}

        \item \textbf{ADEVĂRAT} --- Modelează $\ln(\sigma_t^2)$, deci $\sigma_t^2 = e^{\ln(\sigma_t^2)} > 0$ mereu.

        \vspace{0.2cm}

        \item \textbf{ADEVĂRAT} --- $\text{VaR}_\alpha = z_\alpha \cdot \sigma_{t+1}$ (pentru medie zero).

        \vspace{0.2cm}

        \item EWMA are proprietatea de mean-reversion (revenire la medie) ca și GARCH(1,1).

        \vspace{0.2cm}

        \item Modelele Markov-switching GARCH pot explica persistența aparent infinită (IGARCH spurios).
    \end{enumerate}
    \end{cminipage}
\end{frame}

\begin{frame}{Răspunsuri Adevărat/Fals (continuare)}
    \begin{cminipage}{0.95\textwidth}
    \begin{enumerate}\setlength{\itemsep}{0pt}
    \setcounter{enumi}{6}
        \item \textbf{FALS} --- EWMA = IGARCH cu $\omega = 0$: are $\alpha + \beta = 1$, deci \textbf{nu} revine la o varianță necondiționată. Volatilitatea EWMA rămâne ridicată mai mult timp după crize.

        \vspace{0.2cm}

        \item \textbf{ADEVĂRAT} --- Când regimuri diferite (calm/criză) sunt amestecate, un model GARCH unic estimează $\alpha + \beta \approx 1$. Modelul Markov-switching identifică regimurile și poate arăta că în fiecare regim persistența este mai mică.
    \end{enumerate}
    \end{cminipage}
\end{frame}

\section{Probleme Practice}

\begin{frame}{Problema 1: Calculul Varianței Necondiționate}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Enunț}
        Un model GARCH(1,1) are parametrii estimați:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\omega = 0.000002$
            \item $\alpha = 0.08$
            \item $\beta = 0.90$
        \end{itemize}

        \vspace{0.3cm}
        Calculați:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{(a)} Varianța necondiționată zilnică
            \item \textbf{(b)} Volatilitatea necondiționată zilnică (ca procent)
            \item \textbf{(c)} Volatilitatea anualizată (presupunând 252 zile de tranzacționare)
            \item \textbf{(d)} Half-life-ul volatilității
        \end{itemize}
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Soluție Problema 1}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspunsuri}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{(a)} $\bar{\sigma}^2 = \dfrac{\omega}{1 - \alpha - \beta} = \dfrac{0.000002}{1 - 0.08 - 0.90} = \dfrac{0.000002}{0.02} = 0.0001$

            \vspace{0.2cm}

            \item \textbf{(b)} $\bar{\sigma} = \sqrt{0.0001} = 0.01 = 1\%$ pe zi

            \vspace{0.2cm}

            \item \textbf{(c)} $\sigma_{\text{annual}} = \bar{\sigma} \times \sqrt{252} = 0.01 \times 15.87 = 15.87\%$ pe an

            \vspace{0.2cm}

            \item \textbf{(d)} $HL = \dfrac{\ln(0.5)}{\ln(\alpha + \beta)} = \dfrac{\ln(0.5)}{\ln(0.98)} = \dfrac{-0.693}{-0.0202} \approx 34$ zile
        \end{itemize}
    \end{exampleblock}

    \begin{block}{Interpretare}
        Volatilitatea de 15.87\% pe an este tipică pentru un indice bursier. Half-life de 34 zile înseamnă că un șoc de volatilitate se reduce la jumătate după aproximativ 7 săptămâni.
    \end{block}
    \end{cminipage}
    \quantlet{TSA\_ch5\_convergence}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_convergence}
\end{frame}

\begin{frame}{Problema 2: Prognoză Volatilitate}

            \item $\omega = 0.000002$, $\alpha = 0.08$, $\beta = 0.90$
            \item La momentul $T$: $\varepsilon_T = -0.03$ (scădere de 3\%), $\sigma_T^2 = 0.0004$
        \end{itemize}

        \vspace{0.3cm}
        Calculați prognoza volatilității pentru:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{(a)} $\sigma_{T+1}^2$ (un pas înainte)
            \item \textbf{(b)} $\sigma_{T+5}^2$ (cinci pași înainte)
            \item \textbf{(c)} $\sigma_{T+100}^2$ (o sută de pași înainte)

    \end{block}

\begin{frame}{Soluție Problema 2}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspunsuri}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{(a)} $\sigma_{T+1}^2 = \omega + \alpha \varepsilon_T^2 + \beta \sigma_T^2$
            \[
                = 0.000002 + 0.08 \times (0.03)^2 + 0.90 \times 0.0004 = 0.000434
            \]
            Volatilitate: $\sqrt{0.000434} = 2.08\%$

            \vspace{0.2cm}

            \item \textbf{(b)} $\E_T[\sigma_{T+5}^2] = \bar{\sigma}^2 + (0.98)^4 (\sigma_{T+1}^2 - \bar{\sigma}^2)$
            \[
                = 0.0001 + 0.922 \times (0.000434 - 0.0001) = 0.000408
            \]
            Volatilitate: $\sqrt{0.000408} = 2.02\%$

            \vspace{0.2cm}

            \item \textbf{(c)} $\E_T[\sigma_{T+100}^2] = 0.0001 + (0.98)^{99} \times 0.000334 \approx 0.000145$

            Volatilitate: $\sqrt{0.000145} = 1.20\%$ (aproape de $\bar{\sigma} = 1\%$)
        \end{itemize}
    \end{exampleblock}

\begin{frame}{Problema 5: EWMA vs GARCH --- Persistența și Prognoza}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Enunț}
        Considerați două modele pentru volatilitatea zilnică a EUR/USD:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{EWMA} cu $\lambda = 0.94$: \; $\sigma_t^2 = 0.94\, \sigma_{t-1}^2 + 0.06\, r_{t-1}^2$
            \item \textbf{GARCH(1,1)}: \; $\sigma_t^2 = 0.000001 + 0.05\, \varepsilon_{t-1}^2 + 0.93\, \sigma_{t-1}^2$
        \end{itemize}

        \vspace{0.3cm}
        La momentul $T$: $r_T = -0.025$ (scădere de 2.5\%), $\sigma_T^2 = 0.0002$.

        \vspace{0.3cm}
        Calculați:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{(a)} Prognoza $\sigma_{T+1}^2$ pentru fiecare model
            \item \textbf{(b)} Varianța necondiționată $\bar{\sigma}^2$ pentru GARCH(1,1). De ce nu există pentru EWMA?
            \item \textbf{(c)} Prognoza pe termen lung ($h \to \infty$) pentru fiecare model
        \end{itemize}
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Soluție Problema 5}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Răspunsuri}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{(a)} Prognoza $\sigma_{T+1}^2$:
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{EWMA}: $0.94 \times 0.0002 + 0.06 \times (0.025)^2 = 0.000188 + 0.0000375 = 0.000226$
                \item \textbf{GARCH}: $0.000001 + 0.05 \times (0.025)^2 + 0.93 \times 0.0002 = 0.000218$
            \end{itemize}

            \vspace{0.2cm}

            \item \textbf{(b)} $\bar{\sigma}^2_{\text{GARCH}} = \dfrac{0.000001}{1 - 0.05 - 0.93} = \dfrac{0.000001}{0.02} = 0.00005$

            EWMA nu are varianță necondiționată: $\alpha + \beta = 1$, $\omega = 0$ $\Rightarrow$ formula $\frac{\omega}{1-\alpha-\beta}$ este nedefinită.

            \vspace{0.2cm}

            \item \textbf{(c)} Pe termen lung:
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{GARCH}: $\sigma_{T+h}^2 \to \bar{\sigma}^2 = 0.00005$ (mean-reversion)
                \item \textbf{EWMA}: rămâne la ultimul nivel --- nu revine la un echilibru
            \end{itemize}
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}

\begin{frame}{Soluție Problema 6}

            \vspace{0.15cm}

            \item \textbf{(b)} $\hat{\delta} = 1.3$: puterea optimă este între $\sigma_t$ ($\delta=1$) și $\sigma_t^2$ ($\delta=2$). Piața nu se comportă exact ca GARCH ($\delta=2$) sau TGARCH ($\delta=1$) --- datele preferă o transformare intermediară.

            \vspace{0.15cm}

            \item \textbf{(c)} $\hat{d} = 0.42$: memorie lungă moderată. ACF a $r_t^2$ decade hiperbolic, nu exponențial. Persistența nu este infinită (ca IGARCH), ci lentă --- șocurile vechi contează, dar tot mai puțin.

            \vspace{0.15cm}

            \item \textbf{(d)} \textbf{Recomandare}: APARCH sau GJR-GARCH. APARCH are AIC-ul cel mai bun și captează asimetria + puterea optimă. Dacă simplitatea contează, GJR-GARCH este o alternativă bună (AIC apropiat, mai ușor de interpretat).

VaR_99 = 2.326 * vol_forecast[0] / 100 * portfolio_value
print(f'VaR 99% (1 zi): {VaR_99:,.0f} EUR')

# VaR 10 zile
VaR_99_10d = VaR_99 * np.sqrt(10)
print(f'VaR 99% (10 zile): {VaR_99_10d:,.0f} EUR')

            {\small
            \item Descărcați inflația trimestrială din România (IPC, INS, 1997--2024)
            \item Estimați un model AR(4) pentru inflație: $\pi_t = c + \phi_1 \pi_{t-1} + \cdots + \phi_4 \pi_{t-4} + \varepsilon_t$
            \item Calculați reziduurile $\hat{\varepsilon}_t$ și testați efecte ARCH: regresați $\hat{\varepsilon}_t^2$ pe $\hat{\varepsilon}_{t-1}^2, \ldots, \hat{\varepsilon}_{t-4}^2$
            \item Raportați statistica $LM = T \cdot R^2$ și comparați cu $\chi^2(4)$
            \item Estimați un model ARCH(4) și comparați cu rezultatele originale ale lui Engle
            \item \textbf{Întrebare}: Inflația românească prezintă efecte ARCH mai puternice decât cea britanică? De ce?
            }
        \end{enumerate}
        {\tiny Ref: Engle, R.F.\ (1982). Autoregressive Conditional Heteroscedasticity with Estimates of the Variance of United Kingdom Inflation, \textit{Econometrica}, 50(4), 987--1007.}
