\section{Introducere în Modelarea Volatilității}

\begin{frame}{Volatility clustering}
    \small
    \vspace{-0.2cm}
    \begin{itemize}
        \item Perioadele de volatilitate mare sunt urmate de perioade de volatilitate mare, iar perioadele de volatilitate redusă sunt urmate de perioade de volatilitate redusă
        \item \textbf{Takeaway:} Volatility clustering este vizibil \textbf{chiar înainte de modelare} --- este primul semn că varianța nu este constantă
    \end{itemize}
    \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.50\textheight, keepaspectratio]{garch_volatility_clustering.pdf}
    \end{center}
    \quantlet{TSA\_ch5\_clustering}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_clustering}
\end{frame}

\begin{frame}{De ce modelăm volatilitatea?}
    \small
        \begin{block}{Observații empirice în seriile financiare}
            \begin{itemize}
                \item Randamentele prezintă \textbf{volatility clustering} --- perioadele de volatilitate ridicată sunt urmate de perioade similare
                \item Distribuția randamentelor are \textbf{cozi groase} (leptokurtosis)
                \item Corelația randamentelor $\approx 0$, dar corelația pătratelor este semnificativă
                \item Volatilitatea răspunde \textbf{asimetric} la șocuri (leverage effect)
            \end{itemize}
        \end{block}
        \vspace{0.2cm}
        \begin{alertblock}{Ideea centrală a capitolului}
            \begin{itemize}
                \item \textbf{Chiar dacă randamentele sunt greu de prognozat, volatilitatea lor este adesea prognozabilă.}
                \item Modelele ARIMA presupun \textbf{varianță constantă} (homoscedasticitate) --- ipoteză nerealistă pentru seriile financiare
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{De ce ne pasă de volatilitate condiționată?}
    \small
        \begin{block}{Aplicații directe în finanțe}
            \begin{itemize}
                \item \textbf{Value at Risk (VaR) și Expected Shortfall (ES)}: cerințe de capital Basel III/IV
                \item \textbf{Prețul opțiunilor}: volatilitatea implicită vs realizată
                \item \textbf{Hedging dinamic}: rapoarte de acoperire optime $h_t^{*} = \sigma_{12,t}/\sigma_{22,t}$
                \item \textbf{Margining}: casele de compensare recalculează zilnic marjele
                \item \textbf{Capital economic / reglementar}: FRTB (Fundamental Review of the Trading Book), stress testing
                \item \textbf{Alocarea portofoliului}: ponderi Markowitz cu $\mathbf{H}_t$ variabilă în timp
            \end{itemize}
        \end{block}
        \vspace{0.2cm}
        \begin{alertblock}{Concluzie}
            \begin{itemize}
                \item Volatilitatea condiționată nu este un exercițiu academic --- este \textbf{ingredientul central} al managementului cantitativ al riscului
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Fapte stilizate ale randamentelor financiare}
    \small
    \vspace{-0.2cm}
    \begin{block}{Proprietăți observate}
        \begin{itemize}
            \item \textbf{Absența autocorelației} în randamente
                \begin{itemize}
                    \item $\text{Corr}(r_t, r_{t-k}) \approx 0$ --- randamentele sunt imprevizibile
                \end{itemize}
            \item \textbf{Autocorelație} în $r_t^2$, $|r_t|$
                \begin{itemize}
                    \item $\text{Corr}(r_t^2, r_{t-k}^2) > 0$ --- volatilitatea este predictibilă
                \end{itemize}
            \item \textbf{Cozi groase} (kurtosis $> 3$)
                \begin{itemize}
                    \item Mai multe valori extreme decât prevede distribuția normală
                \end{itemize}
            \item \textbf{Leverage effect}
                \begin{itemize}
                    \item Șocurile negative cresc volatilitatea mai mult decât cele pozitive
                \end{itemize}
            \item \textbf{Volatility clustering}
                \begin{itemize}
                    \item Perioadele de volatilitate mare sunt urmate de volatilitate mare
                \end{itemize}
        \end{itemize}
    \end{block}

\begin{frame}{Heteroscedasticitate condiționată}
	\small
	\begin{defn}[Varianță Condiționată]
		Pentru o serie de randamente $\{r_t\}$, varianța condiționată la momentul $t$:
		\[
		\sigma_t^2 = \Var(r_t \mid \mathcal{F}_{t-1}) = \E\!\bigl[(r_t - \mu_t)^2 \mid \mathcal{F}_{t-1}\bigr]
		\]
		unde $\mathcal{F}_{t-1}$ = informația disponibilă până la $t{-}1$.
	\end{defn}
	
	\begin{block}{Modelul general}
		\[
		r_t = \mu_t + \varepsilon_t, \qquad
		\varepsilon_t = \sigma_t z_t, \qquad
		z_t \sim \text{i.i.d.}(0, 1)
		\]
		\begin{itemize}
			\item $\mu_t$ = media condiționată (ARMA); \;
			$\sigma_t^2$ = varianța condiționată (GARCH)
			\item $z_t$ = inovații standardizate (Normal, Student-$t$, GED)
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}{Volatilitate condiționată vs necondiționată}
    \small
    \vspace{-0.2cm}
    \begin{block}{Interpretare}
        \begin{itemize}
            \item \textbf{Volatilitate condiționată} $\sigma_t$ (GARCH): variabilă în timp, reacționează la informații noi
            \item \textbf{Volatilitate necondiționată} $\bar{\sigma} = \sqrt{\omega/(1-\alpha-\beta)}$: nivel constant pe termen lung
            \item Zone roșii: $\sigma_t > \bar{\sigma}$ (perioade de stres); zone albastre: $\sigma_t < \bar{\sigma}$ (perioade calme)
        \end{itemize}
    \end{block}
        \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.50\textheight, keepaspectratio]{garch_cond_vs_uncond.pdf}
    \end{center}
    \quantlet{TSA\_ch5\_cond\_vs\_uncond}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_cond_vs_uncond}
\end{frame}

\begin{frame}{Modelul ARCH(q) --- Engle (1982)}
    \small
        \begin{defn}[ARCH(q)]
            Modelul \textbf{Autoregressive Conditional Heteroskedasticity} de ordin $q$:
            \[
                \varepsilon_t = \sigma_t z_t, \quad z_t \sim \text{i.i.d.}(0, 1), \quad
                \sigma_t^2 = \omega + \sum_{i=1}^{q} \alpha_i \varepsilon_{t-i}^2
            \]
        \end{defn}

        \vspace{0.2cm}

        \begin{block}{Restricții pentru staționaritate}
            \begin{itemize}
                \item $\omega > 0$ (nivel de bază pozitiv), \quad $\alpha_i \geq 0$ (non-negativitate)
                \item $\sum_{i=1}^{q} \alpha_i < 1$ (staționaritate)
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{Simulare ARCH(1): Efectul parametrului $\alpha$}
    \small
    \vspace{-0.2cm}
    \begin{block}{Interpretare}
        \begin{itemize}
            \item Cu cât $\alpha$ este mai mare, cu atât volatilitatea reacționează mai puternic la șocuri recente
        \end{itemize}
    \end{block}
        \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.58\textheight, keepaspectratio]{garch_arch1_simulation.pdf}
    \end{center}
    \quantlet{TSA\_ch5\_arch\_sim}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_arch_sim}
\end{frame}

\begin{frame}{Simulare ARCH(1): ce observăm în serie?}
    \small
        \begin{block}{Citirea graficului de simulare}
            \begin{itemize}
                \item \textbf{$\alpha$ mic} (e.g., 0.1): seria arată aproape ca zgomot alb --- clustering ușor, spike-uri rare
                \item \textbf{$\alpha$ moderat} (e.g., 0.5): spike-uri vizibile, dar se sting rapid (1--2 perioade) --- ``memorie scurtă''
                \item \textbf{$\alpha$ mare} (e.g., 0.9): volatilitate explozivă --- spike-uri foarte mari, dar tot fără persistență reală
            \end{itemize}
        \end{block}
        \begin{alertblock}{Diferența cheie: ARCH vs GARCH}
            \begin{itemize}
                \item ARCH reacționează doar la \textbf{ultimul șoc} ($\varepsilon_{t-1}^2$) --- volatilitatea crește brusc dar scade la fel de brusc
                \item Nu captează perioadele lungi de volatilitate ridicată observate pe piețe
                \item GARCH adaugă $\beta\sigma_{t-1}^2$ = ``memorie''
            \end{itemize}
        \end{alertblock}
        \begin{exampleblock}{Exemplu vizual}
            \begin{itemize}
                \item Comparați: în simularea ARCH(1), benzile $\pm 2\sigma$ au spike-uri înguste (1 zi)
                \item Pe S\&P 500 real, episoadele de volatilitate durează săptămâni $\Rightarrow$ ARCH este insuficient
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Proprietăți ale modelului ARCH(1)}
    \small
        \begin{prop}[Proprietăți ARCH(1): $\sigma_t^2 = \omega + \alpha_1 \varepsilon_{t-1}^2$]
            \begin{itemize}
                \item \textbf{Varianța necondiționată}: $\E[\varepsilon_t^2] = \frac{\omega}{1 - \alpha_1}$ (dacă $\alpha_1 < 1$)
                \item \textbf{Kurtosis}: $\kappa = 3 \cdot \frac{1 - \alpha_1^2}{1 - 3\alpha_1^2}$ (dacă $\alpha_1^2 < 1/3$; $z_t \sim N(0,1)$)
                \item Kurtosis $> 3$ pentru $\alpha_1 > 0$ $\Rightarrow$ \textbf{cozi groase}!
            \end{itemize}
        \end{prop}
        \begin{rmk}[Dependența de distribuția inovațiilor]
            Formula presupune $z_t \sim N(0,1)$; sub Student-$t$, kurtosisul necondiționat se modifică.
        \end{rmk}
        \vspace{0.1cm}
        \begin{exampleblock}{Exemplu numeric}
            \begin{itemize}
                \item Dacă $\omega = 0.0001$ și $\alpha_1 = 0.3$:
                    \begin{itemize}
                        \item Varianța necondiționată: $\sigma^2 = \frac{0.0001}{1 - 0.3} = 0.000143$
                        \item Kurtosis: $\kappa = 3 \cdot \frac{1 - 0.09}{1 - 0.27} = 3.74 > 3$
                    \end{itemize}
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Demonstrație: varianța necondiționată ARCH(1)}
    \small
        \begin{proof}[Demonstrație]
            Fie $\varepsilon_t = \sigma_t z_t$ cu $z_t \sim N(0,1)$ și $\sigma_t^2 = \omega + \alpha_1 \varepsilon_{t-1}^2$.

            \vspace{0.1cm}
            \textbf{Pasul 1}: Aplicăm media necondiționată:
            \vspace{-2mm}
            \begin{multline*}
                \E[\varepsilon_t^2] = \E[\sigma_t^2 z_t^2] = \E[\sigma_t^2] \cdot \E[z_t^2] = \E[\sigma_t^2] \\
                \text{(independență: $\sigma_t^2$ este $\mathcal{F}_{t-1}$-măsurabil, $z_t \perp \mathcal{F}_{t-1}$)}
            \end{multline*}
            \vspace{-1mm}

            \textbf{Pasul 2}: Aplicăm media ecuației varianței:
            \vspace{-2mm}
            \[
                \E[\sigma_t^2] = \E[\omega + \alpha_1 \varepsilon_{t-1}^2] = \omega + \alpha_1 \E[\varepsilon_{t-1}^2]
            \]
            \vspace{-1mm}

            \textbf{Pasul 3}: Prin staționaritate, $\E[\varepsilon_t^2] = \E[\varepsilon_{t-1}^2] = \sigma^2$:
            \vspace{-2mm}
            \[
                \sigma^2 = \omega + \alpha_1 \sigma^2 \quad \Rightarrow \quad \sigma^2 (1 - \alpha_1) = \omega
            \]
            \vspace{-1mm}

            \textbf{Rezultat}: $\boxed{\sigma^2 = \dfrac{\omega}{1 - \alpha_1}}$ \quad (necesită $\alpha_1 < 1$ pentru staționaritate)
        \end{proof}
\end{frame}

\begin{frame}{Demonstrație: kurtosis ARCH(1)}
    \small
        Pentru $\varepsilon_t = \sigma_t z_t$ cu $z_t \sim N(0,1)$:

        \textbf{Pasul 1}: $\E[\varepsilon_t^4] = \E[\sigma_t^4] \cdot \E[z_t^4] = 3 \E[\sigma_t^4]$ \quad (deoarece $\E[z^4] = 3$)

        \textbf{Pasul 2}: Folosind $\sigma_t^2 = \omega + \alpha_1 \varepsilon_{t-1}^2$:
        \[
            \E[\sigma_t^4] = \E[(\omega + \alpha_1 \varepsilon_{t-1}^2)^2] = \omega^2 + 2\omega\alpha_1 \sigma^2 + \alpha_1^2 \E[\varepsilon_{t-1}^4]
        \]

        \textbf{Pasul 3}: Rezolvând recurența:
        \[
            \boxed{\kappa = \frac{\E[\varepsilon_t^4]}{(\E[\varepsilon_t^2])^2} = 3 \cdot \frac{1 - \alpha_1^2}{1 - 3\alpha_1^2}}
        \]

        \begin{block}{Interpretare}
            \begin{itemize}
                \item $\kappa > 3$ pentru orice $\alpha_1 > 0$ $\Rightarrow$ \textbf{cozi groase} (leptokurtosis)
                \item Necesită $\alpha_1 < 1/\sqrt{3} \approx 0.577$ pentru moment de ordin 4 finit
                \item ARCH generează în mod natural distribuții cu cozi groase.
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{Demonstrație: kurtosis GARCH(1,1)}
    \small
        Pentru $\varepsilon_t = \sigma_t z_t$ cu $z_t \sim N(0,1)$ și $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$:

        \textbf{Pasul 1}: $\E[\varepsilon_t^4] = \E[\sigma_t^4 z_t^4] = 3\E[\sigma_t^4]$ (deoarece $\E[z^4] = 3$)

        \textbf{Pasul 2}: Folosind $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$, ridicăm la pătrat:
        \vspace{-2mm}
        \[
            \E[\sigma_t^4] = \E[(\omega + \alpha\varepsilon_{t-1}^2 + \beta\sigma_{t-1}^2)^2]
        \]
        \vspace{-2mm}

        Expandăm și folosim staționaritatea ($\E[\sigma_t^4] = \E[\sigma_{t-1}^4]$, $\E[\varepsilon_t^4] = 3\E[\sigma_t^4]$):
        \vspace{-2mm}
        \[
            \E[\sigma_t^4] = \omega^2 + 2\omega(\alpha+\beta)\bar{\sigma}^2 + (3\alpha^2 + 2\alpha\beta + \beta^2)\E[\sigma_{t-1}^4]
        \]
        \vspace{-2mm}

        \textbf{Pasul 3}: Rezolvând recurența:
        \vspace{-2mm}
        \[
            \boxed{\kappa_{\varepsilon} = \frac{\E[\varepsilon_t^4]}{(\E[\varepsilon_t^2])^2} = 3 \cdot \frac{1 - (\alpha+\beta)^2}{1 - (\alpha+\beta)^2 - 2\alpha^2}}
        \]
        \vspace{-1mm}
        \begin{block}{Interpretare}
            \begin{itemize}
                \item Necesită $1 - (\alpha+\beta)^2 - 2\alpha^2 > 0$ pentru moment de ordin 4 finit
                \item GARCH generează \textbf{mai mult kurtosis} decât ARCH deoarece $\beta > 0$ amplifică persistența
                \item Cu $\alpha = 0.088$, $\beta = 0.900$: $\kappa \approx 5.6$ (vs.\ ARCH(1): $\kappa \approx 3.05$)
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{Teorie vs.\ practică: decalajul de kurtosis}
    \small
        \vspace{-0.3cm}
        \begin{block}{Kurtosis-ul modelului ARCH(1) este prea mic}
            \begin{itemize}
                \item Pentru randamente tipice de acțiuni: $\alpha_1 \approx 0.09$ (S\&P 500)
                \item Kurtosis-ul teoretic: $\kappa = 3 \cdot \frac{1 - 0.09^2}{1 - 3 \times 0.09^2} \approx \textbf{3.05}$; \quad kurtosis-ul empiric al S\&P 500: $\kappa \approx \textbf{13.8}$
            \end{itemize}
        \end{block}
        \vspace{-0.15cm}
        \begin{alertblock}{Decalajul de kurtosis}
            \vspace{-2mm}
            \begin{center}
            {\large $\underbrace{3.05}_{\text{ARCH(1)-Normal}} \;\ll\; \underbrace{13.8}_{\text{Empiric}}$}
            \end{center}
            \vspace{-1mm}
            \begin{itemize}
                \item ARCH(1) cu inovații gaussiene nu poate reproduce kurtosis-ul extrem observat în datele financiare
            \end{itemize}
        \end{alertblock}
        \vspace{-0.15cm}
        \begin{exampleblock}{Soluții}
            \begin{itemize}
                \item \textbf{Inovații Student-$t$}: $z_t \sim t_\nu$ adaugă kurtosis din distribuția însăși
                \item \textbf{GARCH(1,1)}: persistența ($\beta > 0$) generează kurtosis suplimentar
                \item Ambele combinate: Student-$t$ GARCH capturează $\kappa \gg 3$ observat în practică
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Distribuția Student-$t$}
    \small

                \item \textbf{Avantaj}: cozi groase controlate de $\nu$ --- captează valorile extreme din seriile financiare
            \end{itemize}

\begin{frame}{Distribuția GED (Generalized Error Distribution)}
    \small
    \vspace{-0.2cm}
        \begin{block}{Densitatea de probabilitate (PDF)}
            \begin{itemize}
                \item[] \vspace{-3mm}
                    $f(x \mid \nu) = \frac{\nu \exp\!\left(-\frac{1}{2}\left|\frac{x}{\lambda}\right|^\nu\right)}{\lambda \cdot 2^{1+1/\nu} \cdot \Gamma(1/\nu)}$, \; $\lambda = \sqrt{2^{-2/\nu} \cdot \frac{\Gamma(1/\nu)}{\Gamma(3/\nu)}}$
                    \vspace{-1mm}
                \item Parametru: $\nu > 0$ (forma); \quad $\nu = 2$: Normal; \quad $\nu = 1$: Laplace; \quad $\nu \to \infty$: Uniform
                \item Kurtosis: $\kappa = \frac{\Gamma(1/\nu)\,\Gamma(5/\nu)}{\Gamma(3/\nu)^2}$; \quad $\nu < 2$: cozi groase; \quad $\nu > 2$: cozi subțiri
                \item \textbf{Avantaj}: flexibilitate maximă --- un singur parametru controlează forma întregii distribuții
            \end{itemize}

\begin{frame}{Testarea efectelor ARCH}
    \small
        \begin{block}{Testul Engle pentru efecte ARCH}
            \textbf{Procedură}:
            \begin{enumerate}
                \item Estimează modelul pentru medie și obține reziduurile $\hat{\varepsilon}_t$
                \item Calculează $\hat{\varepsilon}_t^2$
                \item Regresează $\hat{\varepsilon}_t^2$ pe lag-urile sale:
                \[
                    \hat{\varepsilon}_t^2 = \beta_0 + \beta_1 \hat{\varepsilon}_{t-1}^2 + \cdots + \beta_q \hat{\varepsilon}_{t-q}^2 + u_t
                \]
                \item Calculează statistica $LM = T \cdot R^2 \sim \chi^2(q)$
            \end{enumerate}
        \end{block}

        \vspace{0.3cm}

        \begin{alertblock}{Ipoteze}
            \begin{itemize}
                \item $H_0$: Nu există efecte ARCH ($\beta_1 = \cdots = \beta_q = 0$)
                \item $H_1$: Există efecte ARCH (cel puțin un $\beta_i \neq 0$)
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Corelograma pătratelor reziduurilor}
    \small
    \begin{block}{Diagnostic vizual pentru efecte ARCH}
        \begin{itemize}
            \item Dacă $\hat{\varepsilon}_t$ sunt i.i.d., atunci $\hat{\varepsilon}_t^2$ nu prezintă autocorelație
            \item ACF semnificativă a lui $\hat{\varepsilon}_t^2$ la mai multe lag-uri $\Rightarrow$ \textbf{heteroscedasticitate condiționată}
            \item Completează testul Engle LM cu o vizualizare intuitivă
        \end{itemize}
    \end{block}
    \begin{center}
        \includegraphics[width=0.85\textwidth, height=0.45\textheight, keepaspectratio]{ch5_garch_acf_squared.pdf}
    \end{center}
\end{frame}

\begin{frame}{Limitări ale modelului ARCH}
    \small
        \begin{alertblock}{Probleme practice}
            \begin{enumerate}
                \item \textbf{Ordin mare} --- de obicei sunt necesare multe lag-uri ($q$ mare)
                \item \textbf{Mulți parametri} --- dificultăți de estimare
                \item \textbf{Restricții de non-negativitate} --- greu de impus pentru $q$ mare
                \item \textbf{Nu capturează persistența} --- volatilitatea observată este foarte persistentă
            \end{enumerate}
        \end{alertblock}

        \vspace{0.5cm}

        \begin{block}{Soluția}
            \begin{itemize}
                \item \textbf{Modelul GARCH} --- introduce lag-uri ale varianței condiționate pentru a captura persistența cu mai puțini parametri.
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{Modelul GARCH(1,1)}
    \small
    \vspace{-0.2cm}
        \begin{alertblock}{De ce GARCH? Limitările ARCH}
            \begin{itemize}
                \item ARCH($q$) necesită multe lag-uri pentru a captura persistența volatilității $\Rightarrow$ mulți parametri, restricții dificile
                \item \textbf{GARCH} rezolvă problema adăugând un singur termen: $\sigma_{t-1}^2$
            \end{itemize}
        \end{alertblock}
        \vspace{-0.1cm}
        \begin{block}{Cel mai popular model de volatilitate}
            \begin{itemize}
                \item $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$
            \end{itemize}
        \end{block}
        \vspace{-0.1cm}
        \begin{prop}[Restricții și proprietăți GARCH(1,1)]
            \begin{itemize}
                \item $\omega > 0$, $\alpha \geq 0$, $\beta \geq 0$; \quad $\alpha + \beta < 1$ (staționaritate)
                \item $\bar{\sigma}^2 = \frac{\omega}{1 - \alpha - \beta}$; \quad Half-life: $HL = \frac{\ln(0.5)}{\ln(\alpha + \beta)}$
            \end{itemize}
        \end{prop}
\end{frame}

\end{frame}

\begin{frame}{Modelul GARCH(p,q) --- Bollerslev (1986)}
    \small
        \begin{defn}[GARCH(p,q)]
            Modelul \textbf{Generalized ARCH}:
            \[
                \varepsilon_t = \sigma_t z_t, \quad z_t \sim \text{i.i.d.}(0, 1)
            \]
            \[
                \sigma_t^2 = \omega + \sum_{i=1}^{q} \alpha_i \varepsilon_{t-i}^2 + \sum_{j=1}^{p} \beta_j \sigma_{t-j}^2
            \]
        \end{defn}

        \vspace{0.3cm}

        \begin{block}{Interpretare}
            \begin{itemize}
                \item $\omega$ = nivel de bază al volatilității
                \item $\alpha_i$ = reacția la șocuri recente (coeficienți de impact)
                \item $\beta_j$ = persistența volatilității (memorie)
                \item $\alpha + \beta$ = persistența totală
            \end{itemize}
        \end{block}
\end{frame}

%% Rec. 1 + 5: Cum citesc un model GARCH în 30 de secunde
\begin{frame}{Cum citesc un model GARCH în 30 de secunde}
    \small
        \vspace{-0.2cm}
        \begin{block}{Structura unui model GARCH(1,1)}
            \begin{itemize}
                \item \textbf{Ecuația mediei}: $r_t = \mu + \varepsilon_t$, \quad $\varepsilon_t = \sigma_t z_t$, \quad $z_t \sim D(0,1)$
                \item \textbf{Ecuația varianței}:
                    \vspace{-2mm}
                    \[ \sigma_t^2 = \underbrace{\omega}_{\text{bază}} + \underbrace{\alpha \varepsilon_{t-1}^2}_{\text{reacție la șoc}} + \underbrace{\beta \sigma_{t-1}^2}_{\text{persistență}} \]
                    \vspace{-2mm}
            \end{itemize}
        \end{block}
        \vspace{-1.5mm}
        \begin{exampleblock}{Interpretare economică a parametrilor}
            \begin{itemize}
                \item $\alpha$ \textbf{mare} $\Rightarrow$ piață \textbf{foarte sensibilă} la informație nouă (șocuri bruște)
                \item $\beta$ \textbf{mare} $\Rightarrow$ șocurile de volatilitate se \textbf{sting lent} (inerție)
                \item $\alpha + \beta$ aproape de 1 $\Rightarrow$ volatilitate \textbf{foarte persistentă} (aproape de IGARCH)
                \item $\omega / (1 - \alpha - \beta)$ = varianța necondiționată (nivelul de echilibru pe termen lung)
            \end{itemize}
        \end{exampleblock}
        \vspace{-1.5mm}
        \begin{alertblock}{Alegerea distribuției $z_t$}
            \begin{itemize}
                \item \textbf{Normal}: simplu, dar subestimează riscul de coadă
                \item \textbf{Student-t}: cozi mai groase $\Rightarrow$ VaR mai realist; \textbf{alegerea standard} în practică
            \end{itemize}
        \end{alertblock}
\end{frame}

        \begin{itemize}
            \item $\alpha$ controlează reacția la șocuri
            \item $\beta$ controlează persistența
            \item Suma $\alpha + \beta$ determină viteza de revenire la medie
        \end{itemize}
    \end{block}

\begin{frame}{Simulare GARCH(1,1): ce observăm în serie?}
    \small
        \begin{block}{Citirea graficului de simulare}
            \begin{itemize}
                \item \textbf{$\alpha + \beta$ mic} (e.g., 0.7): volatilitatea revine rapid la medie --- seria arată aproape ca un zgomot alb cu variații ușoare de amplitudine
                \item \textbf{$\alpha + \beta$ moderat} (e.g., 0.95): episoade de volatilitate ridicată durează săptămâni --- asemănător cu seriile financiare reale
                \item \textbf{$\alpha + \beta$ aproape de 1} (e.g., 0.99): volatilitatea persistă luni de zile --- half-life $\approx 69$ zile
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Identificare vizuală: half-life}
            \begin{itemize}
                \item Half-life: $HL = \ln(0.5) / \ln(\alpha + \beta)$
                \item Cu $\alpha + \beta = 0.95$: $HL \approx 14$ zile (2 săptămâni)
                \item Cu $\alpha + \beta = 0.988$: $HL \approx 58$ zile (3 luni) --- tipic S\&P 500
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{Regulă practică}
            \begin{itemize}
                \item Dacă benzile $\pm 2\sigma_t$ nu revin la normal în 1--2 luni, piața are persistență ridicată ($\alpha + \beta > 0.97$)
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Demonstrație: varianța necondiționată GARCH(1,1)}
    \small
    \begin{proof}[Demonstrație]
        Pentru $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$:

        \textbf{Pasul 1}: Aplicăm media necondiționată:
        $\E[\sigma_t^2] = \omega + \alpha \E[\varepsilon_{t-1}^2] + \beta \E[\sigma_{t-1}^2]$

        \textbf{Pasul 2}: Prin staționaritate, $\E[\sigma_t^2] = \E[\sigma_{t-1}^2] = \bar{\sigma}^2$ și $\E[\varepsilon_t^2] = \bar{\sigma}^2$:
        $\bar{\sigma}^2 = \omega + (\alpha + \beta) \bar{\sigma}^2$

        \textbf{Pasul 3}: Rezolvăm: $\bar{\sigma}^2 (1 - \alpha - \beta) = \omega \Rightarrow \boxed{\bar{\sigma}^2 = \frac{\omega}{1 - \alpha - \beta}}$
    \end{proof}
    \vspace{-0.1cm}
    \begin{alertblock}{Condiție de staționaritate}
        \begin{itemize}
            \item Necesită $\alpha + \beta < 1$ pentru varianță necondiționată finită
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{GARCH(1,1) ca ARMA pentru $\varepsilon_t^2$}
    \small
    \vspace{-0.2cm}
    \begin{prop}[Reprezentare ARMA(1,1) a GARCH(1,1)]
        \begin{itemize}
            \item Definim $\nu_t = \varepsilon_t^2 - \sigma_t^2$ (șocul varianței). Atunci:
            \[
                \varepsilon_t^2 = \omega + (\alpha + \beta) \varepsilon_{t-1}^2 + \nu_t - \beta \nu_{t-1}
            \]
            \item Aceasta este un \textbf{ARMA(1,1)} pentru $\varepsilon_t^2$.
        \end{itemize}
    \end{prop}

    \vspace{0.3cm}

    \begin{exampleblock}{Implicații}
        \begin{itemize}
            \item ACF al $\varepsilon_t^2$ decade exponențial (ca ARMA)
            \item Persistența este dată de $\alpha + \beta$
            \item PACF poate ajuta la identificarea ordinului
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{Demonstrație: Reprezentarea ARMA a GARCH(1,1)}
    \small
    \begin{proof}[Demonstrație]
        \textbf{Pasul 1}: Definim șocul varianței: $\nu_t = \varepsilon_t^2 - \sigma_t^2$
        \begin{itemize}
            \item $\E[\nu_t | \mathcal{F}_{t-1}] = \E[\varepsilon_t^2 | \mathcal{F}_{t-1}] - \sigma_t^2 = \sigma_t^2 - \sigma_t^2 = 0$
            \item $\nu_t$ este o secvență diferență de martingal
        \end{itemize}

        \vspace{0.2cm}
        \textbf{Pasul 2}: Înlocuim $\sigma_t^2 = \varepsilon_t^2 - \nu_t$ în ecuația GARCH:
        \[
            \varepsilon_t^2 - \nu_t = \omega + \alpha \varepsilon_{t-1}^2 + \beta (\varepsilon_{t-1}^2 - \nu_{t-1})
        \]

        \textbf{Pasul 3}: Rearanjăm:
        \[
            \varepsilon_t^2 = \omega + (\alpha + \beta) \varepsilon_{t-1}^2 + \nu_t - \beta \nu_{t-1}
        \]

        \textbf{Rezultat}: ARMA(1,1) cu coeficient AR $\phi = \alpha + \beta$ și coeficient MA $\theta = -\beta$.
    \end{proof}
\end{frame}

\begin{frame}{Demonstrație: persistența volatilității și half-life}
    \small
    \vspace{-0.2cm}
        \begin{prop}[Prognoză multi-pas GARCH(1,1)]
            \begin{itemize}
                \item $\E_t[\sigma_{t+h}^2] = \bar{\sigma}^2 + (\alpha+\beta)^{h-1}(\sigma_{t+1}^2 - \bar{\sigma}^2)$
            \end{itemize}
        \end{prop}
        \vspace{-0.15cm}
        \begin{block}{Demonstrație}
            \begin{itemize}
                \item \textbf{Pasul 1}: Notăm $\phi = \alpha + \beta$ și $q_t = \sigma_t^2 - \bar{\sigma}^2$ (deviația de la medie)
                \item \textbf{Pasul 2}: Din ecuația GARCH: $\E_t[q_{t+1}] = \phi \cdot q_t$, deci $\E_t[q_{t+h}] = \phi^h \cdot q_t$
                \item \textbf{Pasul 3}: Half-life: $\phi^{HL} = 0.5 \;\Rightarrow\; HL = \frac{-0.693}{\ln(\alpha+\beta)}$
            \end{itemize}
        \end{block}
        \vspace{-0.15cm}
        \begin{exampleblock}{Exemplu: S\&P 500}
            \begin{itemize}
                \item Cu $\alpha + \beta = 0.988$: $HL = \frac{-0.693}{-0.012} \approx \textbf{58 zile}$ (șocurile persistă $\sim$3 luni!)
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{De ce funcționează GARCH: intuiția centrală}
    \small
        \vspace{-0.3cm}
        \begin{block}{Ideea fundamentală}
            \begin{itemize}
                \item Randamentele financiare sunt greu de prognozat, dar \textbf{volatilitatea lor este prognozabilă}:
            \vspace{-3mm}
            \[
                r_t = \sigma_t \cdot z_t, \qquad z_t \sim \text{i.i.d.}(0,1)
            \]
            \vspace{-2mm}
            \begin{itemize}
                \item $r_t$ este impredictibil (piața eficientă), dar $\sigma_t^2$ \textbf{evoluează lent}
                \item GARCH capturează acest lucru: \textbf{șocurile afectează volatilitatea pentru multe perioade}
            \end{itemize}
            \end{itemize}
        \end{block}
        \vspace{-2mm}
        \begin{exampleblock}{Mecanismul GARCH(1,1)}
            \vspace{-3mm}
            \[
                \sigma_t^2 = \underbrace{\omega}_{\text{echilibru}} + \underbrace{\alpha\, r_{t-1}^2}_{\text{șoc nou}} + \underbrace{\beta\, \sigma_{t-1}^2}_{\text{memorie}}
            \]
            \vspace{-2mm}
            \begin{itemize}\setlength{\itemsep}{-1pt}
                \item $r_{t-1}^2$ mare (zi turbulentă) $\Rightarrow$ volatilitatea de azi crește
                \item $\sigma_{t-1}^2$ mare (volatilitate deja ridicată) $\Rightarrow$ persistă
                \item Termen lung: revenire la $\bar{\sigma}^2 = \omega / (1 - \alpha - \beta)$
            \end{itemize}
        \end{exampleblock}
        \vspace{-2mm}
        \begin{alertblock}{De ce este important}
            \begin{itemize}
                \item Fără GARCH: VaR constant $\Rightarrow$ subestimează riscul în crize.
            Cu GARCH: VaR dinamic $\Rightarrow$ se adaptează la condițiile pieței.
            \end{itemize}
        \end{alertblock}
\end{frame}


\begin{frame}{Estimarea modelelor GARCH}
    \small
    \vspace{-0.2cm}
        \begin{block}{Metoda verosimilității maxime (MLE --- Maximum Likelihood Estimation)}
            \begin{itemize}
                \item Log-verosimilitate (normală): $\ell(\theta) = -\frac{T}{2} \ln(2\pi) - \frac{1}{2} \sum_{t=1}^{T} \left[ \ln(\sigma_t^2) + \frac{\varepsilon_t^2}{\sigma_t^2} \right]$
            \end{itemize}
        \end{block}
        \vspace{-0.1cm}
        \begin{block}{Distribuții alternative pentru $z_t$}
            \begin{itemize}
                \item \textbf{Student-t}: capturează cozile groase --- cea mai frecventă alegere
                \item \textbf{GED}: flexibilitate pentru kurtosis
                \item \textbf{Skewed Student-t}: asimetrie și cozi groase
            \end{itemize}
        \end{block}
        \vspace{-0.1cm}
        \begin{alertblock}{Notă practică}
            \begin{itemize}
                \item Student-t oferă de regulă potrivire mai bună datorită cozilor groase (kurtosis $> 3$)
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Log-verosimilitatea Student-$t$}
    \small
        \begin{block}{De ce Student-$t$?}
            \begin{itemize}
                \item Inovațiile gaussiene subestimează cozile groase: kurtosis fixă la 3
                \item Student-$t$ adaugă un parametru $\nu > 2$ (grade de libertate) care controlează grosimea cozilor
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{defn}[Log-verosimilitatea Student-$t$ standardizată]
            \vspace{-2mm}
            \[
                \ell_t(\theta, \nu) = \ln\Gamma\!\left(\frac{\nu+1}{2}\right) - \ln\Gamma\!\left(\frac{\nu}{2}\right) - \frac{1}{2}\ln\bigl[(\nu-2)\pi\sigma_t^2\bigr] - \frac{\nu+1}{2}\ln\!\left(1 + \frac{\varepsilon_t^2}{(\nu-2)\sigma_t^2}\right)
            \]
            \vspace{-2mm}

            unde $\Gamma(\cdot)$ este funcția Gamma, $\nu > 2$, și $\sigma_t^2$ este din ecuația GARCH.
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Proprietăți}
            \begin{itemize}
                \item Când $\nu \to \infty$: converge la log-verosimilitatea normală
                \item Kurtosis Student-$t$: $3(\nu-2)/(\nu-4)$ pentru $\nu > 4$ (e.g., $\nu = 6 \Rightarrow \kappa = 6$)
                \item $\nu$ se estimează simultan cu $(\omega, \alpha, \beta)$ prin MLE
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Quasi-MLE și robustețe}
    \small
        \setlength{\abovedisplayskip}{1pt}\setlength{\belowdisplayskip}{1pt}
        \vspace{-0.4cm}
        \begin{block}{Ce se întâmplă dacă distribuția este greșit specificată?}
            \begin{itemize}
                \item MLE standard presupune $z_t \sim N(0,1)$, dar distribuția reală poate diferi
                \item \textbf{QMLE}: maximizează log-verosimilitatea gaussiană chiar dacă $z_t$ nu este normal
            \end{itemize}
        \end{block}
        \vspace{-0.3cm}
        \begin{thm}[Consistența QMLE --- Bollerslev \& Wooldridge (1992)]
            Dacă media și varianța condiționate sunt corect specificate, QMLE este \textbf{consistent} pentru $(\omega, \alpha, \beta)$ indiferent de distribuția reală a lui $z_t$.
            \textit{Condiții}: interiorul spațiului parametric, identificabilitate, momente finite.
        \end{thm}
        \vspace{-0.3cm}
        \begin{alertblock}{Erori standard sandwich (robuste)}
            \begin{itemize}
                \item $\hat{V}_{\text{robust}} = \hat{A}^{-1} \hat{B} \hat{A}^{-1}$ (White, 1982); în Python: SE robuste implicit
            \end{itemize}
        \end{alertblock}
        \vspace{-0.3cm}
        \begin{exampleblock}{QMLE vs MLE complet}
            \begin{itemize}
                \item \textbf{QMLE}: robust --- consistent chiar sub specificare greșită
                \item \textbf{MLE complet} (Student-$t$): mai eficient dacă distribuția este corectă
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Provocări practice în estimare}
    \small
        \begin{block}{Probleme frecvente}
            \begin{itemize}
                \item \textbf{Eșec de convergență}: log-verosimilitatea este non-convexă; optimizatorul se poate bloca
                \item \textbf{Soluții la frontieră}: $\hat{\alpha} \approx 0$ sau $\hat{\alpha} + \hat{\beta} \approx 1$ (aproape IGARCH --- Integrated GARCH)
                \item \textbf{Sensibilitate la valori inițiale}: inițializări diferite $\Rightarrow$ estimări diferite
            \end{itemize}
        \end{block}
        \vspace{-0.1cm}
        \begin{exampleblock}{Bune practici}
            \begin{enumerate}
                \item Încercați mai mulți optimizatori: \textbf{BFGS} (rapid, bazat pe gradient) vs \textbf{Nelder-Mead} (fără derivate)
                \item Inițializați $\sigma_0^2$ cu varianța eșantionului (backcast)
                \item Verificați $\alpha + \beta < 1$ post-estimare; dacă $\approx 1$, considerați IGARCH sau schimbare de regim
                \item Comparați rezultatele între valori inițiale; estimări consistente $\Rightarrow$ model fiabil
            \end{enumerate}
        \end{exampleblock}
        \vspace{-0.1cm}
        \begin{alertblock}{Regulă practică}
            \begin{itemize}
                \item Dacă BFGS și Nelder-Mead dau estimări \textit{diferite} ale parametrilor, modelul poate fi greșit specificat sau datele insuficiente
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{IGARCH: ce observăm în seria simulată?}
    \small
        \begin{block}{Citirea graficului de comparație}
            \begin{itemize}
                \item \textbf{GARCH standard} ($\alpha + \beta < 1$): volatilitatea crește după șocuri dar \textbf{revine la nivelul de echilibru} $\bar{\sigma}^2$
                \item Observăm „valuri" cu amplitudine descrescătoare
                \item \textbf{IGARCH} ($\alpha + \beta = 1$): volatilitatea \textbf{nu are ancoră}
                \item După un șoc mare, nivelul ridicat persistă indefinit, fără tendință de revenire
                \item Vizual: seria IGARCH ``rătăcește'' (random walk în varianță), pe când GARCH oscilează în jurul mediei
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Cum detectăm IGARCH în practică?}
            \begin{itemize}
                \item Estimăm GARCH(1,1) și verificăm: dacă $\hat{\alpha} + \hat{\beta} > 0.99$, considerăm IGARCH
                \item ACF a lui $\varepsilon_t^2$ decade foarte lent (ca un proces cu memorie lungă)
                \item Frecvent întâlnit pe piețe valutare (FX) și unele mărfuri
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{Atenție}
            \begin{itemize}
                \item IGARCH poate fi un artefact al schimbărilor de regim --- verificați cu modele Markov-switching!
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{IGARCH --- integrated GARCH}
    \small
    \vspace{-0.2cm}
        \begin{alertblock}{De ce IGARCH? Persistența extremă}
            \begin{itemize}
                \item Empiric, estimarea GARCH dă frecvent $\hat{\alpha} + \hat{\beta} \approx 0.99$
                \item IGARCH formalizează cazul limită $\alpha + \beta = 1$: șocurile au efect permanent, similar unei rădăcini unitare în varianță
            \end{itemize}
        \end{alertblock}
        \vspace{-0.15cm}
        \begin{defn}[IGARCH(1,1)]
            Când $\alpha + \beta = 1$:
            \vspace{-2mm}
            \[
                \sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + (1 - \alpha) \sigma_{t-1}^2
            \]
            \vspace{-2mm}
        \end{defn}
        \vspace{-0.15cm}
        \begin{prop}[Proprietăți IGARCH(1,1)]
            \begin{itemize}
                \item Varianța necondiționată nu există (infinită), dar procesul rămâne strict staționar (Nelson, 1990)
                \item Șocurile au efect \textbf{permanent} --- analog cu rădăcina unitară în varianță
            \end{itemize}
        \end{prop}
        \vspace{-0.15cm}
        \begin{rmk}[Aplicație practică]
            Util pentru \textbf{RiskMetrics} (J.P.\ Morgan): $\alpha = 0.06$, $\beta = 0.94$.
        \end{rmk}
        \hfill\quantlet{TSA\_ch5\_igarch}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_igarch}
\end{frame}

\begin{frame}{Teorema lui Nelson (1990): Staționaritate strictă IGARCH}
    \small
        \vspace{-2mm}
        \begin{thm}[Nelson, 1990]
            Fie procesul IGARCH(1,1):
            $\varepsilon_t = \sigma_t z_t$, \; $\sigma_t^2 = \omega + \alpha\,\varepsilon_{t-1}^2 + (1 - \alpha)\,\sigma_{t-1}^2$, \; $z_t$ i.i.d., $\E[z_t^2] = 1$.
            Dacă $\omega > 0$ și $\E\bigl[\ln\!\bigl(\alpha z_t^2 + (1 - \alpha)\bigr)\bigr] < 0$, atunci procesul $\{\varepsilon_t\}$ este \textbf{strict staționar} și \textbf{ergodic}, deși varianța necondiționată este infinită.
        \end{thm}
        \vspace{-1mm}
        \begin{block}{Condiția Lyapunov}
            \begin{itemize}
                \item Condiția $\E[\ln(\alpha z_t^2 + (1-\alpha))] < 0$ este echivalentă cu \textbf{exponentul Lyapunov de vârf negativ}
                \item Se verifică pentru distribuții standard ($z_t \sim \mathcal{N}(0,1)$, $t$-Student) când $\alpha \in (0,1)$
                \item Interpretare: log-volatilitatea $\ln \sigma_t^2$ are o tendință medie de contracție
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{block}{Schiță de demonstrație}
            \begin{itemize}
                \item Se rescrie $\sigma_t^2$ ca recurență stocastică: $Y_t = A_t Y_{t-1} + B_t$, unde $A_t = \alpha z_{t-1}^2 + (1-\alpha)$, $B_t = \omega$
                \item Se aplică teorema Bougerol \& Picard (1992): dacă $\E[\ln A_t] < 0$ și $\E[\ln^+ B_t] < \infty$, seria $Y_t$ converge în distribuție
            \end{itemize}
        \end{block}
        \hfill\quantlet{TSA\_ch5\_nelson\_theorem}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_nelson_theorem}
\end{frame}

\begin{frame}{IGARCH: Staționaritate strictă vs.\ staționaritate în covarianță}
    \small
        \vspace{-2mm}
        \begin{block}{Contrast fundamental}
            \renewcommand{\arraystretch}{1.2}
            \begin{center}
            \begin{tabular}{lcc}
                \toprule
                \textbf{Proprietate} & \textbf{GARCH} ($\alpha + \beta < 1$) & \textbf{IGARCH} ($\alpha + \beta = 1$) \\
                \midrule
                Strict staționar & Da & \textbf{Da} (Nelson, 1990) \\
                Staționar în covarianță & Da & \textbf{Nu} ($\Var[\varepsilon_t] = \infty$) \\
                Varianță necondiționată & $\bar{\sigma}^2 = \frac{\omega}{1 - \alpha - \beta}$ & Nu există (infinită) \\
                Persistența șocurilor & Geometrică (se estompează) & Permanentă \\
                \bottomrule
            \end{tabular}
            \end{center}
        \end{block}
        \vspace{-1mm}
        \begin{alertblock}{Implicație practică}
            \begin{itemize}
                \item IGARCH este un \textbf{model valid}, în ciuda rădăcinii unitare în varianță
                \item Prognoza varianței condiționate: $\E_t[\sigma_{t+h}^2] = \sigma_t^2 + h\omega$ (creștere liniară)
                \item Nu implică explozie a procesului: $\varepsilon_t$ rămâne bine definit
            \end{itemize}
        \end{alertblock}
        \hfill\quantlet{TSA\_ch5\_igarch}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_igarch}
\end{frame}

\begin{frame}{EWMA: GARCH restricționat pentru RiskMetrics}
    \small
        \vspace{-0.2cm}
        \begin{defn}[Exponentially Weighted Moving Average]
            Modelul \textbf{EWMA} (J.P.\ Morgan RiskMetrics, 1996):
            \vspace{-2mm}
            \[
                \sigma_t^2 = \lambda\, \sigma_{t-1}^2 + (1 - \lambda)\, r_{t-1}^2, \qquad \lambda \in (0,1)
            \]
        \end{defn}
        \vspace{-0.2cm}
        \begin{prop}[EWMA ca caz special al IGARCH]
            \begin{itemize}
                \item EWMA = \textbf{IGARCH(1,1) cu $\omega = 0$}: \quad $\alpha = 1 - \lambda$, \; $\beta = \lambda$, \; $\alpha + \beta = 1$
                \item Standard industrial: $\lambda = 0.94$ (date zilnice), $\lambda = 0.97$ (date lunare)
                \item \textbf{Nicio estimare necesară} --- parametru fix $\Rightarrow$ simplu, dar rigid
            \end{itemize}
        \end{prop}
        \vspace{-0.2cm}
        \begin{alertblock}{EWMA vs GARCH(1,1)}
            \vspace{-1mm}
            \begin{center}
            \begin{tabular}{lcc}
                \toprule
                & \textbf{EWMA} & \textbf{GARCH(1,1)} \\
                \midrule
                Parametri estimați & 0 ($\lambda$ fix) & 3 ($\omega, \alpha, \beta$) \\
                Mean-reversion & \textcolor{Crimson}{Nu} (persistență infinită) & \textcolor{Forest}{Da} (dacă $\alpha + \beta < 1$) \\
                Varianță pe termen lung & \textcolor{Crimson}{Nedefinită} & \textcolor{Forest}{$\omega / (1 - \alpha - \beta)$} \\
                Utilizare & Basel II, screening rapid & Estimare precisă, VaR \\
                \bottomrule
            \end{tabular}
            \end{center}
        \end{alertblock}
\end{frame}

\begin{frame}{EWMA: interpretare și ce observăm în serie}
    \small
        \begin{block}{Citirea seriei EWMA}
            \begin{itemize}
                \item Seria volatilității EWMA urmărește ``fidel'' pătratele randamentelor --- reacție rapidă, fără filtrare
                \item Cu $\lambda = 0.94$: ponderea de ieri este $6\%$, restul vine din istoria exponențial atenuată
                \item Vizual: volatilitatea EWMA ``lipește'' de spike-uri --- urcă rapid dar nu revine la un nivel fix
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Când este suficient EWMA?}
            \begin{itemize}
                \item \textbf{Screening rapid}: nu necesită estimare --- calculabil imediat pe orice serie
                \item \textbf{Basel II/III}: RiskMetrics EWMA rămâne acceptat pentru VaR intern
                \item \textbf{Real-time monitoring}: simplu de implementat în sisteme de trading

                \item EWMA nu revine la un nivel de echilibru $\bar{\sigma}^2$ (nu are $\omega > 0$)
                \item După crize, volatilitatea EWMA rămâne ridicată mai mult decât GARCH --- \textbf{suprareacționează pe termen mediu}
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{CGARCH --- Component GARCH (Engle \& Lee, 1999)}
    \small
        \begin{block}{Motivație}
            \begin{itemize}
                \item GARCH(1,1) are o \textbf{singură componentă} de persistență: nu distinge între șocuri de scurtă și lungă durată
                \item CGARCH descompune varianța în componentă \textbf{permanentă} (trendul) + \textbf{tranzitorie}
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{defn}[CGARCH(1,1)]
            \vspace{-2mm}
            \begin{align*}
                \sigma_t^2 &= q_t + \alpha(\varepsilon_{t-1}^2 - q_{t-1}) + \beta(\sigma_{t-1}^2 - q_{t-1}) & \text{(componenta tranzitorie)}\\
                q_t &= \bar{\omega} + \rho\,(q_{t-1} - \bar{\omega}) + \phi\,(\varepsilon_{t-1}^2 - \sigma_{t-1}^2) & \text{(componenta permanentă)}
            \end{align*}
            \vspace{-2mm}

            unde $q_t$ este tendința pe termen lung, $\rho$ controlează persistența permanentă ($\rho > \alpha+\beta$).
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Interpretare}
            \begin{itemize}
                \item $q_t$ variază lent (factori macroeconomici); $\sigma_t^2 - q_t$ reacționează rapid la șocuri
                \item Permite half-life-uri diferite: permanent $\sim$ani, tranzitoriu $\sim$săptămâni
                \item Disponibil în Python: \texttt{arch} cu specificare manuală
            \end{itemize}
        \end{alertblock}
\end{frame}

    \begin{itemize}
        \item $q_t$ (albastru, gros) = trendul pe termen lung; $\sigma_t^2$ (roșu, subțire) = volatilitatea totală
    \end{itemize}

\begin{frame}{CGARCH: ce observăm în practică?}
    \small
    \vspace{-0.2cm}
        \begin{block}{Interpretarea celor două componente}
            \begin{itemize}
                \item \textbf{Componenta permanentă} $q_t$: se mișcă lent --- factori macroeconomici (inflație, ciclul economic)
                \item \textbf{Componenta tranzitorie} $\sigma_t^2 - q_t$: reacționează rapid la știri (earnings, Fed, geopolitică)
                \item Vizual: $q_t$ = medie mobilă netedă; $\sigma_t^2$ oscilează în jurul ei
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{exampleblock}{Exemplu: S\&P 500}
            \begin{itemize}
                \item Componenta permanentă crește 2005--2008, explodează 2008--2009, scade lent după
                \item Componenta tranzitorie: spike-uri la Flash Crash (2010), Brexit (2016), COVID (2020)
                \item Half-life: permanent $\approx 2$--$3$ ani; tranzitoriu $\approx 2$--$4$ săptămâni
            \end{itemize}
        \end{exampleblock}
        \vspace{-1mm}
        \begin{alertblock}{Când folosim CGARCH?}
            \begin{itemize}
                \item VaR pe termen lung (1 an+) --- componenta permanentă contează pentru Solvency II, stress testing
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{FIGARCH --- Fractionally Integrated GARCH}
    \small
        \begin{block}{Motivație: memorie lungă în volatilitate}
            \begin{itemize}
                \item ACF a $r_t^2$ descreşte \textbf{hiperbolic} (nu exponențial) $\Rightarrow$ memorie lungă
                \item GARCH: descreștere exponențială; IGARCH: persistență infinită --- niciunul nu captează memoria lungă
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{defn}[FIGARCH(1,$d$,1) --- Baillie, Bollerslev \& Mikkelsen (1996)]
            \vspace{-2mm}
            \[
                \bigl[1 - \beta L\bigr]\sigma_t^2 = \omega + \bigl[1 - \beta L - (1-\phi L)(1-L)^d\bigr]\varepsilon_t^2
            \]
            \vspace{-2mm}

            unde $d \in [0,1]$ este parametrul de integrare fracționară: $d=0$ $\Rightarrow$ GARCH, $d=1$ $\Rightarrow$ IGARCH.
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Proprietăți cheie}
            \begin{itemize}
                \item Ponderile impulsurilor $\lambda_k \sim k^{d-1}$ descreşte hiperbolic: șocurile vechi contează, dar tot mai puțin
                \item Varianța necondiționată: finită doar dacă $d < 1$ (spre deosebire de IGARCH)
                \item Valori empirice tipice: $\hat{d} \approx 0.3$--$0.5$ pentru acțiuni (Baillie et al., 1996)
            \end{itemize}
        \end{alertblock}
\end{frame}

    \vspace{-0.2cm}
    \begin{itemize}
        \item GARCH: revenire exponențială; FIGARCH: revenire hiperbolică (lentă); IGARCH: persistență infinită
    \end{itemize}

\begin{frame}{FIGARCH: interpretare și aplicații practice}
    \small
        \begin{block}{Ce înseamnă $d$ în practică?}
            \begin{itemize}
                \item \textbf{$d = 0$} (GARCH): ACF a $r_t^2$ decade exponențial --- șocurile se sting rapid
                \item \textbf{$d = 1$} (IGARCH): ACF nu decade --- șocurile persistă infinit
                \item \textbf{$0 < d < 1$} (FIGARCH): ACF decade \textbf{hiperbolic} ($\sim k^{2d-1}$) --- șocurile vechi contează, dar tot mai puțin
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Rezultate empirice}
            \begin{itemize}
                \item \textbf{Acțiuni} (S\&P 500, DAX): $\hat{d} \approx 0.35$--$0.50$ --- memorie lungă moderată
                \item \textbf{Forex} (EUR/USD): $\hat{d} \approx 0.40$--$0.60$ --- persistență mai ridicată
                \item \textbf{Crypto} (Bitcoin): $\hat{d} \approx 0.25$--$0.45$ --- variabil în timp
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{FIGARCH vs IGARCH: cum alegem?}
            \begin{itemize}
                \item Dacă $\hat{\alpha} + \hat{\beta} \approx 1$ în GARCH, testați FIGARCH
                \item Dacă $\hat{d} < 1$, persistența nu este infinită --- ci doar lungă
                \item FIGARCH oferă o imagine mai realistă decât IGARCH spurios
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Schimbări de regim în volatilitate}
    \small
        \setlength{\abovedisplayskip}{1pt}\setlength{\belowdisplayskip}{1pt}
        \vspace{-0.4cm}
        \begin{block}{Problemă}
            \begin{itemize}
                \item GARCH presupune un \textbf{singur regim} cu parametri constanți
                \item Volatilitatea prezintă adesea \textbf{rupturi structurale} (ex.\ criza 2008, COVID-19)
            \end{itemize}
        \end{block}
        \vspace{-0.25cm}
        \begin{defn}[GARCH cu schimbare de regim Markov]
            Parametrii $(\omega, \alpha, \beta)$ comută între $K$ regimuri guvernate de un lanț Markov ascuns:
            \[
                \sigma_t^2 = \omega_{s_t} + \alpha_{s_t} \varepsilon_{t-1}^2 + \beta_{s_t} \sigma_{t-1}^2, \quad s_t \in \{1, \ldots, K\}
            \]
        \end{defn}
        \vspace{-0.25cm}
        \begin{exampleblock}{Testul ICSS (Iterated Cumulative Sums of Squares) --- Inclán-Tiao}
            \begin{itemize}
                \item Detectează \textbf{puncte de schimbare multiple} în varianța necondiționată (tip CUSUM)
                \item Rezultate tipice: S\&P 500 arată rupturi la 2001, 2008, 2020
            \end{itemize}
        \end{exampleblock}
        \vspace{-0.25cm}
        \begin{alertblock}{Implicații practice}
            \begin{itemize}\setlength{\itemsep}{-1pt}
                \item Dacă rupturile sunt prezente, persistența ($\alpha + \beta$) este \textbf{distorsionată în sus} $\Rightarrow$ IGARCH fals
                \item Soluție: estimare per regim, ferestre rulante, sau GARCH Markov ($K=2$)
            \end{itemize}
        \end{alertblock}
\end{frame}

    \small
    \vspace{-0.2cm}
    \begin{itemize}
        \item Regim calm (albastru) vs turbulent (roșu) --- volatilitatea sare între niveluri discrete
    \end{itemize}

\begin{frame}{Schimbări de regim: interpretare și aplicații}
    \small
        \begin{block}{Ce vedem în practică?}
            \begin{itemize}
                \item Markov-switching GARCH identifică 2--3 regimuri:
                    \textbf{calm} ($\sigma \approx 0.5$--$1\%$) vs \textbf{turbulent} ($\sigma \approx 2$--$4\%$)
                \item Probabilitatea regimului turbulent crește brusc la crize (2008, 2020)
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Rupturi detectate cu ICSS pe S\&P 500}
            \begin{itemize}
                \item \textbf{2001}: dot-com crash + 9/11; \; \textbf{2008}: criză financiară globală; \; \textbf{2020}: COVID-19
                \item Între rupturi: GARCH cu parametri diferiți este mai precis decât GARCH pe toată perioada
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{Implicație practică}
            \begin{itemize}
                \item Dacă GARCH dă $\hat{\alpha} + \hat{\beta} \approx 1$, verificați dacă este IGARCH real sau artefact al schimbărilor de regim. Testul ICSS ajută la decizie.
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{De la GARCH simetric la modele asimetrice}
    \small
        \begin{block}{Ce am învățat până acum}
            \begin{itemize}
                \item GARCH(1,1) capturează volatility clustering și cozi groase
                \item IGARCH, CGARCH, FIGARCH --- diferite tipuri de persistență
            \end{itemize}
        \end{block}
        \vspace{-0.15cm}
        \begin{rmk}[Limitația simetriei GARCH standard]
            \begin{itemize}
                \item GARCH tratează \textbf{simetric} șocurile: $\varepsilon_{t-1}^2$ nu distinge semnul $\varepsilon_{t-1}$
                \item Empiric: un șoc negativ de $-5\%$ crește volatilitatea mai mult decât $+5\%$ --- \textbf{leverage effect} (Black, 1976)
            \end{itemize}
        \end{rmk}
        \vspace{-0.15cm}
        \begin{exampleblock}{Modele asimetrice prezentate}
            \begin{itemize}
                \item EGARCH $\cdot$ GJR-GARCH $\cdot$ TGARCH $\cdot$ APARCH --- toate disting semnul șocului
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Leverage effect}
    \small
        \vspace{-0.2cm}
        \begin{block}{Definiție}
            \begin{itemize}
                \item \textbf{Leverage effect}: Șocurile negative cresc volatilitatea \textbf{mai mult} decât cele pozitive de aceeași magnitudine
            \end{itemize}
        \end{block}
        \vspace{-1.5mm}
        \begin{alertblock}{Problema GARCH standard}
            \begin{itemize}
                \item GARCH standard: $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$ --- doar $\varepsilon_{t-1}^2$ contează, semnul se pierde
                \item Intuiție economică: Semnale negative $\Rightarrow$ prețul scade $\Rightarrow$ levierul crește $\Rightarrow$ volatilitatea crește
            \end{itemize}
        \end{alertblock}
        \vspace{-1.5mm}
        \begin{exampleblock}{Evidență empirică}
            \begin{itemize}
                \item Black (1976): prima documentare a efectului de levier pe piața de acțiuni
                \item Efectul este asimetric: căderi de 5\% cresc volatilitatea mai mult decât creșteri de 5\%
                \item Soluții: EGARCH, GJR-GARCH, TGARCH --- modele care disting semnul șocului
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Simulare EGARCH: ce observăm în serie?}
    \small
        \begin{block}{Citirea graficului de simulare}
            \begin{itemize}
                \item \textbf{Panoul stâng} ($\gamma = 0$, simetric): volatilitatea reacționează egal la șocuri pozitive și negative --- similar cu GARCH standard
                \item \textbf{Panoul drept} ($\gamma < 0$, asimetric): după șocuri \textbf{negative}, volatilitatea crește brusc și persistent
                \item După șocuri pozitive, reacția este atenuată
                \item Vizual: spike-urile de volatilitate sunt mai \textbf{înalte} și mai \textbf{largi} după scăderi de piață
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Exemplu numeric: interpretarea parametrului $\gamma$}
            \begin{itemize}
                \item Cu $\gamma = -0.15$, $\alpha = 0.10$:
                    \begin{itemize}
                        \item Șoc negativ $z_{t-1} = -2$: contribuție $= 0.10 \times 2 + (-0.15)(-2) = 0.50$
                        \item Șoc pozitiv $z_{t-1} = +2$: contribuție $= 0.10 \times 2 + (-0.15)(+2) = -0.10$
                        \item Diferența: \textbf{0.60} unități pe scala $\ln(\sigma_t^2)$
                    \end{itemize}
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{Implicație pentru risk management}
            \begin{itemize}
                \item EGARCH captează faptul că piețele ``iau liftul în jos și scările în sus'' --- volatilitatea crește rapid dar scade lent.
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Modelul EGARCH --- Nelson (1991)}
    \small
        \vspace{-0.3cm}
        \begin{alertblock}{De ce EGARCH? Două probleme ale GARCH}
            \begin{itemize}
                \item GARCH tratează șocurile \textbf{simetric}: $\varepsilon_{t-1}^2$ nu distinge semnul $\Rightarrow$ nu captează leverage effect
                \item GARCH necesită restricții $\omega > 0$, $\alpha, \beta \geq 0$ pentru $\sigma_t^2 > 0$ $\Rightarrow$ dificil la estimare
            \end{itemize}
        \end{alertblock}
        \vspace{-0.15cm}
        \begin{defn}[EGARCH(1,1)]
            \textbf{Exponential GARCH}:
            \vspace{-2mm}
            \[
                \ln(\sigma_t^2) = \omega + \alpha \left( |z_{t-1}| - \E[|z_{t-1}|] \right) + \gamma z_{t-1} + \beta \ln(\sigma_{t-1}^2)
            \]
            \vspace{-2mm}

            unde $z_t = \varepsilon_t / \sigma_t$.
        \end{defn}
        \vspace{-0.15cm}
        \begin{block}{Avantaje EGARCH}
            \begin{itemize}
                \item \textbf{Nu necesită restricții de non-negativitate} --- modelează $\ln(\sigma_t^2)$
                \item \textbf{Captează leverage effect}: $\gamma < 0 \Rightarrow$ șocuri negative cresc volatilitatea mai mult
                \item Persistența este dată de $\beta$
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{News impact curve -- EGARCH}
    \small
        \begin{block}{Interpretare}
            \begin{itemize}
                \item \textbf{News Impact Curve}: relația între $\varepsilon_t$ și $\sigma_{t+1}^2$
                \item \textbf{GARCH}: curba simetrică (parabolă)
                    \begin{itemize}
                        \item Șocuri pozitive și negative au același impact
                    \end{itemize}
                \item \textbf{EGARCH}: curba asimetrică
                    \begin{itemize}
                        \item Șocuri negative au impact mai mare asupra volatilității
                    \end{itemize}
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{EGARCH: aplicații practice}
    \small
        \begin{block}{Când folosim EGARCH?}
            \begin{itemize}
                \item \textbf{Piețe de acțiuni și indici} --- leverage effect pronunțat (Black, 1976)
                \item \textbf{Avantaj unic}: nu necesită restricții de pozitivitate ($\ln\sigma_t^2$ poate lua orice valoare reală)
                \item Util când GARCH simetric dă $\hat{\alpha} + \hat{\beta} > 1$ (artefact al ignorării asimetriei)
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Exemplu: S\&P 500 --- GARCH vs EGARCH}
            \begin{itemize}
                \item GARCH(1,1): subestimează volatilitatea după crash-uri (tratează simetric)
                \item EGARCH: volatilitatea crește mai rapid după scăderi, scade mai lent
                \item În criza COVID (martie 2020), EGARCH prognozează VaR cu 15--20\% mai mare
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{Regulă practică}
            \begin{itemize}
                \item Testați EGARCH vs GARCH pe date reale: dacă AIC/BIC favorează EGARCH, piața are leverage semnificativ.
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Modelul GJR-GARCH}
    \small
        \vspace{-0.3cm}
        \begin{alertblock}{De ce GJR? Alternativă la EGARCH}
            \begin{itemize}
                \item EGARCH modelează $\ln(\sigma_t^2)$ $\Rightarrow$ greu de interpretat economic
                \item GJR-GARCH rămâne pe scara $\sigma_t^2$ dar adaugă un indicator $I_{t-1}$ care activează un termen suplimentar la șocuri negative
            \end{itemize}
        \end{alertblock}
        \vspace{-0.15cm}
        \begin{defn}[GJR-GARCH(1,1)]
            Glosten, Jagannathan \& Runkle (1993):
            $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \gamma \varepsilon_{t-1}^2 \cdot I_{t-1} + \beta \sigma_{t-1}^2$
            unde $I_{t-1} = 1$ dacă $\varepsilon_{t-1} < 0$, altfel $0$.
        \end{defn}
        \vspace{-0.1cm}
        \begin{block}{Interpretare}
            \begin{itemize}
                \item Șocuri pozitive: impact = $\alpha$; \quad Șocuri negative: impact = $\alpha + \gamma$
                \item Leverage effect prezent dacă $\gamma > 0$
                \item Staționaritate: $\alpha + \gamma/2 + \beta < 1$
            \end{itemize}
        \end{block}
\end{frame}

    \vspace{-0.2cm}
    \begin{block}{Interpretare}
        \begin{itemize}
            \item GJR-GARCH adaugă un termen indicator pentru a captura răspunsul asimetric la șocuri negative
        \end{itemize}

\begin{frame}{GJR-GARCH: ce observăm în seria simulată?}
    \small
        \begin{block}{Citirea graficului de simulare}
            \begin{itemize}
                \item Indicatorul $I_{t-1} = \mathbb{1}(\varepsilon_{t-1} < 0)$ se ``activează'' doar la șocuri negative
                \item Impact asimetric: $\varepsilon_{t-1} = -3\%$ $\Rightarrow$ contribuție $= (\alpha + \gamma) \times 0.09 = 0.15 \times 0.09$
                \item Impact simetric: $\varepsilon_{t-1} = +3\%$ $\Rightarrow$ contribuție $= \alpha \times 0.09 = 0.05 \times 0.09$
                \item Vizual: după zilele de scădere, benzile de volatilitate se lărgesc de \textbf{3$\times$ mai mult}
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Exemplu numeric: S\&P 500}
            Cu $\alpha = 0.05$, $\gamma = 0.10$, $\beta = 0.88$, $\omega = 0.00001$:
            \begin{itemize}
                \item Zi bună ($+2\%$): $\Delta\sigma_t^2 \approx 0.05 \times 0.04\% = 0.002\%$
                \item Zi rea ($-2\%$): $\Delta\sigma_t^2 \approx (0.05 + 0.10) \times 0.04\% = 0.006\%$ --- \textbf{triplu!}
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{GJR vs EGARCH: când alegem GJR?}
            \begin{itemize}
                \item GJR rămâne pe scala $\sigma_t^2$ $\Rightarrow$ coeficienții sunt direct interpretabili ca impacturi marginale
                \item EGARCH operează pe $\ln(\sigma_t^2)$ $\Rightarrow$ mai greu de citit economic, dar fără restricții de pozitivitate
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Modelul APARCH --- Ding, Granger \& Engle (1993)}
    \small
        \vspace{-0.3cm}
        \begin{defn}[APARCH(1,1)]
            \textbf{Asymmetric Power ARCH} --- modelul unificator al familiei GARCH:
            \vspace{-2mm}
            \[
                \sigma_t^\delta = \omega + \alpha_1 (|\varepsilon_{t-1}| - \gamma_1 \varepsilon_{t-1})^\delta + \beta_1 \sigma_{t-1}^\delta
            \]
            \vspace{-2mm}

            unde $\delta > 0$ este parametrul de putere și $|\gamma_1| < 1$ captează asimetria.
        \end{defn}
        \vspace{-1mm}
        \begin{block}{Cazuri particulare}
            \begin{itemize}
                \item $\delta = 2$, $\gamma_1 = 0$: \textbf{GARCH(1,1)} standard
                \item $\delta = 2$, $\gamma_1 \neq 0$: \textbf{GJR-GARCH}
                \item $\delta = 1$, $\gamma_1 \neq 0$: \textbf{TGARCH} (Zakoian, 1994)
                \item $\delta = 1$, $\gamma_1 = 0$: \textbf{AVGARCH} (Taylor, 1986)
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{alertblock}{De ce contează APARCH}
            \begin{itemize}
                \item Estimarea lui $\delta$ din date (nu fixat la 1 sau 2) poate \textbf{îmbunătăți prognoza}
                \item Studii empirice: $\hat{\delta} \approx 1.0$--$1.5$ pentru acțiuni (Ding et al., 1993)
                \item Oferă un \textbf{cadru unificator} pentru compararea modelelor asimetrice
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Simulare APARCH: cazuri particulare}
    \small
    \vspace{-0.2cm}
    \begin{itemize}
        \item APARCH unifică toată familia GARCH: $\delta$ și $\gamma$ controlează puterea și asimetria
    \end{itemize}
    \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.65\textheight, keepaspectratio]{garch_aparch_simulation.pdf}

\end{frame}

\begin{frame}{APARCH: interpretare și aplicații practice}
    \small
        \begin{block}{Ce spune estimarea lui $\delta$?}
            \begin{itemize}
                \item $\hat{\delta} \approx 2$: piața se comportă ca un GARCH standard (varianța este măsura naturală)
                \item $\hat{\delta} \approx 1$: modelarea deviației standard ($\sigma_t$) este mai potrivită --- ca TGARCH
                \item $\hat{\delta} < 1$: randamentele absolute au structură mai puternică decât pătratele
            \end{itemize}
        \end{block}
        \begin{exampleblock}{Rezultate empirice tipice}
            \begin{itemize}
                \item \textbf{Acțiuni}: $\hat{\delta} \approx 1.0$--$1.5$, $\hat{\gamma} \approx 0.3$--$0.8$ (leverage puternic)
                \item \textbf{Forex}: $\hat{\delta} \approx 1.5$--$2.0$, $\hat{\gamma} \approx 0$ (simetric, fără leverage)
                \item \textbf{Crypto}: $\hat{\delta} \approx 1.0$--$1.3$, $\hat{\gamma}$ variabil (depinde de perioadă)
            \end{itemize}
        \end{exampleblock}
        \begin{alertblock}{Când folosim APARCH?}
            \begin{itemize}
                \item Când nu știm a priori ce putere ($\delta$) sau ce nivel de asimetrie ($\gamma$) este optim --- APARCH lasă datele să decidă
                \item Util ca model de referință pentru testarea restricțiilor (LR test vs GARCH, GJR etc.)
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Staționaritatea modelelor asimetrice}
    \small
        \begin{block}{Condiții de staționaritate --- modele asimetrice}
            \begin{center}
            \begin{tabular}{lll}
                \toprule
                \textbf{Model} & \textbf{Condiție} & \textbf{Observații} \\
                \midrule
                GARCH(1,1) & $\alpha + \beta < 1$ & Covarianță-staționaritate \\
                GJR-GARCH & $\alpha + \gamma/2 + \beta < 1$ & $\gamma/2$ deoarece $P(\varepsilon<0) = 1/2$ \\
                EGARCH(1,1) & $|\beta| < 1$ & $\ln \sigma_t^2$ este AR(1)-like \\
                TGARCH(1,1) & $(\alpha^+ + \alpha^-)/2 + \beta < 1$ & Media impactului pozitiv/negativ \\
                APARCH(1,1) & $\alpha_1 \E[(|z|-\gamma z)^\delta] + \beta_1 < 1$ & Depinde de distribuția lui $z_t$ \\
                \bottomrule
            \end{tabular}
            \end{center}
        \end{block}
        \vspace{-1mm}
                \begin{prop}[Staționaritate EGARCH (Nelson, 1991)]
            \begin{itemize}
                \item EGARCH modelează $\ln \sigma_t^2$ ca proces AR(1) cu inovații $g(z_{t-1})$
                \item $|\beta| < 1$: \textbf{necesară și suficientă} pentru staționaritate strictă
                \item \textbf{Notă}: nu se confundă cu $\alpha + \beta < 1$ din GARCH --- $\alpha$ din EGARCH are alt rol
            \end{itemize}
        \end{prop}        
\end{frame}

\begin{frame}{TGARCH --- threshold GARCH}
    \small
    \vspace{-0.2cm}
    \begin{alertblock}{De ce TGARCH? Modelarea directă a deviației standard}
        \begin{itemize}
            \item GARCH/EGARCH/GJR modelează $\sigma_t^2$. TGARCH modelează direct $\sigma_t$
            \item Permite coeficienți separați $\alpha^+$, $\alpha^-$ pentru șocuri pozitive și negative --- interpretare mai intuitivă
        \end{itemize}
    \end{alertblock}
    \vspace{-0.15cm}
    \begin{defn}[TGARCH(1,1)]
        Zakoian (1994) modelează deviația standard:
        $\sigma_t = \omega + \alpha^+ \varepsilon_{t-1}^+ + \alpha^- \varepsilon_{t-1}^- + \beta \sigma_{t-1}$
    \end{defn}

    \vspace{0.2cm}

    \begin{block}{Comparație modele asimetrice}
        \begin{center}
            \small
            \begin{tabular}{lcc}
                \toprule
                \textbf{Model} & \textbf{Specificație} & \textbf{Leverage} \\
                \midrule
                GARCH & $\sigma_t^2$ & Nu \\
                EGARCH & $\ln(\sigma_t^2)$ & Da ($\gamma < 0$) \\
                GJR-GARCH & $\sigma_t^2$ cu indicător & Da ($\gamma > 0$) \\
                TGARCH & $\sigma_t$ & Da ($\alpha^- > \alpha^+$) \\
                APARCH & $\sigma_t^\delta$ & Da ($\gamma_i \neq 0$) \\
                \bottomrule
            \end{tabular}
        \end{center}
    \end{block}
\end{frame}

            \begin{itemize}

                \item Empiric, diferențele sunt mici; TGARCH este preferat în risk management (VaR folosește $\sigma_t$ direct)
            \end{itemize}
        \end{alertblock}

%% Rec. 4: Când aleg ARCH, GARCH, EGARCH sau GJR-GARCH?
\begin{frame}{Când aleg ARCH, GARCH, EGARCH sau GJR-GARCH?}
    \small
        \vspace{-0.2cm}
        \begin{block}{Ghid de alegere}
            \begin{itemize}\setlength{\itemsep}{1pt}
                \item \textbf{ARCH(q)}: simplu conceptual, dar necesită multe lag-uri $\Rightarrow$ \textcolor{Crimson}{rar folosit în practică}
                \item \textbf{GARCH(1,1)}: \textcolor{Forest}{\textbf{benchmark-ul standard}} --- captează persistența cu doar 3 parametri; suficient în $\sim$90\% din cazuri
                \item \textbf{EGARCH}: modelează $\ln(\sigma_t^2)$ $\Rightarrow$ \textbf{fără restricții de pozitivitate}; captează leverage prin $\gamma$
                \item \textbf{GJR-GARCH}: variabilă indicatoare $I(\varepsilon_{t-1}<0)$ $\Rightarrow$ \textbf{interpretare directă a impactului asimetric}; echivalent cu EGARCH
            \end{itemize}
        \end{block}
        \vspace{-1.5mm}
        \begin{alertblock}{Reguli practice de selecție}
            \begin{itemize}
                \item \textbf{Pasul 1}: Începe cu GARCH(1,1) --- suficient pentru majoritatea seriilor
                \item \textbf{Pasul 2}: Verifică leverage effect (sign bias test, news impact curve)
                \item \textbf{Pasul 3}: Dacă leverage semnificativ $\Rightarrow$ EGARCH sau GJR-GARCH
                \item \textbf{Pasul 4}: Compară cu AIC/BIC; alege distribuția Student-t
            \end{itemize}
        \end{alertblock}
        \vspace{-1mm}
        \begin{center}
            \textcolor{MainBlue}{\textbf{Modelele nu sunt o listă --- sunt un set de alegeri motivate de date.}}
        \end{center}
\end{frame}

\begin{frame}{GARCH-in-Mean (GARCH-M) --- Engle, Lilien \& Robins (1987)}
    \small
    \vspace{-0.3cm}
    \begin{defn}[GARCH-M]
        \begin{itemize}
            \item \textbf{Model}: Volatilitatea intră direct în ecuația mediei:
            \vspace{-0.2cm}
            \begin{align*}
                r_t &= \mu + \delta \cdot g(\sigma_t^2) + \varepsilon_t, \quad \varepsilon_t = \sigma_t z_t \\
                \sigma_t^2 &= \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2
            \end{align*}
            \vspace{-0.4cm}
            \item \textbf{Funcția $g$}: poate fi $\sigma_t^2$, $\sigma_t$, sau $\ln(\sigma_t^2)$
        \end{itemize}
    \end{defn}
    \vspace{-0.25cm}
    \begin{block}{Interpretare economică}
        \begin{itemize}
            \item $\delta > 0$: \textbf{prima de risc} $\Rightarrow$ randamente mai mari când volatilitatea este ridicată
            \item Formalizează relația risc-randament (CAPM, Merton ICAPM); testul $H_0: \delta = 0$ verifică semnificația
        \end{itemize}
    \end{block}
    \vspace{-0.15cm}
    \begin{exampleblock}{Exemplu tipic: acțiuni}
        \begin{itemize}
            \item $r_t = 0.02 + \underbrace{0.15}_{\delta} \cdot \sigma_t + \varepsilon_t$ \quad $\Rightarrow$ La $\sigma_t = 2\%$: $\E[r_t] = 0.023$ (0.3\% primă)
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{GARCH-M: Specificații alternative}
    \small
    \begin{block}{Specificații comune}
        \begin{itemize}
            \item Prima de risc poate intra sub diferite forme:
            \begin{itemize}
                \item (1) $r_t = \mu + \lambda \sigma_t + \varepsilon_t$
                \item (2) $r_t = \mu + \lambda \sigma_t^2 + \varepsilon_t$
                \item (3) $r_t = \mu + \lambda \ln(\sigma_t^2) + \varepsilon_t$
            \end{itemize}
        \end{itemize}
    \end{block}
    \vspace{-0.1cm}
    \begin{exampleblock}{Rezultate tipice pentru piețele de acțiuni}
        \begin{itemize}
            \item $\lambda$ estimat adesea pozitiv dar mic (0.01--0.10)
            \item Semnificația variază în funcție de piață și perioadă
            \item Specificația cu varianță produce estimări $\lambda$ mai mari
        \end{itemize}
    \end{exampleblock}
    \vspace{-0.1cm}
    \begin{rmk}
        \begin{itemize}
            \item GARCH-M este utilizat în evaluarea activelor, optimizarea portofoliului și testarea CAPM
        \end{itemize}
    \end{rmk}
\end{frame}

\begin{frame}{Workflow complet: de la date la VaR validat}
    \small
        \vspace{-0.2cm}
        \begin{block}{Algoritmul practic în 8 pași}
            \begin{enumerate}\setlength{\itemsep}{1pt}
                \item \textbf{Estimează ecuația mediei}: ARMA sau $r_t = \mu + \varepsilon_t$ (verifică ACF randamente)
                \item \textbf{Testează efecte ARCH}: ARCH-LM test + Ljung-Box pe $\hat{\varepsilon}_t^2$
                \item \textbf{Estimează familia GARCH}: GARCH(1,1), EGARCH(1,1), GJR-GARCH(1,1)
                \item \textbf{Compară distribuții ale inovațiilor}: Normal vs Student-t vs GED
                \item \textbf{Verifică reziduurile standardizate}: $\hat{z}_t = \hat{\varepsilon}_t / \hat{\sigma}_t$ i.i.d. (Ljung-Box, ARCH-LM rezidual, Q-Q plot)
                \item \textbf{Compară modele cu AIC/BIC}: cel mai mic AIC/BIC = cel mai bun compromis
                \item \textbf{Prognozează volatilitatea}: $\hat{\sigma}_{T+h}^2$ cu rolling window sau recursive
                \item \textbf{Validează VaR}: backtesting cu Kupiec (acoperire) și Christoffersen (independență)
            \end{enumerate}
        \end{block}
        \vspace{-1.5mm}
        \begin{alertblock}{De reținut}
            \begin{itemize}
                \item Acesta este \textbf{exact} fluxul de lucru pe care îl veți urma în studiile de caz S\&P 500 și Bitcoin
                \item \textbf{Chiar dacă randamentele sunt greu de prognozat, volatilitatea lor este adesea prognozabilă}
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Selectarea ordinului}
    \small
        \begin{block}{Criterii informaționale}
            \begin{itemize}
                \item \textbf{AIC} (Akaike Information Criterion) = $-2\ell + 2k$
                \item \textbf{BIC} (Bayesian Information Criterion) = $-2\ell + k \ln(T)$
                \item \textbf{HQIC} (Hannan-Quinn Information Criterion) = $-2\ell + 2k \ln(\ln(T))$
            \end{itemize}
            unde $\ell$ = maximul log-verosimilității, $k$ = nr.\ parametri.
        \end{block}

        \vspace{0.3cm}

        \begin{alertblock}{Recomandări practice}
            \begin{itemize}
                \item GARCH(1,1) este suficient în \textbf{90\% din cazuri}
                \item Verificați dacă modelul asimetric îmbunătățește semnificativ ajustarea
                \item Alege distribuția inovațiilor care minimizează AIC/BIC
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Exemplu diagnostic}
    \small
    \vspace{-0.2cm}
    \begin{block}{Verificare}
        \begin{itemize}
            \item Reziduurile standardizate trebuie să fie i.i.d. fără efecte ARCH reziduale
        \end{itemize}
    \end{block}
        \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.50\textheight, keepaspectratio]{garch_diagnostics.pdf}
    \end{center}
    \quantlet{TSA\_ch5\_diagnostic}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_diagnostic}
\end{frame}

\begin{frame}{Diagnosticarea modelelor GARCH}
    \small
        \begin{block}{Reziduuri standardizate}
            \[
                \hat{z}_t = \frac{\hat{\varepsilon}_t}{\hat{\sigma}_t}
            \]
            \begin{itemize}
                \item Dacă modelul este corect specificat, $\hat{z}_t$ ar trebui să fie i.i.d.(0,1)
            \end{itemize}
        \end{block}

        \vspace{0.3cm}

        \begin{block}{Verificări diagnostic}
            \begin{enumerate}
                \item \textbf{Ljung-Box pe $\hat{z}_t$}: $Q(m) = T(T+2)\sum_{k=1}^{m}\frac{\hat{\rho}_k^2}{T-k} \sim \chi^2(m)$;\; $H_0$: fără autocorelație; $H_1$: autocorelație prezentă
                \item \textbf{Ljung-Box pe $\hat{z}_t^2$}: aceeași formulă aplicată pe $\hat{z}_t^2$;\; $H_0$: fără efecte ARCH reziduale
                \item \textbf{Test ARCH-LM pe $\hat{z}_t$}: confirmă absența heteroscedasticității
                \item \textbf{Sign-bias test} (Engle \& Ng, 1993): verifică asimetria neexploatată
                \item \textbf{Histogramă + QQ-plot}: verifică distribuția asumată
            \end{enumerate}
        \end{block}
\end{frame}

\begin{frame}{Testul Sign-Bias --- Engle \& Ng (1993)}
    \small
        \begin{block}{Motivație}
            \begin{itemize}
                \item Verifică dacă modelul GARCH captează \textbf{corect asimetria} în volatilitate
                \item Dacă testul respinge, modelul simetric este inadecvat $\Rightarrow$ trebuie EGARCH sau GJR
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{defn}[Regresii Sign-Bias]
            Fie $\hat{z}_t = \hat{\varepsilon}_t / \hat{\sigma}_t$ și $S_t^- = \mathbf{1}(\hat{\varepsilon}_t < 0)$. Se estimează:
            \vspace{-2mm}
            \[
                \hat{z}_t^2 = c_0 + c_1 S_{t-1}^- + c_2 S_{t-1}^- \hat{\varepsilon}_{t-1} + c_3 (1 - S_{t-1}^-) \hat{\varepsilon}_{t-1} + u_t
            \]
            \vspace{-2mm}
            \begin{itemize}\setlength{\itemsep}{-1pt}
                \item \textbf{Sign-bias}: $H_0: c_1 = 0$; \; \textbf{Neg.\ size-bias}: $H_0: c_2 = 0$; \; \textbf{Pos.\ size-bias}: $H_0: c_3 = 0$
                \item \textbf{Test joint}: $H_0: c_1 = c_2 = c_3 = 0$, statistica $TR^2 \sim \chi^2(3)$
            \end{itemize}
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Interpretare practică}
            \begin{itemize}
                \item Respingere $\Rightarrow$ modelul nu captează leverage $\Rightarrow$ treceți la EGARCH/GJR
                \item Acceptare pe model asimetric $\Rightarrow$ asimetria bine specificată
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Testul de stabilitate Nyblom (1989)}
    \small
        \begin{block}{Motivație}
            \begin{itemize}
                \item Testează dacă parametrii GARCH sunt \textbf{constanți pe întreaga perioadă}
                \item Instabilitatea parametrilor distorsionează persistența: $\hat{\alpha} + \hat{\beta}$ umflat artificial
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{defn}[Testul Nyblom]
            $H_0$: $\boldsymbol{\theta} = (\omega, \alpha, \beta, \ldots)$ constanți; $H_1$: $\boldsymbol{\theta}_t$ urmează random walk.
            \vspace{-2mm}
            \[
                L_c = \frac{1}{T^2} \sum_{t=1}^T \mathbf{S}_t' \hat{\mathbf{V}}^{-1} \mathbf{S}_t, \qquad \mathbf{S}_t = \sum_{s=1}^t \frac{\partial \ell_s}{\partial \boldsymbol{\theta}}\bigg|_{\hat{\boldsymbol{\theta}}}
            \]
            \vspace{-2mm}

            $\mathbf{S}_t$ = suma parțială a scorurilor, $\hat{\mathbf{V}}$ = matricea de informație estimată.
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Utilizare practică}
            \begin{itemize}
                \item Disponibil în \texttt{arch} Python: \texttt{result.diagnostic\_summary()}
                \item Respingere $\Rightarrow$ estimare rolling window sau GARCH cu schimbare de regim
                \item Se pot testa parametri \textbf{individual} (care parametru este instabil?)
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Comparație modele GARCH -- validare}
    \small
        \begin{block}{Interpretare}
            \begin{itemize}
                \item GARCH(1,1) obține cel mai mic MAE pe setul de validare
                    \begin{itemize}
                        \item Mai parcimonios și mai stabil decât modelele de ordin mai mare

                \item GARCH(2,1) și GJR-GARCH: performanță similară, dar mai mulți parametri
                \item \textbf{Concluzie}: simplitatea câștigă $\Rightarrow$ GARCH(1,1) este greu de bătut
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{Testul Diebold-Mariano}
    \small
        \setlength{\abovedisplayskip}{1pt}\setlength{\belowdisplayskip}{1pt}
        \vspace{-0.4cm}
        \begin{block}{Motivație}
            \begin{itemize}
                \item Date fiind două modele de volatilitate concurente, care prognozează mai bine?
            \end{itemize}
        \end{block}
        \vspace{-0.3cm}
        \begin{defn}[Diebold-Mariano (1995)]
            Diferența de pierdere: $d_t = L(e_{1t}) - L(e_{2t})$, unde $e_{it} = \sigma_t^2 - \hat{\sigma}_{it}^2$
            \[
                DM = \frac{\bar{d}}{\sqrt{\hat{V}(\bar{d})}} \xrightarrow{d} N(0,1) \quad \text{sub } H_0: \; \E[d_t] = 0
            \]
        \end{defn}
        \vspace{-0.35cm}
        \begin{alertblock}{Note practice}
            \begin{itemize}\setlength{\itemsep}{-1pt}
                \item $H_0$: acuratețe predictivă egală; respingere $\Rightarrow$ un model semnificativ mai bun
                \item Funcții de pierdere: MSE, QLIKE (robustă, Patton 2011); SE HAC dacă $d_t$ autocorelat
            \end{itemize}
        \end{alertblock}
        \vspace{-0.35cm}
        \begin{exampleblock}{Exemplu: S\&P 500 (2015--2024, rolling 500 zile, QLIKE)}
            \begin{itemize}\setlength{\itemsep}{-1pt}
                \item EGARCH-t vs GARCH-N: $\bar{d}\!=\!-0.047$, $DM\!=\!-3.12$, p $=0.002$ $\Rightarrow$ respinge $H_0$
                \item EGARCH-t vs GJR-t: $DM\!=\!-0.84$, p $=0.40$ $\Rightarrow$ nu respinge: acuratețe echivalentă
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Funcții de pierdere pentru evaluarea prognozei volatilității}
    \small
        \begin{block}{De ce contează funcția de pierdere?}
            \begin{itemize}
                \item Volatilitatea reală $\sigma_t^2$ este \textbf{neobservabilă} --- folosim proxy-uri ($r_t^2$, $RV_t$)
                \item Funcția de pierdere trebuie să fie \textbf{consistentă}: clasamentul nu depinde de proxy (Patton, 2011)
            \end{itemize}
        \end{block}
        \vspace{-2mm}
        \begin{defn}[Funcții de pierdere standard]
            \vspace{-3mm}
            \begin{align*}
                L_{\text{MSE}} &= (\sigma_t^2 - \hat{\sigma}_t^2)^2 & \text{(sensibilă la extreme)}\\
                L_{\text{QLIKE}} &= \ln \hat{\sigma}_t^2 + \sigma_t^2 / \hat{\sigma}_t^2 & \text{(robustă, consistentă --- \textbf{recomandată})}\\
                L_{\text{MAE}} &= |\sigma_t - \hat{\sigma}_t| & \text{(nu este consistentă pentru $\sigma^2$)}
            \end{align*}
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Recomandare}
            \begin{itemize}
                \item \textbf{QLIKE}: robustă la proxy zgomotos; disponibilă în DM și MCS
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Model Confidence Set --- Hansen, Lunde \& Nason (2011)}
    \small
        \begin{block}{Motivație: de la comparație pereche la comparație multiplă}
            \begin{itemize}
                \item Diebold-Mariano compară \textbf{2 modele}; ce facem cu 5, 10, sau 20 candidate?
                \item MCS: identifică \textbf{setul de modele echivalente statistic} (``cele mai bune'')
            \end{itemize}
        \end{block}
        \vspace{-1mm}
        \begin{defn}[Model Confidence Set (MCS)]
            Pornind de la $\mathcal{M}_0 = \{M_1, \ldots, M_m\}$ (toate modelele candidate):
            \begin{enumerate}\setlength{\itemsep}{-1pt}
                \item Testează $H_0$: toate modelele din set au acuratețe predictivă egală
                \item Dacă respinsă: elimină modelul cel mai slab ($t_{\max}$ sau $T_R$)
                \item Repetă până $H_0$ nu mai este respinsă $\Rightarrow$ $\hat{\mathcal{M}}^*_{1-\alpha}$
            \end{enumerate}
        \end{defn}
        \vspace{-1mm}
        \begin{alertblock}{Avantaj față de Diebold-Mariano}
            \begin{itemize}\setlength{\itemsep}{-1pt}
                \item Controlează \textbf{eroarea de testare multiplă} (familywise error rate)
                \item Fiecare model primește un \textbf{MCS p-value}; disponibil: \texttt{arch.bootstrap.MCS}
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Regresia Mincer-Zarnowitz}
    \small
        \setlength{\abovedisplayskip}{1pt}\setlength{\belowdisplayskip}{1pt}
        \vspace{-0.4cm}
        \begin{block}{Test de eficiență a prognozei}
            \begin{itemize}
                \item Regresăm volatilitatea realizată pe volatilitatea prognozată:
                $\sigma_{t,\text{realized}}^2 = a + b \, \hat{\sigma}_t^2 + u_t$
            \end{itemize}
        \end{block}
        \vspace{-0.2cm}
        \begin{defn}[Ipoteze]
            \begin{itemize}
                \item $H_0: a = 0, \; b = 1$ (prognoză nedirecționată și eficientă)
                \item Test prin F-test sau Wald; respingere $\Rightarrow$ prognoza este biasată sau ineficientă
            \end{itemize}
        \end{defn}
        \vspace{-0.2cm}
        \begin{exampleblock}{Exemplu: S\&P 500, GARCH(1,1)-t (2015--2024)}
            \begin{itemize}
                \item $\hat{a} = 0.12$, $\hat{b} = 0.87$, $R^2 = 0.08$; F-test: $p = 0.03$ $\Rightarrow$ respinge $H_0$
                \item Prognoza ușor biasată ($a > 0$, $b < 1$); cu $RV_t$ ca proxy: $R^2 = 0.42$, $\hat{b} = 0.95$
            \end{itemize}
        \end{exampleblock}
        \vspace{-0.2cm}
        \begin{alertblock}{Atenție}
            \begin{itemize}
                \item Varianța realizată ($r_t^2$) este un proxy zgomotos; $R^2$ de regulă mic ($< 0.10$); preferați $RV_t$
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Prognoza cu GARCH(1,1)}
    \small
        \begin{block}{Prognoză un pas înainte}
            \begin{itemize}
                \item $\hat{\sigma}_{T+1}^2 = \omega + \alpha \varepsilon_T^2 + \beta \sigma_T^2$
            \end{itemize}
        \end{block}

        \begin{block}{Prognoză multi-pas}
            \begin{itemize}
                \item Pentru $h > 1$: $\E_T[\sigma_{T+h}^2] = \bar{\sigma}^2 + (\alpha + \beta)^{h-1} (\sigma_{T+1}^2 - \bar{\sigma}^2)$
                \item $\bar{\sigma}^2 = \frac{\omega}{1 - \alpha - \beta}$ = varianța necondiționată
            \end{itemize}
        \end{block}

        \vspace{0.2cm}

        \begin{exampleblock}{Convergență}
            \begin{itemize}
                \item $\lim_{h \to \infty} \E_T[\sigma_{T+h}^2] = \bar{\sigma}^2$ --- prognoza converge către varianța necondiționată.
            \end{itemize}
        \end{exampleblock}
\end{frame}

\begin{frame}{Convergența prognozei GARCH către varianța necondiționată}
    \small
        \begin{block}{Interpretare}
            \begin{itemize}
                \item Prognoza multi-pas converge exponențial către $\bar{\sigma}^2 = \frac{\omega}{1-\alpha-\beta}$
                \item Cu cât $\alpha + \beta$ este mai aproape de 1, cu atât convergența este mai lentă
                    \begin{itemize}
                        \item S\&P 500: $\alpha + \beta \approx 0.99$ $\Rightarrow$ convergență în $\sim$50 zile
                        \item Bitcoin: $\alpha + \beta \approx 0.95$ $\Rightarrow$ convergență mai rapidă
                    \end{itemize}
            \end{itemize}
        \end{block}
\end{frame}

\begin{frame}{Volatilitate realizată și HAR-RV}
    \small
        \setlength{\abovedisplayskip}{1pt}\setlength{\belowdisplayskip}{1pt}
        \vspace{-0.4cm}
        \begin{block}{Volatilitate realizată (RV)}
            \begin{itemize}
                \item Cu randamente intra-zilnice $r_{t,i}$ ($M$ obs./zi): $RV_t = \sum_{i=1}^{M} r_{t,i}^2$ --- măsură \textbf{fără model}, \textbf{observabilă} (vs.\ GARCH: latentă)
            \end{itemize}
        \end{block}
        \vspace{-0.25cm}
        \begin{defn}[Modelul HAR-RV --- Corsi (2009)]
            \[
                RV_t = \beta_0 + \beta_D \, RV_{t-1} + \beta_W \, RV_{t-1}^{(w)} + \beta_M \, RV_{t-1}^{(m)} + u_t
            \]
            \vspace{-0.3cm}
            unde $RV^{(w)} = \frac{1}{5}\sum_{j=1}^{5} RV_{t-j}$ (săptămânală) și $RV^{(m)} = \frac{1}{22}\sum_{j=1}^{22} RV_{t-j}$ (lunară).
        \end{defn}
        \vspace{-0.25cm}
        \begin{alertblock}{Avantaj principal}
            \begin{itemize}
                \item Regresie OLS simplă care capturează \textbf{memoria lungă}; traderi eterogeni explică dinamica multi-scală
                \item Adesea depășește GARCH când datele intra-zilnice sunt disponibile (5 min, $M \approx 78$/zi)
                \item Atenție: zgomot de microstructură la frecvență foarte înaltă $\Rightarrow$ estimatori kernel
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Realized GARCH}
    \small
        \vspace{-0.4cm}
        \begin{block}{Motivație}
            \begin{itemize}
                \item GARCH standard tratează volatilitatea ca \textbf{latentă}; cu date de frecvență înaltă, o putem \textbf{observa} prin $RV_t$
            \end{itemize}
        \end{block}
        \vspace{-2.5mm}
        \begin{defn}[Realized GARCH --- Hansen, Huang \& Shek (2012)]
            \textbf{Randamente}: $r_t = \mu + \sigma_t z_t$, $z_t \sim D(0,1)$ \quad
            \textbf{GARCH}: $\log \sigma_t^2 = \omega + \beta \log \sigma_{t-1}^2 + \gamma \log RV_{t-1}$

            \textbf{Măsurare}: $\log RV_t = \xi + \varphi \log \sigma_t^2 + \tau(z_t) + u_t$
        \end{defn}
        \vspace{-2.5mm}
        \begin{exampleblock}{Beneficii}
            \begin{itemize}
                \item \textbf{Model comun}: leagă randamentele, volatilitatea latentă și măsurile realizate
                \item Funcția de levier $\tau(z_t)$ capturează asimetria; prognoze \textbf{mai precise} decât GARCH standard
            \end{itemize}
        \end{exampleblock}
        \vspace{-2.5mm}
        \begin{alertblock}{Comparație: GARCH vs HAR-RV vs Realized GARCH}
            \begin{itemize}
                \item \textbf{GARCH}: nu necesită date intra-zilnice; cel mai utilizat
                \item \textbf{HAR-RV}: simplu, fără model; necesită date la 5 min
                \item \textbf{Realized GARCH}: combină avantajele; necesită date la 5 min + structură parametrică
            \end{itemize}
        \end{alertblock}
\end{frame}

                \item \textbf{Atenție}: Sub modelele GARCH, randamentele \textbf{nu} sunt i.i.d.\ $\Rightarrow$ regula $\sqrt{h}$ este doar o \textbf{aproximare}; pentru VaR multi-perioadă, folosiți prognoza GARCH iterată

\begin{frame}{Value at risk --- distribuție Student-t}
    \small
        \begin{block}{De ce Student-t?}
            \begin{itemize}
                \item Distribuția normală \textbf{subestimează} riscul de coadă
                \item Student-t cu $\nu$ grade de libertate capturează mai bine cozile groase (kurtosis $> 3$)
            \end{itemize}
        \end{block}

        \begin{exampleblock}{Comparație VaR 1\% (1 zi): $\sigma = 1.5\%$, portofoliu = 1M EUR}
            \begin{center}
                \begin{tabular}{lcc}
                    \toprule
                    \textbf{Distribuție} & \textbf{Cuantilă} & \textbf{VaR (EUR)} \\
                    \midrule
                    Normal & 2.326 & 34.890 \\
                    Student-t ($\nu = 6$) & 3.143 & 47.145 \\
                    Student-t ($\nu = 4$) & 3.747 & 56.205 \\
                    \bottomrule
                \end{tabular}
            \end{center}
        \end{exampleblock}

        \begin{alertblock}{Observație}
            \begin{itemize}
                \item Cu $\nu = 6$ (tipic pentru acțiuni), VaR este cu \textbf{35\% mai mare} decât cel normal.
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{VaR --- exemplu complet cu GARCH}
    \small
        \begin{block}{Procedura de calcul VaR}
            \begin{enumerate}
                \item Estimează modelul GARCH(1,1) cu distribuție Student-t
                \item Obține prognoza volatilității: $\hat{\sigma}_{T+1}$
                \item Calculează VaR: $\text{VaR}_\alpha = t_\alpha(\nu) \cdot \hat{\sigma}_{T+1} \cdot \sqrt{\frac{\nu-2}{\nu}}$
            \end{enumerate}
        \end{block}

        \vspace{0.2cm}

        \begin{exampleblock}{Exemplu: S\&P 500}
            \begin{itemize}
                \item Parametri estimați: $\alpha = 0.088$, $\beta = 0.900$, $\nu = 6.4$
                \item Volatilitate prognozată: $\hat{\sigma}_{T+1} = 1.2\%$
                \item Portofoliu: 10.000.000 EUR
            \end{itemize}

            \vspace{0.2cm}

            \textbf{VaR 1\% (1 zi):}
            $\text{VaR} = 3.05 \times 0.012 \times 10{,}000{,}000 = \textbf{366{,}000 EUR}$
        \end{exampleblock}
\end{frame}

\begin{frame}{Ce este VaR backtesting?}
    \small
        \begin{block}{Definiție}
            \begin{itemize}
                \item \textbf{Backtesting} = verificarea ex-post a calității modelului VaR
                \item Compară pierderile realizate cu pragul VaR prognozat
                    \begin{itemize}
                        \item O \textbf{încălcare} (violation) apare când $r_t < -\text{VaR}_t$
                    \end{itemize}
            \end{itemize}
        \end{block}
        \vspace{-0.2cm}
        \begin{block}{Principiul Backtesting-ului}
            \begin{itemize}
                \item Indicatorul de încălcare: $I_t = \mathbf{1}(r_t < -\text{VaR}_{\alpha,t})$
                \item Pentru un model corect la nivel $\alpha$:
                    \begin{itemize}
                        \item Frecvența: $\hat{p} = \frac{1}{T}\sum I_t \approx \alpha$; încălcări \textbf{independente}
                    \end{itemize}
                \item VaR 1\% pe 250 zile $\Rightarrow$ așteptăm $\sim$2.5 încălcări/an
            \end{itemize}
        \end{block}
        \vspace{-0.2cm}
        \begin{alertblock}{Importanță}
            \begin{itemize}
                \item Cerință regulamentară \textbf{Basel III/IV} pentru bănci: backtesting obligatoriu
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Metodologia Rolling Window pentru VaR}
    \small
        \vspace{-0.2cm}
        \begin{block}{Conceptul Rolling Window}
            \begin{itemize}
                \item O \textbf{fereastră mobilă} de dimensiune fixă $W$ (ex. 500 zile) se deplasează zi cu zi
                \item La fiecare pas $t$: re-estimare GARCH pe $[t-W,\; t-1]$, prognoză $\hat{\sigma}_{t|t-1}$, calcul VaR$_t$
            \end{itemize}
        \end{block}
        \vspace{-0.2cm}
        \begin{exampleblock}{Procedura pas cu pas (pentru fiecare zi $t = W+1, \ldots, T$)}
            \begin{enumerate}
                \item Estimează GARCH pe $\{r_{t-W}, \ldots, r_{t-1}\}$ $\Rightarrow$ parametri $\hat{\omega}, \hat{\alpha}, \hat{\beta}, \hat{\nu}$
                \item Prognozează: $\hat{\sigma}_{t|t-1}^2 = \hat{\omega} + \hat{\alpha} r_{t-1}^2 + \hat{\beta} \hat{\sigma}_{t-1}^2$
                \item Calculează: $\text{VaR}_{\alpha,t} = -t_{\alpha}(\hat{\nu}) \cdot \sqrt{\frac{\hat{\nu}-2}{\hat{\nu}}} \cdot \hat{\sigma}_{t|t-1}$
                \item Verifică încălcarea: $I_t = \mathbf{1}(r_t < -\text{VaR}_{\alpha,t})$
            \end{enumerate}
        \end{exampleblock}
        \vspace{-0.2cm}
        \begin{alertblock}{Rolling vs Expanding window}
            \begin{itemize}
                \item \textbf{Rolling} ($W$ fix): reflectă regimul curent; recomandat dacă sunt schimbări structurale
                \item \textbf{Expanding}: folosește toate datele; mai stabil, dar mai lent la adaptare
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{VaR backtesting: testul Kupiec}
    \small
    \begin{block}{Testul acoperirii necondiționate}
        \begin{itemize}
            \item Testează dacă rata de încălcare observată este egală cu rata așteptată $p$ (ex.\ 1\% pentru VaR 1\%)
            \item Fie $N$ = numărul de încălcări VaR, $T$ = totalul observațiilor, $\hat{p} = N/T$
            \item \textbf{Statistica Likelihood Ratio:}
            \[
                LR_{uc} = -2\ln\left[\frac{(1-p)^{T-N} p^N}{(1-\hat{p})^{T-N} \hat{p}^N}\right] \sim \chi^2(1)
            \]
        \end{itemize}
    \end{block}

    \vspace{0.2cm}

    \begin{alertblock}{Ipoteze}
        \begin{itemize}
            \item $H_0$: $\hat{p} = p$ (modelul VaR este corect calibrat)
            \item $H_1$: $\hat{p} \neq p$ (modelul VaR sub- sau supra-estimează riscul)
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{VaR backtesting: testul Christoffersen}
    \small
        \vspace{-0.3cm}
        \begin{prop}[Proprietăți ale unui model VaR corect calibrat (Christoffersen, 1998)]
            \begin{itemize}
                \item \textbf{Acoperire necondiționată}: frecvența încălcărilor = nivelul nominal $\alpha$
                \item \textbf{Independență}: încălcările trebuie să fie serial independente (fără clusterizare)
            \end{itemize}
        \end{prop}
        \vspace{-0.1cm}
        \begin{block}{Componente ale testului}
            \begin{itemize}
                \item \textbf{Testul de independență} ($LR_{ind}$): Testează dacă încălcările sunt serial independente
                \item \textbf{Acoperire condiționată}: $LR_{cc} = LR_{uc} + LR_{ind} \sim \chi^2(2)$
            \end{itemize}
        \end{block}
        \vspace{-0.1cm}
        \begin{alertblock}{Interpretare}
            \begin{itemize}
                \item Respinge $LR_{uc}$: frecvență greșită
                \item Respinge $LR_{ind}$: încălcări clusterizate
                \item Respinge $LR_{cc}$: modelul nu este validat
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Backtesting complet -- rezultate și decizie}
    \small
    \begin{block}{Aplicare S\&P 500 (T=500, VaR 1\%)}
        \begin{itemize}
            \item Rolling window: $W = 500$ zile, model EGARCH(1,1)-t
            \item Fereastră de test: $T - W$ zile out-of-sample
            \item Se compară VaR prognozat cu randamentul realizat în fiecare zi
        \end{itemize}
    \end{block}
    \begin{exampleblock}{Output tipic}
        \begin{itemize}
            \item Număr încălcări vs așteptat $\Rightarrow$ zona Basel (verde/galben/roșu)
            \item Kupiec (frecvență): $H_0$: $\hat{p} = \alpha$ $\Rightarrow$ p-value $> 0.05$ $\Rightarrow$ acceptat
            \item Christoffersen (independență): $H_0$: încălcări i.i.d. $\Rightarrow$ p-value $> 0.05$ $\Rightarrow$ acceptat
            \item Test combinat: ambele condiții îndeplinite $\Rightarrow$ \textcolor{Forest}{\textbf{model validat}}
        \end{itemize}
    \end{exampleblock}
    \quantlet{TSA\_ch5\_backtest\_full}{https://github.com/QuantLet/TSA/tree/main/TSA_ch5/TSA_ch5_backtest_full}
\end{frame}

\begin{frame}{Backtesting-ul Expected Shortfall}
    \small
        \setlength{\abovedisplayskip}{1pt}\setlength{\belowdisplayskip}{1pt}
        \vspace{-0.4cm}
        \begin{block}{De ce backtesting ES?}
            \begin{itemize}
                \item VaR măsoară doar \textit{pragul} pierderilor extreme, nu \textit{magnitudinea} lor
                \item ES = pierderea medie \textit{dacă} VaR este depășit $\Rightarrow$ mai informativ pentru riscul de coadă
            \end{itemize}
        \end{block}
        \vspace{-0.25cm}
        \begin{defn}[Testul Z Acerbi-Sz\'ekely (2014)]
            $Z_{\text{ES}} = \frac{1}{n_\alpha} \sum_{t: r_t < -\text{VaR}_t} \frac{r_t}{\text{ES}_t} + 1$.\quad $H_0$: model ES corect ($\E[Z_{\text{ES}}] = 0$); $H_1$: ES subestimat ($Z_{\text{ES}} < 0$); se respinge dacă $Z_{\text{ES}} < z_{\alpha}$
        \end{defn}
        \vspace{-1.5mm}
        \begin{exampleblock}{Exemplu: S\&P 500, ES 2.5\%, EGARCH-t (2015--2024, T=2015)}
            \begin{itemize}
                \item $n_\alpha = 51$ încălcări VaR, $Z_{\text{ES}} = -0.38$, p-value $= 0.35$ $\Rightarrow$ nu respinge $H_0$: ES corect
                \item Sub GARCH-Normal: $Z_{\text{ES}} = -1.92$, p-value $= 0.03$ $\Rightarrow$ respinge: ES subestimat (cozi prea subțiri)
            \end{itemize}
        \end{exampleblock}
        \vspace{-1.5mm}
        \begin{alertblock}{Observație}
            \begin{itemize}
                \item Spre deosebire de backtesting VaR (binar), backtesting ES folosește \textit{magnitudinea} pierderilor
                \item Necesită eșantioane mari ($T > 1000$); alternativă: bootstrap McNeil \& Frey (2000)
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Basel IV: de la VaR la ES}
    \small
        \vspace{-0.3cm}
        \begin{block}{Schimbarea reglementară}
            \begin{itemize}
                \item \textbf{Basel II/III}: Capital bazat pe VaR 99\% (10 zile)
                \item \textbf{Basel IV} (FRTB, 2023+): Capital bazat pe \textbf{ES 97.5\%} (10 zile)
                \item Motivație: ES este \textbf{coerent} (subaditivitate), VaR nu este
            \end{itemize}
        \end{block}
        \vspace{-0.15cm}
        \begin{exampleblock}{Echivalență sub normalitate}
            \[
                \text{ES}_{97.5\%}^{\text{Normal}} = \frac{\phi(z_{0.025})}{0.025} \cdot \sigma \approx 2.338\,\sigma \quad \approx \quad \text{VaR}_{99\%}^{\text{Normal}} = 2.326\,\sigma
            \]
            \vspace{-0.3cm}
            \begin{itemize}
                \item Sub normalitate, ES 97.5\% $\approx$ VaR 99\%; sub cozi groase, ES $\gg$ VaR
            \end{itemize}
        \end{exampleblock}
        \vspace{-0.15cm}
        \begin{alertblock}{Implicație practică}
            \begin{itemize}
                \item Modelele GARCH Student-$t$ devin \textbf{esențiale} pentru estimarea precisă a ES
                \item Băncile trebuie să modeleze \textit{întreaga coadă}, nu doar o singură cuantilă
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{ARMA-GARCH: modelarea combinată a mediei și varianței}
    \small
        \begin{block}{De ce modelare combinată?}
            \begin{itemize}
                \item \textbf{Corelație serială} $\Rightarrow$ ARMA pentru medie
                \item \textbf{Volatility clustering} $\Rightarrow$ GARCH pentru varianță
            \end{itemize}
        \end{block}
        \vspace{-0.1cm}
        \begin{defn}[ARMA(p,q)-GARCH(r,s)]
            \textbf{Ecuația mediei}:
            \vspace{-2mm}
            $$r_t = \mu + \sum_{i=1}^{p} \phi_i (r_{t-i} - \mu) + \sum_{j=1}^{q} \theta_j \varepsilon_{t-j} + \varepsilon_t$$
            \vspace{-2mm}

            \textbf{Ecuația varianței}: $\varepsilon_t = \sigma_t z_t$,
            \vspace{-2mm}
            $$\sigma_t^2 = \omega + \sum_{i=1}^{r} \alpha_i \varepsilon_{t-i}^2 + \sum_{j=1}^{s} \beta_j \sigma_{t-j}^2$$
            \vspace{-2mm}
        \end{defn}
\end{frame}

\begin{frame}{ARMA-GARCH: strategie de selecție a modelului}
    \small
        \begin{block}{Abordare pas cu pas}
            \begin{enumerate}
                \item \textbf{Identifică modelul pentru medie}: Verifică ACF/PACF al randamentelor
                \item \textbf{Testează efectele ARCH}: Aplică testul ARCH-LM pe reziduuri
                \item \textbf{Specifică modelul pentru varianță}: De obicei GARCH(1,1) este suficient
                \item \textbf{Estimare combinată}: Estimează ambele ecuații prin MLE
                \item \textbf{Diagnostic}: Reziduurile standardizate trebuie să fie i.i.d.
            \end{enumerate}
        \end{block}

        \vspace{0.2cm}

        \begin{alertblock}{Specificații comune}
            \begin{itemize}
                \item \textbf{Randamente acțiuni}: AR(1)-GARCH(1,1) sau ARMA(1,1)-GARCH(1,1)
                \item \textbf{Cursuri de schimb}: Adesea doar GARCH(1,1) (fără dinamică în medie)
                \item \textbf{Rate ale dobânzii}: AR(1)-EGARCH(1,1) pentru efecte de levier
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Python: Estimarea GARCH cu \texttt{arch}}
    \small
        \vspace{-3mm}
        \begin{block}{Pași principali}
    \begin{lstlisting}
from arch import arch_model
import numpy as np; from scipy.stats import norm

# Fit GARCH(1,1) with Student-t innovations
model = arch_model(returns, vol='Garch', p=1, q=1,
                   dist='StudentsT', mean='AR', lags=1)
result = model.fit(disp='off')
print(result.summary())

# Forecast 20-step ahead volatility
forecast = result.forecast(horizon=20)
sigma_forecast = np.sqrt(forecast.variance.iloc[-1])

# Compute 1-day VaR (1%) from conditional distribution
sigma_1 = sigma_forecast.iloc[0]
VaR_1pct = norm.ppf(0.01) * sigma_1  # Normal approx
    \end{lstlisting}
        \end{block}
        \vspace{-3mm}
        \textbf{Notă}: Pentru VaR Student-$t$, înlocuiți \texttt{norm.ppf} cu \texttt{scipy.stats.t.ppf(0.01, df=nu)} scalat cu $\sqrt{(\nu-2)/\nu}$.
\end{frame}

\begin{frame}{Checklist practic: modelarea GARCH (I)}
    \small
        \begin{block}{Workflow recomandat --- Pașii 1--5}
            \begin{enumerate}\setlength{\itemsep}{1pt}
                \item \textbf{Date}: Calculați randamentele logaritmice: $r_t = \ln(P_t / P_{t-1}) \times 100$
                \item \textbf{Explorare}: Verificați faptele stilizate --- kurtosis $\gg 3$, ACF($r_t^2$) semnificativ, QQ-plot
                \item \textbf{Test ARCH}: Testul Engle LM pe reziduurile unui model AR(p) --- $H_0$: fără efecte ARCH
                \item \textbf{Estimare}: Începeți cu GARCH(1,1)-t; comparați cu EGARCH, GJR-GARCH
                \item \textbf{Distribuție}: \textcolor{Crimson}{\textbf{Nu}} Normal --- folosiți \textbf{Student-$t$} (cozi groase) sau Skewed-$t$
            \end{enumerate}
        \end{block}
        \vspace{0.1cm}
        \begin{alertblock}{Observație}
            \begin{itemize}
                \item Pașii 1--3 sunt \textbf{obligatorii} înainte de orice estimare GARCH
                \item Pasul 4: GARCH(1,1) este suficient în 90\% din cazuri --- ordine mai mari rareori îmbunătățesc prognoza
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Checklist practic: modelarea GARCH (II)}
    \small
        \begin{block}{Workflow recomandat --- Pașii 6--9}
            \begin{enumerate}\setcounter{enumi}{5}\setlength{\itemsep}{1pt}
                \item \textbf{Persistență}: Verificați $\alpha + \beta$ --- dacă $\approx 1$, considerați IGARCH
                \item \textbf{Diagnostice}: Reziduuri standardizate $\hat{z}_t = r_t / \hat{\sigma}_t$ trebuie să fie i.i.d.
                \item \textbf{Selecție}: AIC/BIC + Diebold-Mariano pe forecasting out-of-sample
                \item \textbf{Aplicare}: VaR/ES cu rolling window (250--500 zile) + backtesting Kupiec/Christoffersen
            \end{enumerate}
        \end{block}
        \vspace{0.1cm}
        \begin{alertblock}{Capcane frecvente}
            \begin{itemize}
                \item \textbf{Convergență}: Dacă optimizatorul eșuează, schimbați algoritmul (\texttt{method='Nelder-Mead'}) sau valorile inițiale
                \item \textbf{Rupturi structurale}: Pe serii lungi (2000--2024), parametrii GARCH \textbf{nu sunt constanți} --- rolling windows sunt esențiale
            \end{itemize}
        \end{alertblock}
\end{frame}

                \item Leverage effect: modelele asimetrice \textbf{obligatorii} pentru acțiuni
                \item Backtesting sistematic: nu doar „arată bine'', ci \textbf{testat statistic}
                \item \textbf{Chiar dacă randamentele sunt greu de prognozat, volatilitatea lor este prognozabilă}
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Formule principale}
    \small
        \begin{block}{Modele de volatilitate}
            \begin{itemize}
                \item \textbf{ARCH(q):} $\sigma_t^2 = \omega + \sum_{i=1}^{q} \alpha_i \varepsilon_{t-i}^2$
                \item \textbf{GARCH(1,1):} $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \beta \sigma_{t-1}^2$
                \item \textbf{EGARCH:} $\ln(\sigma_t^2) = \omega + \alpha(|z_{t-1}| - \E[|z|]) + \gamma z_{t-1} + \beta \ln(\sigma_{t-1}^2)$
                \item \textbf{GJR-GARCH:} $\sigma_t^2 = \omega + \alpha \varepsilon_{t-1}^2 + \gamma \varepsilon_{t-1}^2 I_{t-1} + \beta \sigma_{t-1}^2$
            \end{itemize}
        \end{block}

        \begin{block}{Proprietăți și măsuri}
            \begin{itemize}
                \item \textbf{Var. necond.:} $\bar{\sigma}^2 = \frac{\omega}{1-\alpha-\beta}$ \quad \textbf{Half-life:} $HL = \frac{\ln(0.5)}{\ln(\alpha+\beta)}$
                \item \textbf{VaR:} $\text{VaR}_\alpha = z_\alpha \cdot \sigma_{T+1}$ \quad \textbf{Staționaritate:} $\alpha + \beta < 1$
                \item \textbf{ARCH-LM:} $LM = T \cdot R^2 \sim \chi^2(q)$
            \end{itemize}
        \end{block}
\end{frame}

\section{Rezumat}

\begin{frame}{Rezumat -- capitolul 5: Modele de volatilitate}
    \small
    \vspace{-0.2cm}
        \begin{block}{Concepte fundamentale}
            \begin{itemize}
                \item \textbf{ARCH(q)}: varianța condiționată depinde de pătratele erorilor trecute (Nobel 2003)
                \item \textbf{GARCH(p,q)}: adaugă lag-uri ale varianței pentru persistență (GARCH(1,1) în 90\% din cazuri)
                \item \textbf{EGARCH}: permite leverage effect, fără restricții de pozitivitate
                \item \textbf{GJR-GARCH/TGARCH}: captură asimetria cu variabile indicador
            \end{itemize}
        \end{block}
        \vspace{-0.2cm}
        \begin{block}{Aplicații}
            \begin{itemize}
                \item Măsurarea și prognoza riscului (VaR, ES)
                \item Prețul derivatelor, hedging dinamic, managementul portofoliului
            \end{itemize}
        \end{block}
        \vspace{-0.2cm}
        \begin{alertblock}{Sfat practic}
            \begin{itemize}
                \item Începe cu GARCH(1,1), verifică leverage effect, alege distribuția care minimizează AIC/BIC.
                    \begin{itemize}
                        \item Student-t adesea superior distribuției normale
                    \end{itemize}
            \end{itemize}
        \end{alertblock}
\end{frame}

\begin{frame}{Întrebarea 5}
    \small
        \begin{alertblock}{Întrebare computațională aplicată}
            Un model GARCH(1,1) estimat pe randamente zilnice logaritmice (\%) dă:
            $\omega = 0.05$, $\alpha = 0.15$, $\beta = 0.82$.

            \begin{enumerate}\setlength{\itemsep}{1pt}
                \item[(a)] Calculați varianța necondiționată $\bar{\sigma}^2$ și volatilitatea anualizată.
                \item[(b)] Calculați half-life-ul unui șoc de volatilitate.
                \item[(c)] Pentru un portofoliu de 2\,000\,000 EUR, calculați VaR pe 1 zi la 1\% sub:
                    \begin{itemize}
                        \item Distribuția normală
                        \item Distribuția Student-$t$ cu $\nu = 5$ grade de libertate
                    \end{itemize}
                    (Folosiți $\sigma_{T+1} = \bar{\sigma}$ pentru simplitate.)
            \end{enumerate}
        \end{alertblock}
\end{frame}

\begin{frame}{Întrebarea 5: Răspuns}
    \small
        \vspace{-0.3cm}
        \begin{exampleblock}{Rezolvare}
            \begin{itemize}
                \item[(a)] $\bar{\sigma}^2 = \frac{\omega}{1 - \alpha - \beta} = \frac{0.05}{1 - 0.97} = 1.667$ $\Rightarrow$ $\bar{\sigma} = 1.291\%$ zilnic, $\sigma_{\text{an}} = 1.291\% \times \sqrt{252} \approx 20.5\%$
                \item[(b)] $HL = \frac{\ln 0.5}{\ln(\alpha + \beta)} = \frac{-0.693}{\ln(0.97)} = \frac{-0.693}{-0.0305}$ $\approx \textbf{23 zile}$
                \item[(c)] Valoare portofoliu $V = 2{,}000{,}000$ EUR, $\sigma_{T+1} = 1.291\%$:
                    \begin{itemize}
                        \item \textbf{Normal}: $\text{VaR}_{1\%} = 2.326 \times 0.01291 \times 2{,}000{,}000 = \textbf{60\,066}$ EUR
                        \item \textbf{Student-$t$(5)}: $t_{0.01,5} = 3.365$ scalat cu $\sqrt{(\nu\!-\!2)/\nu}$: $\text{VaR}_{1\%} = \frac{3.365}{\sqrt{5/3}} \times 0.01291 \times 2\text{M}$ $= \textbf{67\,300}$ EUR (\textbf{+12\%}!)
                    \end{itemize}
            \end{itemize}
        \end{exampleblock}
        \begin{center}

\begin{frame}{Competențe dobândite}
    \small
        \begin{block}{La finalul capitolului, studentul trebuie să poată:}
            \begin{enumerate}\setlength{\itemsep}{2pt}
                \item \textbf{Recunoaște} efecte ARCH în date (ARCH-LM, Ljung-Box pe $r_t^2$, Q-Q plot)
                \item \textbf{Estima și interpreta} un GARCH(1,1) --- rolul lui $\alpha$, $\beta$, varianța necondiționată, half-life
                \item \textbf{Decide} dacă este nevoie de asimetrie (leverage effect $\Rightarrow$ EGARCH / GJR-GARCH)
                \item \textbf{Alege} o distribuție a inovațiilor (Normal vs Student-t vs GED)
                \item \textbf{Produce} un forecast de volatilitate (rolling window, convergența la $\bar{\sigma}^2$)
                \item \textbf{Calcula și valida} un VaR (backtesting Kupiec + Christoffersen)
            \end{enumerate}
        \end{block}
        \vspace{0.2cm}
        \begin{alertblock}{Mesajul central al capitolului}
            \begin{itemize}
                \item \textbf{Chiar dacă randamentele sunt greu de prognozat, volatilitatea lor este adesea prognozabilă.}
                \item Aceasta este ideea fundamentală care justifică întreaga econometrie a volatilității.
            \end{itemize}
        \end{alertblock}
\end{frame}
