\section{Motivație}
%=============================================================================

\begin{frame}{De ce analiză în domeniul frecvenței?}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Domeniul timpului vs.\ domeniul frecvenței}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item \textbf{Domeniul timpului}: descrie cum evoluează seria în timp --- ACF, ARIMA, GARCH
            \item \textbf{Domeniul frecvenței}: descrie care \textit{componente periodice} determină variația
            \item Acestea sunt \textbf{reprezentări duale} --- echivalente matematic, dar fiecare dezvăluie structuri diferite
        \end{itemize}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Când analiza în domeniul frecvenței este esențială}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Detectarea \textbf{periodicităților ascunse} (cicluri de afaceri, sezonalitate, cicluri astronomice)
            \item Înțelegerea \textbf{la ce frecvențe} se concentrează cea mai mare energie (varianță)
            \item Proiectarea \textbf{filtrelor optime} pentru extragerea trendului, ciclului sau sezonalității
            \item Analiza \textbf{co-mișcării} între serii la frecvențe specifice (coerență)
        \end{itemize}
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{Impact istoric}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Fourier (1807): orice funcție periodică poate fi descompusă în unde sinusoidale
            \item Schuster (1898): a inventat periodograma pentru detectarea periodicităților ascunse
            \item Wiener--Khintchine (anii 1930): au legat autocovarianța de densitatea spectrală
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\section{Fundamentele analizei Fourier}
%=============================================================================

\begin{frame}{Serii Fourier: Descompunerea funcțiilor periodice}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Ideea fundamentală a lui Fourier}
        {\small Orice funcție periodică $g(t)$ cu perioada $T$ poate fi scrisă ca:
        \[
            g(t) = a_0 + \sum_{k=1}^{\infty} \left[ a_k \cos\!\left(\frac{2\pi k t}{T}\right) + b_k \sin\!\left(\frac{2\pi k t}{T}\right) \right]
        \]
        unde coeficienții Fourier sunt:
        \[
            a_0 = \frac{1}{T}\int_0^T g(t)\,dt, \quad a_k = \frac{2}{T}\int_0^T g(t)\cos\!\left(\frac{2\pi k t}{T}\right)dt, \quad b_k = \frac{2}{T}\int_0^T g(t)\sin\!\left(\frac{2\pi k t}{T}\right)dt
        \]}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Terminologie cheie}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Frecvența}: $f_k = k/T$ cicluri pe unitate de timp; \textbf{Frecvența unghiulară}: $\omega_k = 2\pi f_k$ radiani pe unitate de timp
            \item \textbf{Frecvența fundamentală}: $f_1 = 1/T$; \textbf{Armonice}: $f_k = k \cdot f_1$ pentru $k = 2, 3, \ldots$
            \item \textbf{Amplitudinea}: $R_k = \sqrt{a_k^2 + b_k^2}$; \textbf{Faza}: $\phi_k = \arctan(b_k / a_k)$
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Ilustrare descompunere Fourier}
    \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.72\textheight, keepaspectratio]{ch12_fourier_decomposition.pdf}
    \end{center}
    \vspace{-3mm}
    {\small
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Semnal periodic complex (sus) descompus în componentele fundamentală + armonice
        \item Fiecare rând = o componentă de frecvență; suma reconstruiește semnalul original exact
    \end{itemize}
    }

\end{frame}

\begin{frame}{Forma exponențială complexă}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Formula lui Euler și seria Fourier complexă}
        {\small Utilizând formula lui Euler $e^{i\omega t} = \cos(\omega t) + i\sin(\omega t)$, putem scrie:
        \[
            g(t) = \sum_{k=-\infty}^{\infty} c_k \, e^{i 2\pi k t / T}, \qquad c_k = \frac{1}{T}\int_0^T g(t)\, e^{-i 2\pi k t / T}\,dt
        \]
        unde $c_k = (a_k - ib_k)/2$ pentru $k > 0$ și $c_{-k} = \overline{c_k}$ (conjugatul complex).}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{De ce notația complexă?}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Mai compactă: o singură formulă în locul termenilor separați sinus/cosinus
            \item Algebric mai simplă: convoluția devine multiplicare
            \item Standard în procesarea semnalelor, fizică și analiza modernă a seriilor de timp
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Teorema lui Parseval (Conservarea energiei)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Energia totală este conservată între reprezentări}
        {\small
        \[
            \frac{1}{T}\int_0^T |g(t)|^2\,dt = \sum_{k=-\infty}^{\infty} |c_k|^2
        \]
        Fiecare $|c_k|^2$ măsoară \textbf{contribuția frecvenței $k/T$ la varianța totală}.}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Implicație}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Energia din domeniul timpului = energia din domeniul frecvenței
            \item Acesta este fundamentul descompunerii spectrale a varianței
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Transformata Fourier Discretă (DFT)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{De la continuu la discret}
        {\small Fie date $n$ observații $y_1, y_2, \ldots, y_n$; DFT este:
        \[
            J(\omega_j) = \frac{1}{\sqrt{n}} \sum_{t=1}^{n} y_t \, e^{-i\omega_j t}, \qquad \omega_j = \frac{2\pi j}{n}, \quad j = 0, 1, \ldots, n-1
        \]
        Frecvențele $\omega_j$ se numesc \textbf{frecvențe Fourier} sau \textbf{frecvențe fundamentale}.}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Proprietăți cheie ale DFT}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Ortogonalitate}: $\sum_{t=1}^n e^{i(\omega_j - \omega_k)t} = n \cdot \mathbf{1}_{j=k}$ --- frecvențele sunt ortogonale
            \item \textbf{Inversabilitate}: $y_t = \frac{1}{\sqrt{n}} \sum_{j=0}^{n-1} J(\omega_j) \, e^{i\omega_j t}$ --- reconstrucție perfectă
            \item \textbf{Frecvența Nyquist}: $\omega_{n/2} = \pi$ --- cea mai mare frecvență detectabilă ($\geq 2$ obs.\ per ciclu)
        \end{itemize}
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{Transformata Fourier Rapidă (FFT)}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item DFT naivă: $O(n^2)$ operații. FFT (Cooley--Tukey, 1965): $O(n \log n)$
            \item Unul dintre cele mai importante algoritme ale secolului XX
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Aliasing și frecvența Nyquist}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Teorema eșantionării (Shannon--Nyquist)}
        {\small Un semnal continuu cu frecvența maximă $f_{\max}$ poate fi reconstruit perfect din eșantioane discrete dacă rata de eșantionare satisface:
        \[
            f_s > 2 f_{\max} \qquad \text{(criteriul Nyquist)}
        \]
        \textbf{Frecvența Nyquist} $f_N = f_s / 2$ este cea mai mare frecvență detectabilă.}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Aliasing: efectele subeșantionării}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Dacă un semnal conține frecvențe \textit{peste} $f_N$, acestea sunt „repliate" și apar ca frecvențe mai joase
            \item \textbf{Exemplu}: date zilnice ($f_s = 1$/zi, $f_N = 0{,}5$/zi). Un ciclu intra-zilnic la 0,7 cicluri/zi apare ca 0,3 cicluri/zi
            \item Aliasing-ul este \textbf{ireversibil} --- odată ce datele sunt eșantionate, frecvențele repliate nu pot fi separate
        \end{itemize}
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{Implicații practice}
        {\small Date lunare: $f_N = 0{,}5$ cicluri/lună $\Rightarrow$ cicluri $\geq 2$ luni. Date trimestriale: cicluri $\geq$ 6 luni.}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

    \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.65\textheight, keepaspectratio]{ch12_aliasing.pdf}

\section{Periodograma}
%=============================================================================

\begin{frame}{Definiție și interpretare}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{defn}[Periodograma]
        {\small \textbf{Periodograma} seriei $\{y_t\}_{t=1}^n$ la frecvența $\omega_j = 2\pi j / n$ este:
        \[
            I(\omega_j) = |J(\omega_j)|^2 = \frac{1}{n}\left|\sum_{t=1}^{n} y_t \, e^{-i\omega_j t}\right|^2
        \]
        Echivalent, în formă trigonometrică:
        \[
            I(\omega_j) = \frac{n}{2}\left(\hat{a}_j^2 + \hat{b}_j^2\right)
        \]
        unde $\hat{a}_j = \frac{2}{n}\sum_t y_t \cos(\omega_j t)$ și $\hat{b}_j = \frac{2}{n}\sum_t y_t \sin(\omega_j t)$.}
    \end{defn}
    \vspace{-2mm}
    \begin{exampleblock}{Interpretare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $I(\omega_j)$ măsoară \textbf{contribuția frecvenței $\omega_j$ la varianța totală a eșantionului}
            \item $I(\omega_j)$ mare $\Rightarrow$ oscilație puternică la perioada $T_j = 2\pi/\omega_j = n/j$
            \item Periodograma descompune varianța: $\frac{1}{n}\sum_{t=1}^n (y_t - \bar{y})^2 = \sum_{j=1}^{\lfloor n/2 \rfloor} I(\omega_j)$ (până la constante)
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_periodogram}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_periodogram}
\end{frame}

\begin{frame}{Periodograma și autocovarianța eșantionului}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    \begin{thm}[Relația periodogramă--autocovarianță]
        {\small Periodograma este transformata Fourier a funcției de autocovarianță a eșantionului:
        \vspace{-0.5em}
        \[
            I(\omega) = \sum_{h=-(n-1)}^{n-1} \hat{\gamma}(h) \, e^{-i\omega h}
        \]
        \vspace{-0.5em}
        unde $\hat{\gamma}(h) = \frac{1}{n}\sum_{t=1}^{n-|h|}(y_t - \bar{y})(y_{t+|h|} - \bar{y})$ este autocovarianța eșantionului la lag-ul $h$.}
    \end{thm}
    \vspace{-3mm}
    \begin{block}{Analogul de eșantion al teoremei Wiener--Khintchine}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Populație: $f(\omega) = \frac{1}{2\pi}\sum_{h=-\infty}^{\infty} \gamma(h)\,e^{-i\omega h}$ (densitate spectrală)
            \item Eșantion: $I(\omega) = \sum_{h} \hat{\gamma}(h)\,e^{-i\omega h}$ (periodogramă)
            \item Periodograma este un \textbf{estimator natural} al densității spectrale --- dar este oare unul \textit{bun}?
        \end{itemize}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Atenție: Periodograma este inconsistentă!}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item $\E[I(\omega_j)] \to f(\omega)$ când $n \to \infty$ (asimptotic nedeplasată)
            \item Dar $\Var[I(\omega_j)] \not\to 0$ --- varianța \textit{nu} scade cu mai multe date!
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_periodogram}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_periodogram}
\end{frame}

\begin{frame}{Citirea periodogramei: Tipare comune}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    {\small
    \begin{block}{Semnăturile spectrale ale proceselor comune}
    \vspace{-2mm}
    \begin{center}
    \begin{tabular}{l|l|l}
        \toprule
        \textbf{Proces} & \textbf{Forma spectrală} & \textbf{Tiparul periodogramei} \\
        \midrule
        Zgomot alb & Plat: $f(\omega) = \sigma^2/(2\pi)$ & Aproximativ uniform, fără vârfuri \\
        AR(1), $\phi > 0$ & Vârf la $\omega = 0$, descrescător & Valori mari la frecvențe joase \\
        AR(1), $\phi < 0$ & Vârf la $\omega = \pi$, descrescător & Valori mari la frecvențe înalte \\
        AR(2), rădăcini complexe & Vârf la frecvența ciclică & Vârf atenuat la frecvența ciclică \\
        Mers aleator & $f(\omega) \propto \omega^{-2}$ & Dominanță puternică la frecvențe joase \\
        Sezonier ($s = 12$) & Vârfuri la $\omega = 2\pi k/12$ & Vârfuri ascuțite la armonicele sezoniere \\
        \bottomrule
    \end{tabular}
    \end{center}
    \end{block}
    }
    \vspace{-2mm}
    \begin{exampleblock}{Regulă practică}

            \item \textbf{Spectru albastru} (putere concentrată la $\omega$ mare): dominat de zgomot, autocorelat negativ
            \item \textbf{Vârfuri ascuțite}: componente periodice deterministe sau cvasi-deterministe
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

    \quantlet{TSA\_ch12\_spectral\_density}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_spectral_density}
\end{frame}

\begin{frame}{Proprietățile statistice ale periodogramei}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Distribuția asimptotică (proces gaussian)}
        {\small Pentru un proces gaussian staționar cu densitate spectrală $f(\omega)$:
        \[
            \frac{2\,I(\omega_j)}{f(\omega_j)} \;\xrightarrow{d}\; \chi^2_2 \qquad \text{pentru } 0 < \omega_j < \pi
        \]
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\E[I(\omega_j)] \approx f(\omega_j)$ --- aproximativ nedeplasată
            \item $\Var[I(\omega_j)] \approx f(\omega_j)^2$ --- varianța este egală cu media la pătrat!
            \item Valorile periodogramei la \textbf{frecvențe Fourier diferite sunt asimptotic independente}
        \end{itemize}}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Problema inconsistenței}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Coeficientul de variație $\text{CV} = \text{SD}/\text{Media} = 1$ indiferent de dimensiunea eșantionului
            \item Adăugarea mai multor date adaugă mai multe frecvențe dar \textbf{nu} reduce zgomotul la nicio frecvență
            \item Periodograma brută \textbf{nu este un estimator consistent} al densității spectrale
            \item \textbf{Soluție}: netezirea periodogramei (Secțiunea~5)
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

    \begin{itemize}\setlength{\itemsep}{0pt}

\begin{frame}{Testul Fisher pentru periodicități ascunse}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Acel vârf este real sau doar zgomot?}
        {\small Fisher (1929) a propus testarea $H_0$: seria este zgomot alb contra $H_1$: există o periodicitate ascunsă. Statistica de test este:
        \[
            g = \frac{\max_j I(\omega_j)}{\sum_{j=1}^{\lfloor n/2 \rfloor} I(\omega_j)}
        \]
        Sub $H_0$, distribuția exactă a lui $g$ este cunoscută. Pentru $n$ mare:
        \[
            P(g > x) \approx m(1-x)^{m-1} - \binom{m}{2}(1-2x)_+^{m-1} + \cdots
        \]
        unde $m = \lfloor n/2 \rfloor$ și $(a)_+ = \max(a, 0)$.}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Interpretare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Dacă cea mai mare valoare a periodogramei depășește semnificativ pragul așteptat sub ipoteza de zgomot alb, se respinge $H_0$
            \item Poate fi aplicat iterativ: se elimină frecvența detectată, se testează din nou
            \item Alternativă modernă: testarea frecvențelor individuale folosind distribuția $\chi^2_2$
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Definiția și teorema Wiener--Khintchine}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{thm}[Teorema Wiener--Khintchine]
        {\small Fie $\{Y_t\}$ un proces slab staționar cu funcția de autocovarianță $\gamma(h)$ satisfăcând $\sum_{h=-\infty}^{\infty} |\gamma(h)| < \infty$. Atunci \textbf{funcția de densitate spectrală} (FDS) există și este dată de:
        \[
            f(\omega) = \frac{1}{2\pi} \sum_{h=-\infty}^{\infty} \gamma(h) \, e^{-i\omega h}, \qquad -\pi \leq \omega \leq \pi
        \]
        Relația \textbf{inversă} este:
        \[
            \gamma(h) = \int_{-\pi}^{\pi} f(\omega) \, e^{i\omega h}\,d\omega
        \]}
    \end{thm}
    \vspace{-2mm}
    \begin{block}{Proprietățile densității spectrale}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $f(\omega) \geq 0$ pentru orice $\omega$ (non-negativă --- reprezintă contribuția la varianță)
            \item $f(\omega) = f(-\omega)$ (simetrică pentru procese cu valori reale)
            \item $\gamma(0) = \Var(Y_t) = \int_{-\pi}^{\pi} f(\omega)\,d\omega$ (varianța totală = aria sub $f$)
            \item $f$ este \textbf{periodică} cu perioada $2\pi$, deci avem nevoie doar de $\omega \in [-\pi, \pi]$ sau $[0, \pi]$
        \end{itemize}
    \end{block}
    \end{cminipage}
    \quantlet{TSA\_ch12\_spectral\_density}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_spectral_density}
\end{frame}

\begin{frame}{Densitatea spectrală: Zgomot alb și AR(1)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Zgomot alb: $Y_t = \varepsilon_t$, $\varepsilon_t \sim \text{WN}(0, \sigma^2)$}
        {\small
        \[
            \gamma(h) = \sigma^2 \mathbf{1}_{h=0} \qquad\Longrightarrow\qquad f(\omega) = \frac{\sigma^2}{2\pi} \quad \text{(plat --- putere egală la toate frecvențele)}
        \]
        Densitatea spectrală este constantă: zgomotul alb contribuie egal la orice frecvență.}
    \end{block}
    \vspace{-2mm}
    \begin{block}{AR(1): $Y_t = \phi Y_{t-1} + \varepsilon_t$}
        {\small
        \[
            f(\omega) = \frac{\sigma^2}{2\pi} \cdot \frac{1}{|1 - \phi e^{-i\omega}|^2} = \frac{\sigma^2}{2\pi(1 - 2\phi\cos\omega + \phi^2)}
        \]
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\phi > 0$: \textbf{spectru roșu} --- vârf la $\omega = 0$, putere concentrată la frecvențe joase (similar tendințelor)
            \item $\phi < 0$: \textbf{spectru albastru} --- vârf la $\omega = \pi$, putere la frecvențe înalte (oscilatoriu)
            \item $|\phi| \to 1$: spectrul devine mai concentrat, apropiindu-se de un impuls
        \end{itemize}}
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Densitatea spectrală: MA(1) și ARMA general}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{MA(1): $Y_t = \varepsilon_t + \theta\varepsilon_{t-1}$}
        {\small
        \[
            f(\omega) = \frac{\sigma^2}{2\pi} \cdot |1 + \theta e^{-i\omega}|^2 = \frac{\sigma^2}{2\pi}(1 + 2\theta\cos\omega + \theta^2)
        \]
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\theta > 0$: mai multă putere la frecvențe joase; $\theta < 0$: mai multă putere la frecvențe înalte
            \item Forma spectrală este o modulare cosinusoidală simplă a spectrului de zgomot alb
        \end{itemize}}
    \end{block}
    \vspace{-2mm}
    \begin{block}{ARMA$(p,q)$ general: $\phi(B)Y_t = \theta(B)\varepsilon_t$}
        {\small
        \[
            f(\omega) = \frac{\sigma^2}{2\pi} \cdot \frac{|\theta(e^{-i\omega})|^2}{|\phi(e^{-i\omega})|^2} \qquad \text{(densitate spectrală rațională)}
        \]
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Polii AR $\Rightarrow$ vârfuri în spectru; zerourile MA $\Rightarrow$ minime (zerouri spectrale)
            \item Forma rațională face estimarea spectrală parametrică naturală pentru procese ARMA
        \end{itemize}}
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Teorema reprezentării spectrale}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{thm}[Reprezentarea spectrală / Reprezentarea Cram\'er]
        {\small Orice proces slab staționar $\{Y_t\}$ cu $\E[Y_t] = 0$ poate fi reprezentat ca:
        \[
            Y_t = \int_{-\pi}^{\pi} e^{i\omega t}\,dZ(\omega)
        \]
        unde $Z(\omega)$ este un proces cu incrementuri ortogonale satisfăcând:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\E[dZ(\omega)] = 0$
            \item $\E[|dZ(\omega)|^2] = dF(\omega)$ unde $F(\omega)$ este funcția de distribuție spectrală
            \item $\E[dZ(\omega_1)\,\overline{dZ(\omega_2)}] = 0$ pentru $\omega_1 \neq \omega_2$
        \end{itemize}
        }
    \end{thm}
    \vspace{-2mm}
    \begin{exampleblock}{Interpretare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Orice proces staționar este o \textbf{superpoziție de oscilații necorelate} la diferite frecvențe
            \item $dZ(\omega)$ dă amplitudinea și faza (aleatoare) la frecvența $\omega$
            \item Densitatea spectrală $f(\omega) = dF(\omega)/d\omega$ măsoară \textbf{varianța pe unitate de frecvență}
            \item Acesta este \textbf{analogul în domeniul frecvenței al descompunerii Wold}
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Problema estimării}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Obiectiv: Estimarea lui $f(\omega)$ din date}
        {\small Avem nevoie de un estimator \textbf{consistent}: atât biasul $\to 0$, cât și varianța $\to 0$ când $n \to \infty$. Două abordări principale:}
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{Metode neparametrice}
            \begin{itemize}\setlength{\itemsep}{-2pt}
                \item Netezesc periodograma brută
                \item Fără ipoteze de model
                \item \textbf{Fereastră lag} (Blackman--Tukey)
                \item \textbf{Mediere pe segmente} (Bartlett, Welch)
                \item \textbf{Multitaper} (Thomson)
                \item Controlul compromisului bias-varianță prin alegerea bandwidth-ului
            \end{itemize}
        \end{block}

        \column{0.50\textwidth}
        \begin{alertblock}{Metode parametrice}
            \begin{itemize}\setlength{\itemsep}{-2pt}
                \item Se ajustează un model, se derivă densitatea spectrală
                \item Presupune o formă specifică de model
                \item \textbf{Estimare spectrală AR}
                \item Rezoluție mai mare pentru serii scurte
                \item Risc de specificare greșită
                \item AIC/BIC pentru selecția modelului
            \end{itemize}
        \end{alertblock}
    \end{columns}
    \vspace{1mm}
    {\small \textbf{Observație cheie}: Estimarea spectrală consistentă necesită sacrificarea rezoluției în frecvență. Acesta este un compromis fundamental bias-varianță --- nu putem avea simultan rezoluție perfectă și varianță mică.}
    \end{cminipage}
    \quantlet{TSA\_ch12\_estimation}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_estimation}
\end{frame}

\begin{frame}{Periodograma netezită (Nucleul Daniell)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Ideea: Medierea valorilor periodogramei vecine}
        {\small \textbf{Periodograma netezită} la frecvența $\omega$ este:
        \[
            \hat{f}(\omega) = \sum_{k=-m}^{m} w_k \, I(\omega + \omega_k)
        \]
        unde $\{w_k\}$ sunt ponderi non-negative cu sumă 1, iar $m$ este parametrul de \textbf{bandwidth}.
        Cea mai simplă alegere este nucleul Daniell (uniform): $w_k = 1/(2m+1)$ pentru $|k| \leq m$.
        Grade de libertate echivalente: $\nu \approx 2(2m+1)$.}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Alegerea bandwidth-ului: Compromisul fundamental}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $m$ mic: bias mic, varianță mare (aproape de periodograma brută)
            \item $m$ mare: varianță mică, bias mare (supra-netezit, vârfurile sunt estompate)
            \item Aproximativ $\hat{f}(\omega) \sim f(\omega) \cdot \chi^2_\nu / \nu$
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Estimarea Blackman--Tukey (fereastră lag)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Alternativă: Ponderarea autocovarianțelor}
        {\small În loc de netezire în domeniul frecvenței, se aplică o fereastră pe autocovarianțele eșantionului:
        \[
            \hat{f}(\omega) = \frac{1}{2\pi}\sum_{h=-(M-1)}^{M-1} w(h/M)\,\hat{\gamma}(h)\,e^{-i\omega h}
        \]
        unde $w(\cdot)$ este o \textbf{fereastră lag} (nucleu) și $M$ este \textbf{punctul de trunchiere}.}
    \end{block}
    \vspace{-2mm}
    {\scriptsize
    \begin{block}{Ferestre lag comune}
    \vspace{-2mm}
    \begin{center}
    \begin{tabular}{l|l|l}
        \toprule
        \textbf{Fereastră} & \textbf{Formula $w(x)$, $|x| \leq 1$} & \textbf{Proprietăți} \\
        \midrule
        Bartlett (triunghi) & $1 - |x|$ & Simplă, spectru non-negativ \\
        Parzen & $1 - 6x^2 + 6|x|^3$ pt.\ $|x| \leq 1/2$, altfel $2(1-|x|)^3$ & Netedă, non-negativă \\
        Tukey--Hanning & $\frac{1}{2}(1 + \cos(\pi x))$ & Rezoluție bună \\
        Bartlett--Priestley & $\frac{3}{(\pi x)^2}\left(\frac{\sin(\pi x)}{\pi x} - \cos(\pi x)\right)$ & Optimală pt.\ spectre netede \\
        \bottomrule
    \end{tabular}
    \end{center}
    \end{block}
    }
    \vspace{-2mm}
    \begin{exampleblock}{Alegerea lui $M$}
        {\small Regulă practică: $M \approx 2\sqrt{n}$. Validare încrucișată sau selectori plug-in pentru alegere automată.}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Metoda Welch (medierea pe segmente)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Ideea: Împărțire și mediere}
        {\small Welch (1967): împărțirea seriei în $K$ segmente suprapuse de lungime $L$, aplicarea unei ferestre (ex.\ Hanning) pe fiecare, calcularea periodogramei fiecărui segment, apoi medierea:
        $\hat{f}_{\text{Welch}}(\omega) = \frac{1}{K}\sum_{k=1}^{K} I^{(k)}(\omega)$}
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{Avantaje}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Varianța redusă cu factorul $\approx 1/K$
                \item Ferestrele reduc scurgerea spectrală
                \item Suprapunere de 50\% este standard
            \end{itemize}
        \end{exampleblock}

        \column{0.50\textwidth}
        \begin{alertblock}{Compromis}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Rezoluție în frecvență $\Delta f = 1/L$
                \item Segmente mai scurte $\Rightarrow$ rezoluție mai slabă
                \item Segmente mai lungi $\Rightarrow$ mai puține medieri
                \item Tipic: $L = n/4$ până la $n/8$ cu suprapunere 50\%
            \end{itemize}
        \end{alertblock}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Estimarea spectrală Multitaper}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Metoda Multitaper a lui Thomson (1982)}
        {\small În loc de o singură fereastră, se folosesc $K$ \textbf{ferestre ortogonale} $\{h_k(t)\}_{k=0}^{K-1}$:
        \[
            \hat{f}_{\text{MT}}(\omega) = \frac{1}{K}\sum_{k=0}^{K-1} \left|\sum_{t=1}^{n} h_k(t)\,y_t\,e^{-i\omega t}\right|^2
        \]
        Ferestrele sunt \textbf{secvențe sferoidale prolate discrete} (DPSS / secvențe Slepian), care concentrează optim energia în banda de frecvență $[-W, W]$.}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{De ce Multitaper este superioară}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Utilizează \textbf{toate datele} (fără segmentare) --- rezoluție maximă în frecvență
            \item $K$ ferestre ortogonale furnizează $K$ estimări spectrale aproximativ independente
            \item Compromis bias-varianță aproape optim controlat de bandwidth $W$; $K = 2nW - 1$
            \item $NW = 3$--$4$ este standard ($K = 5$--$8$). Python: \texttt{scipy.signal.windows.dpss()}
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_multitaper}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_multitaper}
\end{frame}

\begin{frame}{Estimarea spectrală parametrică}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Estimarea spectrală bazată pe AR}
        {\small Dacă ajustăm un model AR($p$): $Y_t = \sum_{k=1}^p \hat{\phi}_k Y_{t-k} + \varepsilon_t$, densitatea spectrală este:
        \[
            \hat{f}_{\text{AR}}(\omega) = \frac{\hat{\sigma}^2}{2\pi\,|1 - \hat{\phi}_1 e^{-i\omega} - \cdots - \hat{\phi}_p e^{-ip\omega}|^2}
        \]
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{Avantaje}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Rezoluție înaltă chiar pentru serii scurte
                \item Estimare netedă (fără fluctuații eratice)
                \item Poate rezolva vârfuri apropiate
                \item Algoritm Yule--Walker sau Burg
            \end{itemize}
        \end{exampleblock}

        \column{0.50\textwidth}
        \begin{alertblock}{Dezavantaje}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Presupune că modelul AR este corect
                \item Poate rata caracteristici necaptate de AR
                \item Selecția ordinului $p$ este critică (AIC, BIC)
                \item Nu poate reprezenta zerouri spectrale ascuțite
            \end{itemize}
        \end{alertblock}
    \end{columns}
    \vspace{2mm}
    {\small \textbf{Recomandare practică}: Utilizați estimarea spectrală AR pentru serii scurte ($n < 100$) și când procesul subiacent este presupus a fi autoregresiv. Folosiți metode neparametrice pentru serii mai lungi sau când forma spectrală este necunoscută.}
    \end{cminipage}
\end{frame}

\begin{frame}{Comparația metodelor de estimare spectrală}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    {\scriptsize
    \begin{block}{Rezumatul comparativ al metodelor}
    \vspace{-2mm}
    \begin{center}
    \begin{tabular}{l|c|c|c|c}
        \toprule
        \textbf{Metoda} & \textbf{Rezoluție} & \textbf{Varianță} & \textbf{Ipoteze} & \textbf{Optim pentru} \\
        \midrule
        Periodogramă brută & Cea mai mare & Cea mai mare & Niciuna & Explorare inițială \\
        Netezire Daniell & Medie & Medie & $f$ local netedă & Estimare rapidă \\
        Blackman--Tukey & Medie & Mică--medie & $f$ netedă & Analiză clasică \\
        Metoda Welch & Mică--medie & Mică & Local staționare & Serii lungi, timp real \\
        Multitaper & Medie--mare & Mică & Niciuna & Standard de aur \\
        AR parametric & Foarte mare & Foarte mică & Model AR corect & Serii scurte \\
        \bottomrule
    \end{tabular}
    \end{center}
    \end{block}
    }
    \vspace{-2mm}
    \begin{exampleblock}{Flux de lucru practic}

            \item[\textcolor{MainBlue}{\textbf{2.}}] Aplicați estimarea multitaper pentru analiza principală

            \item[\textcolor{MainBlue}{\textbf{4.}}] Folosiți intervale de încredere: $\hat{f}(\omega) \cdot \chi^2_\nu / \nu$ cu $\nu$ grade de libertate efective

\section{Memorie lungă și spectrul}
%=============================================================================

\begin{frame}{Procese cu Memorie Lungă}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{ARFIMA și comportamentul spectral în vecinătatea originii}
        {\small Un proces ARFIMA$(p,d,q)$ cu parametrul de diferențiere fracționară $d \in (-0{,}5;\; 0{,}5)$:
        \[
            \phi(B)(1-B)^d Y_t = \theta(B)\varepsilon_t
        \]
        are densitatea spectrală care satisface:
        \[
            f(\omega) \sim C_f \, |\omega|^{-2d} \quad \text{când } \omega \to 0
        \]
        Pentru $0 < d < 0{,}5$: memorie lungă --- spectrul \textbf{diverge} la origine (putere nelimitată la frecvențe joase).}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Legătura cu exponentul Hurst}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Exponentul Hurst: $H = d + 1/2$, deci $H > 1/2$ indică memorie lungă
            \item Autocovarianțele descresc hiperbolic: $\gamma(h) \sim C \, |h|^{2d-1}$ (nesumabile pentru $d > 0$)
            \item Teorema Wiener--Khintchine se aplică în continuare, dar spectrul este nemărginit la $\omega = 0$
        \end{itemize}
    \end{exampleblock}
    \vspace{-2mm}
    \begin{alertblock}{Exemple în economie și finanțe}
        {\small Volatilitate realizată ($d \approx 0{,}4$), persistența inflației, spread-uri ale ratelor dobânzii, volum tranzacționat.}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Estimatori spectrali de regresie ai lui $d$}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    \begin{block}{Estimatorul GPH (Geweke \& Porter-Hudak, 1983)}
        {\small Regresați $\log I(\omega_j)$ față de $\log|2\sin(\omega_j/2)|^2$ la cele mai mici $m$ frecvențe Fourier:
        \[
            \log I(\omega_j) = \beta_0 - d \cdot \log\!\big(4\sin^2(\omega_j/2)\big) + \varepsilon_j, \qquad j = 1, \ldots, m
        \]
        unde $m = n^\alpha$ cu $\alpha \in (0,1)$, de regulă $\alpha = 0{,}5$--$0{,}8$.
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Regresie simplă pe log-periodogramă; consistent dar ineficient
            \item $\hat{d}_{\text{GPH}} \xrightarrow{d} N(d,\; \pi^2/(24m))$ în condiții de regularitate
        \end{itemize}}
    \end{block}
    \vspace{-3mm}
    \begin{exampleblock}{Estimatorul Local Whittle (Robinson, 1995)}
        {\small Maximizează funcția de verosimilitate locală Whittle la cele $m$ frecvențe inferioare:
        \[
            \hat{d}_{\text{LW}} = \arg\min_d \left[\log\!\left(\frac{1}{m}\sum_{j=1}^{m} \frac{I(\omega_j)}{\omega_j^{-2d}}\right) - \frac{2d}{m}\sum_{j=1}^{m}\log\omega_j\right]
        \]
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Mai eficient decât GPH; $\sqrt{m}(\hat{d}_{\text{LW}} - d) \xrightarrow{d} N(0,\; 1/4)$
            \item Versiuni extinse: Local Whittle exact (Shimotsu \& Phillips, 2005) tratează nestationaritatea
        \end{itemize}}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Estimarea prin funcția de verosimilitate Whittle}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Funcția de verosimilitate Whittle}
        {\small Pentru un model spectral parametric $f_\theta(\omega)$, aproximarea log-verosimilității Whittle este:
        \[
            \ell_W(\theta) = -\sum_{j=1}^{\lfloor n/2 \rfloor} \left[\log f_\theta(\omega_j) + \frac{I(\omega_j)}{f_\theta(\omega_j)}\right], \qquad \hat{\theta}_W = \arg\max_\theta \, \ell_W(\theta)
        \]}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Proprietăți asimptotice}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $\sqrt{n}(\hat{\theta}_W - \theta_0) \xrightarrow{d} N(0, \Sigma^{-1})$ unde $\Sigma = \frac{1}{4\pi}\int_{-\pi}^{\pi} \bigl(\frac{\partial \log f_\theta}{\partial \theta}\bigr)\bigl(\frac{\partial \log f_\theta}{\partial \theta}\bigr)^{\!\top} d\omega$
            \item Asimptotic echivalent cu MLE gaussian în domeniul timpului
        \end{itemize}}
    \end{exampleblock}
    \vspace{-2mm}
    \begin{alertblock}{Avantaje față de MLE în domeniul timpului}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Calcul: $O(n \log n)$ față de $O(n^2)$ pentru verosimilitatea exactă în domeniul timpului
            \item Robustețe: necesită doar specificarea corectă a spectrului, nu ipoteze complete de distribuție
            \item Modularitate: direct aplicabil pentru ARFIMA sau modele cu structuri spectrale complexe
        \end{itemize}}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Memorie lungă: Aplicație practică}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Exemplu: Persistența volatilității financiare}
        {\small
        \begin{itemize}\setlength{\itemsep}{-1pt}
            \item Volatilitatea realizată a randamentelor bursiere prezintă memorie lungă cu $d \approx 0{,}35$--$0{,}45$
            \item Spectrul diverge în vecinătatea $\omega = 0$: revenire la medie extrem de lentă
            \item Modelele ARMA standard necesită un ordin $p$ foarte ridicat pentru a aproxima acest comportament
            \item ARFIMA surprinde aceleași dinamici parsimonios cu un singur parametru $d$
        \end{itemize}
        }
    \end{block}
    \vspace{-3mm}
    \begin{exampleblock}{Flux de lucru practic}
        \begin{itemize}\setlength{\itemsep}{-1pt}
            \item[\textcolor{MainBlue}{\textbf{1.}}] Reprezentați log-periodograma: dacă este aproximativ liniară lângă origine, suspectați memorie lungă
            \item[\textcolor{MainBlue}{\textbf{2.}}] Estimați $d$ prin GPH și Local Whittle; comparați pentru robustețe
            \item[\textcolor{MainBlue}{\textbf{3.}}] Ajustați ARFIMA$(p,d,q)$; comparați AIC/BIC cu ARMA standard
            \item[\textcolor{MainBlue}{\textbf{4.}}] Folosiți verosimilitatea Whittle pentru estimare parametrică eficientă
        \end{itemize}
    \end{exampleblock}
    \vspace{-2mm}
    {\small \textbf{Atenție}: Rupturi structurale și tendințe deterministe pot imita memoria lungă în periodogramă --- testați întotdeauna ambele ipoteze (Qu, 2011; Shimotsu, 2006).}
    \end{cminipage}
    \quantlet{TSA\_ch12\_long\_memory}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_long_memory}
\end{frame}

\section{Analiza cross-spectrală}
%=============================================================================

\begin{frame}{Analiza spectrală bivariată}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{De la univariat la bivariat}
        {\small Pentru două procese conjunct staționare $\{X_t\}$ și $\{Y_t\}$, \textbf{densitatea cross-spectrală} este:
        \[
            f_{XY}(\omega) = \frac{1}{2\pi}\sum_{h=-\infty}^{\infty} \gamma_{XY}(h)\,e^{-i\omega h}
        \]
        unde $\gamma_{XY}(h) = \Cov(X_{t+h}, Y_t)$ este funcția de cross-covarianță.}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Diferența față de cazul univariat}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $f_{XY}(\omega)$ este \textbf{cu valori complexe}: $f_{XY}(\omega) = c_{XY}(\omega) - i\,q_{XY}(\omega)$
            \item $c_{XY}(\omega)$: \textbf{co-spectrul} (co-mișcare în fază la frecvența $\omega$)
            \item $q_{XY}(\omega)$: \textbf{spectrul de cuadratură} (co-mișcare defazată)
            \item Acestea sunt mai greu de interpretat direct $\Rightarrow$ se folosesc măsurile derivate
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Coerența pătratică}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    \begin{block}{Coerența pătratică: $R^2$ în domeniul frecvenței}
        {\small
        \[
            \kappa^2_{XY}(\omega) = \frac{|f_{XY}(\omega)|^2}{f_X(\omega)\,f_Y(\omega)}, \qquad 0 \leq \kappa^2_{XY}(\omega) \leq 1
        \]
        }
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Interpretare}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Măsoară \textbf{tăria asocierii liniare} între $X$ și $Y$ la frecvența $\omega$
            \item $\kappa^2 = 1$: relație liniară perfectă la această frecvență; $\kappa^2 = 0$: nicio relație
            \item Analog al $R^2$ din regresie, dar \textbf{specific frecvenței}
            \item Spre deosebire de corelația în domeniul timpului, coerența poate arăta că două serii sunt puternic legate la frecvențele ciclului de afaceri dar necorelate la frecvențe mai înalte
        \end{itemize}
    \end{exampleblock}
    \vspace{-2mm}
    \begin{alertblock}{Estimare}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Cross-periodograma brută dă $\hat{\kappa}^2 = 1$ peste tot --- trebuie netezită în prealabil
            \item Se folosesc estimații cross-spectrale cu multitaper sau netezire prin nuclee
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_coherence}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_coherence}
\end{frame}

\begin{frame}{Spectrul de fază și câștigul}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{Spectrul de fază}
            {\small
            \[
                \phi_{XY}(\omega) = \arctan\!\left(\frac{-q_{XY}(\omega)}{c_{XY}(\omega)}\right)
            \]
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{Avans/întârziere} între $X$ și $Y$ la frecvența $\omega$
                \item $\phi > 0$: $X$ conduce $Y$
                \item Decalaj temporal: $\tau(\omega) = \phi(\omega)/\omega$
                \item Faza are sens doar unde coerența este semnificativă
            \end{itemize}}
        \end{block}

        \column{0.50\textwidth}
        \begin{block}{Câștigul (raportul de amplitudine)}
            {\small
            \[
                G_{XY}(\omega) = \frac{|f_{XY}(\omega)|}{f_X(\omega)}
            \]
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Măsoară cât de mult frecvența $\omega$ din $X$ este \textbf{amplificată sau atenuată} în $Y$
                \item Dacă $Y = \beta X + \varepsilon$, câștigul este $|\beta|$ la toate frecvențele
                \item Util pentru evaluarea transmisiei dependente de frecvență
            \end{itemize}}
        \end{block}
    \end{columns}
    \vspace{2mm}
    {\small \textbf{Împreună}: Coerența vă spune \textit{cât de puternic} sunt legate două serii la fiecare frecvență, faza vă spune \textit{care conduce}, iar câștigul vă spune \textit{cu cât} sunt amplificate oscilațiile.}
    \end{cminipage}
    \quantlet{TSA\_ch12\_coherence}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_coherence}
\end{frame}

\section{Filtrarea în domeniul frecvenței}
%=============================================================================

\begin{frame}{Filtre liniare și funcții de transfer I}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Un filtru liniar}
        {\small Un filtru liniar invariant în timp transformă $\{X_t\}$ în $\{Y_t\}$:
        \[
            Y_t = \sum_{j=-\infty}^{\infty} a_j\,X_{t-j} = A(B)\,X_t
        \]
        \textbf{Funcția de răspuns în frecvență} (funcția de transfer) este:
        \[
            A(\omega) = \sum_{j=-\infty}^{\infty} a_j\,e^{-i\omega j}
        \]
        }
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Rezultatul fundamental: Filtrarea = Multiplicare}
        {\small Dacă $X_t$ are densitate spectrală $f_X(\omega)$, atunci $Y_t = A(B)X_t$ are densitate spectrală:
        \[
            f_Y(\omega) = |A(\omega)|^2 \cdot f_X(\omega)
        \]
        \textbf{Un filtru multiplică spectrul cu câștigul său pătratic} --- de aceea domeniul frecvenței face analiza filtrelor trivială!}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Filtre liniare și funcții de transfer II}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{exampleblock}{Exemple clasice de filtre}
        \begin{itemize}\setlength{\itemsep}{1pt}
            \item \textbf{Prima diferență} $(1-B)$: \quad $|A(\omega)|^2 = 4\sin^2(\omega/2)$\\
                  $\Rightarrow$ amplifică frecvențele înalte, elimină $\omega = 0$ (elimină trendul)
            \item \textbf{Media mobilă de ordin $m$}: \quad $|A(\omega)|^2 = \left(\dfrac{\sin(m\omega/2)}{m\sin(\omega/2)}\right)^2$\\
                  $\Rightarrow$ filtru trece-jos (lasă componentele lente, elimină oscilațiile rapide)
            \item \textbf{Diferența sezonieră} $(1-B^{12})$: \quad vârfuri ale $|A(\omega)|^2 = 0$ la armonicele sezoniere\\
                  $\Rightarrow$ elimină tiparul sezonier lunar
        \end{itemize}
    \end{exampleblock}
    \vspace{-2mm}
    \begin{block}{Filtrul band-pass ideal}
        {\small Lasă să treacă frecvențele din $[\omega_L, \omega_H]$ și blochează restul:
        \[
            A(\omega) = \begin{cases} 1 & \text{dacă } \omega_L \leq |\omega| \leq \omega_H \\ 0 & \text{altfel} \end{cases}
            \qquad a_j = \frac{\sin(\omega_H j) - \sin(\omega_L j)}{\pi j}
        \]
        \textbf{Problemă}: necesită infinit de multe avansuri/întârzieri $\Rightarrow$ trebuie trunchiat în practică.}
    \end{block}
    \end{cminipage}

\end{frame}

    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Filtrul Baxter--King (BK) (1999)}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Aproximare simetrică, de lungime fixă a filtrului band-pass ideal
            \item Utilizează $K$ avansuri și întârzieri (de obicei $K = 12$ pentru date trimestriale)
            \item Coeficienți: filtru ideal trunchiat + constrâns să aibă suma zero (elimină trendul)
            \item \textbf{Dezavantaj}: pierde $K$ observații la fiecare capăt al seriei
        \end{itemize}
        }
    \end{block}

    \begin{exampleblock}{Filtrul Christiano--Fitzgerald (CF) (2003)}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}

            \item Aproximare optimală sub ipoteza că $X_t$ este un mers aleator
            \item Ponderile filtrului se adaptează la capetele seriei
            \item În general preferat față de BK pentru majoritatea aplicațiilor economice
        \end{itemize}
        }
    \end{exampleblock}

    {\small \textbf{Filtrul HP} ($\lambda=1600$): filtru trece-sus cu $|A_{\text{HP}}(\omega)|^2 = \frac{4\lambda(1 - \cos\omega)^2}{1 + 4\lambda(1 - \cos\omega)^2}$, lasă să treacă cicluri $< 32$ trimestre.}

\begin{frame}{Scurgerea spectrală și ponderarea datelor}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Problema scurgerii spectrale}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Calculul DFT presupune implicit că datele sunt periodice (se „înfășoară")
            \item Dacă seria nu începe și nu se termină la aceeași valoare, aceasta creează \textbf{discontinuități}
            \item Energia de la vârfurile spectrale puternice „se scurge" în frecvențele vecine
            \item Rezultat: periodograma arată humps largi în loc de vârfuri ascuțite
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Soluție: Ponderarea datelor (aplicarea ferestrei)}
        {\small Se aplică o pondere $\{d_t\}$ care aduce lin datele la zero la ambele capete:
        \[
            y_t^{\text{ponderat}} = d_t \cdot y_t, \qquad \text{unde } d_1 \approx d_n \approx 0, \quad d_{n/2} \approx 1
        \]
        Ferestre comune:
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Hanning}: $d_t = 0{,}5(1 - \cos(2\pi t / n))$ --- bun uz general
            \item \textbf{Hamming}: $d_t = 0{,}54 - 0{,}46\cos(2\pi t / n)$ --- reducere ușor mai mică a scurgerii
            \item \textbf{Blackman}: $d_t = 0{,}42 - 0{,}5\cos(2\pi t/n) + 0{,}08\cos(4\pi t/n)$ --- excelentă suprimare
            \item \textbf{DPSS (Slepian)}: concentrare optimală a energiei (folosit în multitaper)
        \end{itemize}}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{De la Fourier la Wavelets: Motivație}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Limitarea analizei Fourier}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Analiza Fourier descompune un semnal în componente sinus/cosinus \textit{globale}
            \item Fiecare componentă de frecvență se extinde pe \textbf{întreaga} perioadă de timp
            \item Problemă: \textbf{semnale nestaționare} --- conținutul de frecvență se schimbă în timp!
            \item Exemplu: volatilitatea financiară este calmă în unele perioade, turbulentă în altele
        \end{itemize}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Principiul incertitudinii (Heisenberg--Gabor)}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item \textbf{Nu puteți} avea rezoluție perfectă simultan în timp și frecvență
            \item $\Delta t \cdot \Delta f \geq \frac{1}{4\pi}$ (incertitudine timp-frecvență)
            \item Fourier: $\Delta f$ este minimă dar $\Delta t = \infty$ (fără localizare în timp)
            \item Wavelets: adaptivă --- rezoluție bună în timp la $f$ înaltă, rezoluție bună în frecvență la $f$ joasă
        \end{itemize}
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{Wavelets: Localizate atât în timp, cât și în frecvență}
        \begin{itemize}\setlength{\itemsep}{-2pt}
            \item Folosesc funcții bază \textbf{scurte, oscilatorii} („wavelets") în loc de sinusoide infinite
            \item Wavelets se întind pentru frecvențe joase, se comprimă pentru frecvențe înalte
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Transformata Wavelet Continuă (CWT)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Definiție}
        {\small CWT a unui semnal $x(t)$ în raport cu un wavelet mamă $\psi(t)$ este:
        \[
            W(a, b) = \frac{1}{\sqrt{|a|}}\int_{-\infty}^{\infty} x(t)\,\overline{\psi\!\left(\frac{t-b}{a}\right)}\,dt
        \]
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $a$: parametrul de \textbf{scală} (invers proporțional cu frecvența: $a$ mare = frecvență joasă)
            \item $b$: parametrul de \textbf{translație} (localizare în timp)
            \item $|W(a,b)|^2$: \textbf{puterea wavelet} la scala $a$ și momentul $b$
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Wavelets mamă comune}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Morlet}: $\psi(t) = \pi^{-1/4}e^{i\omega_0 t}e^{-t^2/2}$ (exponențială complexă modulată gaussian, $\omega_0 = 6$)
            \item \textbf{Pălăria mexicană}: $\psi(t) = (1-t^2)e^{-t^2/2}$ (derivata a doua a gaussianei)
            \item \textbf{Haar}: cel mai simplu wavelet --- constant pe porțiuni (bun pentru detectarea salturilor)
            \item Pentru serii de timp: \textbf{Morlet este standard} --- echilibru bun timp/frecvență
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_morlet}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_morlet}
\end{frame}

\begin{frame}{Spectrul de putere wavelet și scalograma}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Spectrul de putere wavelet}
        {\small
        \[
            \text{WPS}(a, b) = |W(a, b)|^2
        \]
        Afișat ca o \textbf{scalogramă}: o hartă termică 2D cu timpul pe axa $x$ și scala (perioada) pe axa $y$. Intensitatea culorii = puterea.}
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{Spectrul wavelet global}
            {\small Puterea medie în timp la fiecare scală:
            \[
                \overline{W}^2(a) = \frac{1}{n}\sum_{b=1}^{n} |W(a,b)|^2
            \]
            Comparabil cu densitatea spectrală Fourier. Util pentru identificarea periodicităților dominante.}
        \end{exampleblock}

        \column{0.50\textwidth}
        \begin{block}{Conul de influență (COI)}
            \begin{itemize}\setlength{\itemsep}{-2pt}
                \item Lângă capetele seriei, coeficienții wavelet sunt nesiguri (efecte de margine)
                \item COI marchează regiunea unde efectele de margine sunt semnificative
                \item Interpretați puterea doar \textbf{în interiorul} COI
            \end{itemize}
        \end{block}
    \end{columns}
    \vspace{1mm}
    {\small \textbf{Semnificație statistică}: Sub $H_0$ de zgomot roșu (AR(1)), puterea wavelet urmează o distribuție $\chi^2_2$ scalată de spectrul de fundal. Contururile la nivelul de semnificație 95\% sunt standard.}
    \end{cminipage}
    \quantlet{TSA\_ch12\_scalogram}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_scalogram}
\end{frame}

\begin{frame}{Coerența wavelet}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Transformata cross-wavelet și coerența}
        {\small Pentru două serii $X_t$ și $Y_t$ cu CWT-urile $W_X(a,b)$ și $W_Y(a,b)$:
        \[
            W_{XY}(a,b) = W_X(a,b)\,\overline{W_Y(a,b)} \qquad \text{(spectrul cross-wavelet)}
        \]
        \textbf{Coerența wavelet} este:
        \[
            R^2_{XY}(a,b) = \frac{|S(W_{XY}(a,b))|^2}{S(|W_X(a,b)|^2) \cdot S(|W_Y(a,b)|^2)}
        \]
        unde $S(\cdot)$ este un operator de netezire atât în timp, cât și în scală.}
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Interpretare și aplicații}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $R^2_{XY}(a,b) \in [0, 1]$: măsură variabilă în timp și scală a co-mișcării
            \item \textbf{Săgeți de fază} pe graficul de coerență indică relații de avans/întârziere
            \item Aplicație: prețul petrolului și piețele de acțiuni --- coerența poate fi ridicată la frecvențele ciclului în timpul crizelor dar scăzută în perioade normale
            \item Aplicație: inflația și masa monetară --- coerența variază între regimuri istorice
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
    \quantlet{TSA\_ch12\_wavelet\_coherence}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_wavelet_coherence}
\end{frame}

\begin{frame}{Procese local staționare}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Dahlhaus (1997): Un cadru pentru spectre nestaționare}
        {\small Un \textbf{proces local staționar} are reprezentarea:
        \[
            X_{t,T} = \int_{-\pi}^{\pi} A\!\left(\frac{t}{T},\,\omega\right) e^{i\omega t}\,dZ(\omega)
        \]
        unde $A(u,\omega)$ este \textbf{funcția de transfer variabilă în timp} și $Z(\omega)$ are incrementi ortogonali.
        \textbf{Densitatea spectrală variabilă în timp} este:
        \[
            f(u,\,\omega) = |A(u,\,\omega)|^2, \qquad u \in [0,1],\; \omega \in [-\pi,\pi]
        \]
        }
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Ideea cheie}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item La fiecare punct de timp rescalat $u = t/T$, procesul ``arată'' staționar cu spectrul $f(u,\omega)$
            \item Generalizează reprezentarea Cramér clasică pentru a permite structuri variabile în timp
            \item Estimare prin periodograma locală sau metode bazate pe wavelets
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Detecția punctelor de schimbare spectrală}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Testarea staționarității spectrale}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item $H_0$: $f(u, \omega) = f(\omega)$ pentru toți $u$ \quad (spectrul este invariant în timp)
            \item $H_1$: $f(u, \omega)$ se schimbă la unul sau mai multe puncte de timp (puncte de schimbare spectrală)
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{Abordări}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{CUSUM pe periodograme locale}: Compararea estimărilor spectrale din sub-segmente
            \item \textbf{Teste bazate pe wavelets}: Utilizarea varianței la fiecare scală pentru detectarea schimbărilor în benzi de frecvență specifice
            \item \textbf{Statistici de tip Kolmogorov--Smirnov}: Compararea distribuțiilor spectrale între ferestre
        \end{itemize}
    \end{exampleblock}
    \vspace{-2mm}
    \begin{alertblock}{Aplicație: Detectarea regimurilor în economie}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Detectarea începutului și sfârșitului Marii Moderații prin schimbări spectrale
            \item Identificarea rupturilor structurale în dinamica inflației sau volatilitatea financiară
            \item Schimbări în frecvența sau amplitudinea ciclului de afaceri între regimuri de politică economică
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{STFT vs.\ CWT: Legătura dintre Fourier și Wavelets}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Transformata Fourier pe termen scurt (STFT)}
        {\small Aplicați DFT pe segmente cu fereastră: $\text{STFT}(\tau, \omega) = \sum_{t} x(t)\,w(t - \tau)\,e^{-i\omega t}$, unde $w(\cdot)$ este o fereastră de lățime fixă centrată la momentul $\tau$.}
    \end{block}
    \vspace{-2mm}
    {\scriptsize
    \begin{center}
    \begin{tabular}{l|c|c}
        \toprule
        & \textbf{STFT} & \textbf{CWT} \\
        \midrule
        Fereastră & Lățime fixă pentru toți $\omega$ & Adaptivă (îngustă la $\omega$ mare) \\
        Rezoluție & Uniformă $\Delta t \times \Delta \omega$ & Variabilă: $\Delta t \propto 1/\omega$ \\
        Bază & Sinusoide cu fereastră & Wavelets scalate/deplasate \\
        Cel mai bun pentru & Semnale cu frecvență stabilă & Semnale cu elemente tranzitorii \\
        Spectrogramă & $|\text{STFT}|^2$ = spectrogramă & $|W(a,b)|^2$ = scalogramă \\
        \bottomrule
    \end{tabular}
    \end{center}
    }
    \vspace{-2mm}
    \begin{exampleblock}{Perspectivă practică}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item STFT adaugă localizare temporală analizei Fourier; CWT adaptează în plus lățimea ferestrei la frecvență
            \item Pentru date economice: CWT este preferată deoarece dinamicile ciclului de afaceri și cele de înaltă frecvență necesită rezoluții diferite
        \end{itemize}}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Fourier vs.\ Wavelets: Când se folosește fiecare}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    {\small
    \begin{block}{Ghid de decizie}
    \vspace{-2mm}
    \begin{center}
    \begin{tabular}{l|c|c}
        \toprule
        \textbf{Criteriu} & \textbf{Fourier / Spectral} & \textbf{Wavelet} \\
        \midrule
        Staționaritate & Necesară (sau presupusă) & Nu este necesară \\
        Localizare în timp & Nu & Da \\
        Rezoluția în frecvență & Uniformă la toate $\omega$ & Variază cu scala \\
        Cel mai bun pentru & Serii staționare, proiectare filtre & Nestaționare, evenimente tranzitorii \\
        Testarea ipotezelor & Bine dezvoltată (F-test, $\chi^2$) & Monte Carlo, nul AR(1) \\
        Cost computațional & $O(n \log n)$ & $O(n \log n)$ la $O(n^2)$ \\
        Comun în & Economie, procesarea semnalelor & Geofizică, finanțe, climă \\
        \bottomrule
    \end{tabular}
    \end{center}
    \end{block}
    }
    \vspace{-2mm}
    \begin{exampleblock}{Recomandare practică}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Folosiți \textbf{analiza Fourier/spectrală} când credeți că structura de frecvență este \textit{stabilă} în timp
            \item Folosiți \textbf{wavelets} când suspectați că structura de frecvență \textit{se schimbă} în timp (schimbări de regim, cicluri în evoluție)
            \item De multe ori este util să folosiți \textbf{ambele}: Fourier pentru forma spectrală generală, wavelets pentru comportamentul variabil în timp
        \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Concluzii cheie}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    \begin{block}{Ce am învățat}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Densitatea spectrală} $f(\omega)$ descompune varianța pe frecvențe (Wiener--Khintchine: $\gamma(h) \leftrightarrow f(\omega)$)
            \item \textbf{Periodograma} este estimatorul natural dar \textit{inconsistentă} --- trebuie netezită (multitaper = standard de aur)
            \item Măsurile \textbf{cross-spectrale} (coerență, fază, câștig) dezvăluie relații specifice frecvenței
            \item \textbf{Filtrele band-pass} (BK, CF) extrag ciclurile de afaceri; HP este un filtru trece-sus
            \item \textbf{Wavelets} oferă localizare timp-frecvență pentru semnale nestaționare
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Recomandări practice}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Eliminați trendul întotdeauna înainte de analiza spectrală; folosiți scala logaritmică pentru grafice
            \item Alegeți multitaper pentru estimări finale; periodograma brută doar pentru explorare
            \item Pentru date nestaționare, preferați analiza wavelet în locul metodelor Fourier
        \end{itemize}
        }
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 1}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Ce măsoară funcția de densitate spectrală $f(\omega)$ și care este relația sa cu funcția de autocovarianță $\gamma(h)$?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Variante de răspuns}

        \textcolor{MainBlue}{\textbf{(A)}} $f(\omega)$ măsoară media procesului la frecvența $\omega$; este transformata Laplace a lui $\gamma(h)$\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} $f(\omega)$ măsoară contribuția frecvenței $\omega$ la varianța totală; este transformata Fourier a lui $\gamma(h)$ (teorema Wiener--Khintchine)\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} $f(\omega)$ măsoară defazajul la frecvența $\omega$; este inversa lui $\gamma(h)$\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} $f(\omega)$ măsoară componenta de trend la frecvența $\omega$; este derivata lui $\gamma(h)$

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 1: Răspuns}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\small
    \begin{exampleblock}{Răspuns: (B)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Funcția de densitate spectrală $f(\omega) = \frac{1}{2\pi}\sum_{h} \gamma(h)e^{-i\omega h}$ descompune varianța totală a unui proces staționar pe frecvențe. Varianța totală este $\gamma(0) = \int_{-\pi}^{\pi} f(\omega)\,d\omega$. Această pereche de transformate Fourier între $\gamma(h)$ și $f(\omega)$ este teorema Wiener--Khintchine --- rezultatul fundamental al analizei spectrale.
    \end{itemize}
    \end{exampleblock}
    }
    \hfill\quantlet{TSA\_ch12\_quiz1}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_quiz1}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 2}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item De ce periodograma brută este un estimator \textit{inconsistent} al densității spectrale și cum poate fi remediat acest lucru?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Variante de răspuns}

        \textcolor{MainBlue}{\textbf{(A)}} Periodograma este deplasată; se remediază prin creșterea dimensiunii eșantionului\\[1pt]

        \textcolor{MainBlue}{\textbf{(B)}} Varianța periodogramei nu scade cu dimensiunea eșantionului ($\Var[I(\omega)] \approx f(\omega)^2$ pentru orice $n$); se remediază prin netezire (ex.\ multitaper, Welch, nucleu Daniell)\\[1pt]

        \textcolor{MainBlue}{\textbf{(C)}} Periodograma funcționează doar pentru zgomot alb; se remediază prin diferențierea datelor\\[1pt]

        \textcolor{MainBlue}{\textbf{(D)}} Periodograma este inconsistentă deoarece folosește FFT; se remediază prin calculul direct al DFT

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 2: Răspuns}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\small
    \begin{exampleblock}{Răspuns: (B)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Periodograma este aproximativ nedeplasată ($\E[I(\omega)] \to f(\omega)$), dar varianța sa $\Var[I(\omega)] \approx f(\omega)^2$ nu scade când $n \to \infty$. Aceasta se datorează faptului că adăugarea mai multor date introduce mai multe frecvențe Fourier, în loc să îmbunătățească estimările la frecvențele existente. Netezirea (medierea valorilor vecine ale periodogramei, multitaper sau metode cu fereastră lag) sacrifică rezoluția în frecvență în schimbul varianței reduse, producând un estimator consistent.
    \end{itemize}
    \end{exampleblock}
    }
    \hfill\quantlet{TSA\_ch12\_quiz2}{https://github.com/danpele/TSA/tree/main/TSA_ch12/TSA_ch12_quiz2}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 3}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Observați o densitate spectrală cu un vârf ascuțit lângă $\omega = 0$ și vârfuri mai mici la $\omega = \pi/6, \pi/3, \pi/2, 2\pi/3, 5\pi/6, \pi$. Ce tip de proces a generat probabil aceste date?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Variante de răspuns}

        \textcolor{MainBlue}{\textbf{(A)}} Zgomot alb --- spectru plat cu fluctuații aleatorii\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} AR(1) cu $\phi > 0$ --- spectru roșu cu vârf la zero\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Proces sezonier lunar cu trend --- vârf la zero (trend) plus vârfuri la armonicele sezoniere $\omega_k = 2\pi k/12$\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Mers aleator --- toată puterea la $\omega = 0$

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 3: Răspuns}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\small
    \begin{exampleblock}{Răspuns: (C)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Vârfurile la $\omega = \pi/6, \pi/3, \pi/2, 2\pi/3, 5\pi/6, \pi$ corespund exact lui $2\pi k/12$ pentru $k = 1, 2, \ldots, 6$ --- armonicele sezoniere ale datelor lunare ($s = 12$). Vârful la $\omega \approx 0$ indică o componentă puternică de trend sau pe termen lung. Aceasta este semnătura spectrală clasică a unei serii lunare cu atât trend, cât și sezonalitate, cum ar fi pasagerii companiilor aeriene sau vânzările cu amănuntul.
    \end{itemize}
    \end{exampleblock}

\begin{frame}{Întrebarea 4}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Care este avantajul principal al analizei wavelet față de analiza spectrală Fourier pentru serii de timp economice?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Variante de răspuns}

        \textcolor{MainBlue}{\textbf{(A)}} Wavelets oferă rezoluție mai mare în frecvență decât Fourier pentru serii staționare\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} Wavelets pot localiza caracteristici spectrale atât în timp, cât și în frecvență, fiind potrivite pentru procese nestaționare unde conținutul de frecvență se schimbă în timp\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Wavelets sunt computațional mai rapide decât FFT\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Wavelets nu necesită alegerea niciunui parametru, spre deosebire de metodele Fourier

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 4: Răspuns}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\small
    \begin{exampleblock}{Răspuns: (B)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Analiza wavelet oferă localizare simultană în timp și frecvență. Spre deosebire de analiza Fourier, care produce un spectru global unic, wavelets creează o scalogramă care arată cum puterea spectrală evoluează în timp. Aceasta este crucială pentru seriile economice supuse rupturilor structurale, schimbărilor de regim și dinamicii ciclurilor de afaceri în evoluție. Principiul incertitudinii Heisenberg limitează rezoluția comună, dar wavelets echilibrează adaptiv rezoluția în timp și frecvență pe scale diferite.
    \end{itemize}
    \end{exampleblock}
    }

\begin{frame}{Întrebarea 5}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Întrebare}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Un cercetător aplică filtrul Hodrick--Prescott (HP) cu $\lambda = 1600$ pe date trimestriale de PIB. Care este interpretarea spectrală a acestui filtru?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Variante de răspuns}

        \textcolor{MainBlue}{\textbf{(A)}} Este un filtru band-pass care extrage cicluri cu perioada 6--32 trimestre\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} Este un filtru trece-jos care extrage trendul păstrând doar frecvențele joase\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Componenta ciclică HP este un filtru trece-sus: $|A(\omega)|^2 = \frac{4\lambda(1-\cos\omega)^2}{1+4\lambda(1-\cos\omega)^2}$, lasând să treacă toate ciclurile mai scurte de $\approx$ 32 trimestre inclusiv zgomotul neregulat\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Este un filtru notch care elimină doar frecvența sezonieră

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Întrebarea 5: Răspuns}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.3cm}
    {\small
    \begin{exampleblock}{Răspuns: (C)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Componenta ciclică a filtrului HP este un filtru trece-sus care lasă să treacă toate frecvențele peste un prag determinat de $\lambda$. Cu $\lambda = 1600$ și date trimestriale, lasă să treacă cicluri mai scurte de $\approx 32$ trimestre. Spre deosebire de un filtru band-pass (BK, CF), ciclul HP include atât componenta ciclului de afaceri, \textit{cât și} componenta neregulată de frecvență înaltă. Aceasta este o limitare bine cunoscută: „ciclul" HP este contaminat cu zgomot, motiv pentru care filtrele band-pass dedicate sunt preferate pentru extragerea ciclului de afaceri.
    \end{itemize}
    \end{exampleblock}
    }
