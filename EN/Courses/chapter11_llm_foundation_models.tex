\section{Transformer Architecture}
%=============================================================================

\begin{frame}{Researcher Spotlight: Vaswani et al.\ (2017)}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{``Attention Is All You Need''}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Published at NeurIPS 2017 by researchers at Google Brain
            \item Over \textbf{130{,}000 citations} --- one of the most influential ML papers ever
            \item Originally designed for \textbf{machine translation} (NLP)
            \item Key innovation: replaced recurrence entirely with \textbf{self-attention}
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Why It Matters for Time Series}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Time series are \textbf{sequential} data --- same fundamental challenge as language
            \item Transformers handle \textbf{variable-length} sequences naturally
            \item \textbf{Parallel computation}: unlike RNNs, all positions processed simultaneously
            \item Foundation models (GPT, BERT, T5) all build on this architecture
        \end{itemize}
        }
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{Impact Timeline}
        {\small 2017: Attention Is All You Need $\to$ 2018: BERT, GPT $\to$ 2020: GPT-3 $\to$ 2023: TimeGPT, Chronos $\to$ 2024: TimesFM, Moirai}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Self-Attention: Intuition}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    \begin{block}{Why Attention Replaces Recurrence}
        {\scriptsize
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{RNNs}: process tokens sequentially $\Rightarrow$ $O(T)$ sequential steps, information bottleneck
            \item \textbf{Self-Attention}: every position attends to every other position \textbf{in parallel}
            \item Direct connections between distant positions: no vanishing gradients
        \end{itemize}
        }
    \end{block}
    \vspace{-3mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{alertblock}{RNN: Sequential}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Path length between $x_1$ and $x_T$: $O(T)$ steps
                \item Cannot parallelize across time steps
                \item Gradient must flow through $T$ multiplications
                \item Information ``compresses'' through hidden state
            \end{itemize}
            }
        \end{alertblock}

        \column{0.48\textwidth}
        \begin{exampleblock}{Attention: Parallel}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Path length between any two positions: $O(1)$
                \item Fully parallelizable $\Rightarrow$ faster on GPUs
                \item Each position directly ``sees'' all others
                \item Learns which positions to attend to (data-driven)
            \end{itemize}
            }
        \end{exampleblock}
    \end{columns}
    \vspace{-2mm}
    \begin{center}
        \includegraphics[width=0.90\textwidth, height=0.22\textheight, keepaspectratio]{ch11_quiz1_attention_vs_rnn.pdf}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Self-Attention: Mathematics}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Scaled Dot-Product Attention}
        Given input embeddings $X \in \mathbb{R}^{T \times d}$, compute:
        \vspace{-2mm}
        $$Q = XW^Q, \quad K = XW^K, \quad V = XW^V$$
        \vspace{-4mm}
        $$\text{Attention}(Q,K,V) = \text{softmax}\!\left(\frac{QK^\top}{\sqrt{d_k}}\right)V$$
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.50\textwidth}
        {\scriptsize
        \begin{exampleblock}{Interpretation}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item $Q$ (Queries): ``What am I looking for?''
                \item $K$ (Keys): ``What do I contain?''
                \item $V$ (Values): ``What information do I provide?''
                \item $\sqrt{d_k}$: scaling prevents softmax saturation
                \item Output: weighted average of values, weights = compatibility scores
            \end{itemize}
        \end{exampleblock}
        }

        \column{0.48\textwidth}
        \begin{center}
            \includegraphics[width=0.95\textwidth, height=0.38\textheight, keepaspectratio]{ch11_attention_heatmap.pdf}
        \end{center}
        {\scriptsize Attention weights heatmap: rows attend to columns.}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Multi-Head Attention}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Multiple Attention Heads Capture Diverse Patterns}
        \vspace{-2mm}
        \begin{align*}
            \text{MultiHead}(Q,K,V) &= \text{Concat}(\text{head}_1, \ldots, \text{head}_h)\, W^O \\
            \text{where head}_i &= \text{Attention}(QW_i^Q,\; KW_i^K,\; VW_i^V)
        \end{align*}
        \vspace{-4mm}
        {\small Each head projects to a lower dimension: $W_i^Q, W_i^K \in \mathbb{R}^{d \times d_k}$, $W_i^V \in \mathbb{R}^{d \times d_v}$ with $d_k = d_v = d/h$.}
    \end{block}

    \vspace{-1mm}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{What Different Heads Learn}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Head 1: \textbf{Trend} --- attend to recent values
                \item Head 2: \textbf{Seasonality} --- attend to same period last cycle
                \item Head 3: \textbf{Outliers} --- attend to extreme values
                \item Head 4: \textbf{Long-range} --- attend to distant correlations
            \end{itemize}
            }
        \end{exampleblock}

        \column{0.48\textwidth}
        \begin{alertblock}{Computational Cost}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Self-attention complexity: $O(T^2 \cdot d)$
                \item Memory: $O(T^2)$ for the attention matrix
                \item For long sequences ($T > 4096$): quadratic cost becomes prohibitive
                \item Solutions: patching, sparse attention, linear attention
            \end{itemize}
            }
        \end{alertblock}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Positional Encoding}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Why Position Matters}
        {\small Self-attention is \textbf{permutation-equivariant}: without positional info, it treats $\{x_1, x_2, x_3\}$ and $\{x_3, x_1, x_2\}$ identically. Time series \textbf{order matters}, so we inject position.}
    \end{block}
    \vspace{-2mm}
    \begin{alertblock}{Sinusoidal Positional Encoding (Vaswani et al., 2017)}
        \vspace{-2mm}
        \begin{align*}
            PE_{(pos,\, 2i)} &= \sin\!\left(\frac{pos}{10000^{2i/d}}\right) \\
            PE_{(pos,\, 2i+1)} &= \cos\!\left(\frac{pos}{10000^{2i/d}}\right)
        \end{align*}
        \vspace{-4mm}
        {\small Each dimension uses a different frequency $\Rightarrow$ unique positional signature. Relative positions can be computed via dot products.}
    \end{alertblock}
    \vspace{-2mm}
    \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.28\textheight, keepaspectratio]{ch11_positional_encoding.pdf}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{The Full Transformer Block}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \begin{block}{Transformer Block Components}
            {\scriptsize
            \begin{enumerate}\setlength{\itemsep}{0pt}
                \item \textbf{Input embedding} + Positional encoding
                \item \textbf{Layer Normalization} (pre-norm variant)
                \item \textbf{Multi-Head Self-Attention}
                \item \textbf{Residual connection}: $x + \text{Attn}(x)$
                \item \textbf{Layer Normalization}
                \item \textbf{Feed-Forward Network} (FFN): two linear layers with GELU/ReLU
                $$\text{FFN}(x) = W_2 \cdot \text{GELU}(W_1 x + b_1) + b_2$$
                \item \textbf{Residual connection}: $x + \text{FFN}(x)$
            \end{enumerate}
            }
        \end{block}
        {\scriptsize Stacking $N$ blocks ($N=6$--$24$) gives depth. Residual connections enable gradient flow.}

        \column{0.48\textwidth}
        \begin{center}

\end{frame}

\begin{frame}{Encoder vs Decoder Architectures}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    {\footnotesize
    \begin{block}{Three Architectural Paradigms}
    \vspace{-2mm}
    \begin{center}
    \begin{tabular}{l|p{3cm}|p{3cm}|p{3.5cm}}
        \toprule
        & \textbf{Encoder-only} & \textbf{Decoder-only} & \textbf{Encoder-Decoder} \\
        \midrule
        \textbf{NLP origin} & BERT & GPT & T5, BART \\
        \textbf{Attention} & Bidirectional & Causal (masked) & Cross-attention \\
        \textbf{TS models} & Moirai, PatchTST & TimesFM, Lag-Llama & Chronos, TimeGPT \\
        \textbf{Strengths} & Learns rich representations & Natural for autoregressive generation & Flexible input-output mapping \\
        \textbf{TS use} & Representation, classification & Forecasting, generation & Forecasting, seq2seq \\
        \bottomrule
    \end{tabular}
    \end{center}
    \vspace{-2mm}
    \end{block}
    }
    \vspace{-2mm}
    \begin{exampleblock}{Key Distinction}
        {\small \textbf{Causal masking} (decoder): position $t$ can only attend to $\{1, \ldots, t\}$. Prevents ``peeking'' into the future --- natural fit for time series forecasting. \textbf{Bidirectional} (encoder): every position sees all others --- useful for representation learning and classification.}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\section{From Transformers to Foundation Models}
%=============================================================================

\begin{frame}{The Foundation Model Paradigm}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Three-Stage Pipeline}
        {\small
        \begin{enumerate}\setlength{\itemsep}{0pt}
            \item \textbf{Pre-training}: Train on massive, diverse corpora (billions of time points across domains)
            \item \textbf{Fine-tuning} (optional): Adapt to a specific domain or dataset with a small learning rate
            \item \textbf{Inference}: Deploy on new, unseen series --- zero-shot or few-shot
        \end{enumerate}
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \begin{center}
            \includegraphics[width=0.95\textwidth, height=0.40\textheight, keepaspectratio]{ch11_foundation_paradigm.pdf}
        \end{center}

        \column{0.48\textwidth}
        \begin{alertblock}{Key Insight (Bommasani et al., 2021)}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item ``A foundation model is any model trained on broad data that can be adapted to a wide range of downstream tasks.''
                \item For TS: train on \textit{energy + finance + retail + health + weather} $\Rightarrow$ forecast \textit{anything}
                \item Critical assumption: \textbf{shared temporal patterns} exist across domains (trend, seasonality, noise)
            \end{itemize}
            }
        \end{alertblock}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Tokenization Strategies for Time Series}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    {\scriptsize
    \begin{block}{Challenge: Continuous Values $\to$ Discrete Tokens}
        NLP models process discrete tokens (words). Time series are \textbf{continuous} $\Rightarrow$ need a mapping strategy.
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{Strategy 1: Raw Value Embeddings}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Project each $x_t \in \mathbb{R}$ to $\mathbb{R}^d$ via linear layer
                \item Used by: vanilla Transformer TS models
                \item Simple but ignores local structure
            \end{itemize}
        \end{exampleblock}
        \vspace{-2mm}
        \begin{exampleblock}{Strategy 2: Scaling + Quantization}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Normalize, then map to $B$ discrete bins
                \item Used by: \textbf{Chronos} (4096 bins)
                \item Leverages NLP machinery directly
            \end{itemize}
        \end{exampleblock}

        \column{0.48\textwidth}
        \begin{exampleblock}{Strategy 3: Patching}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Group $P$ consecutive points into one token
                \item Used by: \textbf{PatchTST}, \textbf{TimesFM}
                \item Reduces sequence length by factor $P$
            \end{itemize}
        \end{exampleblock}
        \vspace{-2mm}
        \begin{exampleblock}{Strategy 4: Variate-Independent Channels}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Each variable processed independently
                \item Used by: \textbf{PatchTST}, \textbf{Moirai}
                \item Scales to any number of variates
            \end{itemize}
        \end{exampleblock}
    \end{columns}
    }
    \vspace{-1mm}
    \begin{center}
        \includegraphics[width=0.95\textwidth, height=0.18\textheight, keepaspectratio]{ch11_tokenization_strategies.pdf}
    \end{center}
    \end{cminipage}
\end{frame}

\begin{frame}{Patching: Why It Works}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{PatchTST Insight (Nie et al., 2023)}
        {\small Divide a time series of length $L$ into $N = \lfloor L / P \rfloor$ non-overlapping patches of size $P$:
        $$\mathbf{p}_i = (x_{(i-1)P+1},\; x_{(i-1)P+2},\; \ldots,\; x_{iP}) \in \mathbb{R}^P, \quad i = 1, \ldots, N$$
        Each patch is linearly projected to dimension $d$ and fed as a single token to the Transformer.}
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{Benefits of Patching}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{Reduces complexity}: attention $O(T^2) \to O((T/P)^2)$
                \item \textbf{Local semantics}: each patch captures a short sub-pattern (e.g., daily cycle)
                \item \textbf{Longer context}: with $P=32$, a 512-token window covers 16{,}384 time steps
                \item \textbf{Better representations}: patch-level tokens are more informative than point-level
            \end{itemize}
            }
        \end{exampleblock}

        \column{0.48\textwidth}
        \begin{center}
            \includegraphics[width=0.95\textwidth, height=0.45\textheight, keepaspectratio]{ch11_patching_illustration.pdf}
        \end{center}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Quantization: The Chronos Approach}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Scale-and-Quantize Tokenization (Ansari et al., 2024)}
        {\small
        \begin{enumerate}\setlength{\itemsep}{0pt}
            \item \textbf{Normalize}: $\tilde{x}_t = \frac{x_t - \bar{x}}{s}$ where $\bar{x}$ is mean and $s$ is the mean absolute value
            \item \textbf{Quantize}: map to $B$ bins via the CDF: $b_t = \left\lfloor (B-1) \cdot \Phi(\tilde{x}_t) \right\rfloor$ where $\Phi$ is the standard normal CDF
            \item \textbf{Embed}: look up bin $b_t$ in the T5 embedding table ($B = 4096$)
        \end{enumerate}
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{alertblock}{Why Quantize?}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Reuses \textbf{NLP vocabulary machinery}: embeddings, softmax output layer
                \item Cross-entropy loss (categorical) is well-understood
                \item Probabilistic: output is a distribution over bins $\Rightarrow$ full predictive distribution
                \item Trade-off: resolution limited by number of bins
            \end{itemize}
            }
        \end{alertblock}

        \column{0.48\textwidth}
        \begin{center}
            \includegraphics[width=0.95\textwidth, height=0.40\textheight, keepaspectratio]{ch11_chronos_tokenization.pdf}
        \end{center}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Pre-Training Objectives}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    {\scriptsize
    \begin{block}{Paradigm 1: Next-Token Prediction (Autoregressive, GPT-style)}
        \vspace{-1mm}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Predict $x_{t+1}$ given $x_1, \ldots, x_t$ via causal masking. Loss: $\mathcal{L} = -\sum_{t=1}^{T-1} \log p(x_{t+1} \mid x_1, \ldots, x_t)$
            \item Used by: \textbf{TimesFM}, \textbf{Lag-Llama}, \textbf{TimeGPT}
        \end{itemize}
        \vspace{-1mm}
    \end{block}
    \vspace{-2.5mm}
    \begin{alertblock}{Paradigm 2: Masked Prediction (BERT-style)}
        \vspace{-1mm}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Randomly mask 15--40\% of tokens, predict masked ones. Loss: $\mathcal{L} = -\sum_{t \in \mathcal{M}} \log p(x_t \mid x_{\setminus \mathcal{M}})$
            \item Used by: \textbf{Moirai}, \textbf{PatchTST} (self-supervised)
        \end{itemize}
        \vspace{-1mm}
    \end{alertblock}
    \vspace{-2.5mm}
    \begin{exampleblock}{Paradigm 3: Denoising / Seq2Seq (T5-style)}
        \vspace{-1mm}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Encoder processes context, decoder generates forecast. Loss: cross-entropy over quantized tokens
            \item Used by: \textbf{Chronos}
        \end{itemize}
        \vspace{-1mm}
    \end{exampleblock}
    }
    \vspace{-3mm}

\begin{frame}{Data Augmentation for Pre-Training}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{KernelSynth (Ansari et al., 2024)}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Generate \textbf{synthetic} time series by composing Gaussian Process (GP) kernels:
            $$k(t, t') = k_{\text{RBF}}(t,t') + k_{\text{periodic}}(t,t') + k_{\text{linear}}(t,t') + \ldots$$
            \item Produces realistic patterns: trends, seasonality, changepoints, noise
            \item Chronos uses \textbf{billions} of synthetic series to augment real data
        \end{itemize}
        }
    \end{block}

    \vspace{-1mm}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{TSMix (TimesFM)}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Mix existing series: $\tilde{x} = \lambda x_a + (1-\lambda) x_b$
                \item Augments diversity without needing GP expertise
                \item Combined with window slicing and frequency mixing
            \end{itemize}
            }
        \end{exampleblock}

        \column{0.48\textwidth}
        \begin{alertblock}{Why Augmentation Matters}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Prevents overfitting to specific domains
                \item Increases pattern diversity during pre-training
                \item Chronos: 50\% real data + 50\% synthetic $\Rightarrow$ \textbf{better} than 100\% real
            \end{itemize}
            }
        \end{alertblock}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Chronos: How It Forecasts}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Step-by-Step Pipeline}
        {\small
        \begin{enumerate}\setlength{\itemsep}{0pt}
            \item \textbf{Input}: raw time series $(x_1, x_2, \ldots, x_T)$
            \item \textbf{Normalize}: mean-scale $\tilde{x}_t = x_t / \bar{|x|}$
            \item \textbf{Quantize}: map each $\tilde{x}_t$ to one of 4096 bins via $\Phi$ (normal CDF)
            \item \textbf{Encode}: T5 encoder processes context tokens $\to$ hidden representations
            \item \textbf{Decode}: T5 decoder generates forecast tokens \textbf{autoregressively}
            \item \textbf{Sample}: repeat decoding $N$ times (e.g., $N=20$) with temperature sampling
            \item \textbf{Dequantize}: convert bin indices back to continuous values
            \item \textbf{Output}: median forecast + quantile intervals (10\%, 90\%)
        \end{enumerate}
        }
    \end{block}

\begin{frame}{Lag-Llama: LLM Backbone for Time Series}

        \begin{alertblock}{Why TS Scaling Differs from NLP}
            \begin{itemize}\setlength{\itemsep}{0pt}

\begin{frame}{When Foundation Models Fail}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-3mm}
    \begin{alertblock}{Failure Modes to Watch For}
        {\scriptsize
        \begin{enumerate}\setlength{\itemsep}{0pt}
            \item \textbf{Distribution shift}: Financial regime changes (COVID crash, rate hikes). Pre-training data may not cover such regimes.
            \item \textbf{Missing exogenous variables}: Weather affects energy demand; policy changes affect economics. Most foundation models are \textbf{univariate}.
            \item \textbf{Rare frequencies}: Irregular sampling, event-driven or tick-by-tick data. Pre-training favors regular frequencies.
            \item \textbf{Very short series}: Fewer than $\sim$50 observations $\Rightarrow$ insufficient context. Classical methods (ETS, ARIMA) often dominate.
            \item \textbf{Multivariate dependencies}: Cross-variable correlations (e.g., VAR) ignored by univariate models. Only Moirai attempts multivariate support.
        \end{enumerate}
        }
    \end{alertblock}
    \vspace{-3mm}
    \begin{exampleblock}{Rule of Thumb}
        {\scriptsize If your data has \textbf{known causal structure}, \textbf{domain-specific features}, or \textbf{regime changes}, consider combining foundation model forecasts with domain-informed models (ensemble or hierarchical).}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Zero-Shot: When and Why}
    \begin{cminipage}{0.95\textwidth}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{exampleblock}{When to Use Zero-Shot}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{Limited labeled data}: new product launch, cold-start
                \item \textbf{Rapid prototyping}: quick baseline in minutes
                \item \textbf{Cross-domain deployment}: same model for finance, retail, energy
                \item \textbf{Many series}: 10{,}000+ series to forecast, no time to train each
                \item \textbf{No ML expertise}: simple API call
            \end{itemize}
            }
        \end{exampleblock}

        \column{0.48\textwidth}
        \begin{center}
            \includegraphics[width=0.95\textwidth, height=0.45\textheight, keepaspectratio]{ch11_zeroshot_vs_finetuning.pdf}

    \vspace{-2mm}
    \begin{alertblock}{Surprising Finding}
        {\small Ansari et al.\ (2024): Chronos zero-shot outperforms task-specific models (DeepAR, N-BEATS) that were \textit{trained on the target data}. Pre-training transfers temporal knowledge effectively.}
    \end{alertblock}

\end{frame}

\begin{frame}{Fine-Tuning: Domain Adaptation}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Full Fine-Tuning vs LoRA}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Full fine-tuning}: update all parameters on domain data (expensive, risk of forgetting)
            \item \textbf{LoRA} (Low-Rank Adaptation, Hu et al., 2022): freeze base weights, add small trainable matrices
            $$W' = W + \Delta W = W + BA, \quad B \in \mathbb{R}^{d \times r},\; A \in \mathbb{R}^{r \times d},\; r \ll d$$
            \item LoRA: 0.1--1\% of trainable parameters, similar performance, much cheaper
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{alertblock}{Risks of Fine-Tuning}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{Catastrophic forgetting}: model loses pre-trained knowledge
                \item \textbf{Overfitting}: small domain dataset $\Rightarrow$ poor generalization
                \item \textbf{Hyperparameter sensitivity}: learning rate, epochs, warmup critical
            \end{itemize}
            }
        \end{alertblock}

        \column{0.48\textwidth}
        \begin{exampleblock}{Best Practices}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Start with \textbf{low learning rate} ($10^{-5}$--$10^{-4}$)
                \item Use \textbf{learning rate warmup} (500--1000 steps)
                \item \textbf{Early stopping} on validation loss
                \item \textbf{Freeze early layers}, fine-tune later layers
                \item Prefer \textbf{LoRA} for small datasets ($<$10K series)
            \end{itemize}
            }
        \end{exampleblock}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Few-Shot and In-Context Learning}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Context Length as ``Few-Shot'' for Time Series}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item In NLP, ``few-shot'' = provide examples in the prompt. In TS, ``context'' = historical observations
            \item \textbf{Longer context $\to$ better forecasts} (up to the model's context window)
            \item The model conditions on historical patterns to infer trend, seasonality, and noise structure
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \begin{center}
            \includegraphics[width=0.95\textwidth, height=0.38\textheight, keepaspectratio]{ch11_context_length_effect.pdf}
        \end{center}
        {\scriptsize Forecast error vs.\ context length: diminishing returns beyond the model's window.}

        \column{0.48\textwidth}
        {\scriptsize
        \begin{exampleblock}{Context Windows by Model}
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Chronos: 512 tokens ($\approx$ 512 time steps)
                \item TimesFM: 512 patches ($\approx$ 16{,}384 steps)
                \item Moirai: 5000 tokens (configurable)
                \item Lag-Llama: 1024 steps (with lags extending further)
                \item TimeGPT: configurable via API
            \end{itemize}
        \end{exampleblock}
        \vspace{-2mm}
        \begin{alertblock}{Practical Tip}
            Provide as much context as the model allows. For daily data, $\geq$1 year is ideal. For monthly data, $\geq$5 years captures seasonal cycles.
        \end{alertblock}
        }
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Decision Framework}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{When to Choose Zero-Shot, Fine-Tuning, or Classical}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Step 1}: Do you have labeled domain data? No $\to$ Zero-shot
            \item \textbf{Step 2}: Is zero-shot performance sufficient? Yes $\to$ Deploy; No $\to$ Fine-tune
            \item \textbf{Step 3}: Is the domain very different from pre-training data? Yes $\to$ Full fine-tune; No $\to$ LoRA
            \item \textbf{Step 4}: Need interpretability or causal reasoning? Yes $\to$ Classical models (ARIMA, VAR)
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \begin{center}

\section{Limitations, Ethics, and Future}
%=============================================================================

\begin{frame}{Current Limitations}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{alertblock}{Five Key Limitations of Foundation Models for Time Series}
        {\small
        \begin{enumerate}\setlength{\itemsep}{2pt}
            \item \textbf{Most models are univariate only}: Cannot capture cross-variable interactions (e.g., price-volume, temperature-demand). Only Moirai attempts multivariate.
            \item \textbf{Limited context windows (512--4096 tokens)}: Cannot leverage very long histories. For daily data with yearly seasonality, 512 steps $\approx$ 1.5 years only.
            \item \textbf{No causal reasoning}: Models learn correlations, not causation. Cannot answer ``what if interest rates rise by 1\%?'' --- no structural model underneath.
            \item \textbf{Black box nature}: Attention weights are not easily interpretable. Difficult to explain \textit{why} a particular forecast was made --- problematic for regulated industries.
            \item \textbf{``Hallucination'' in probabilistic forecasts}: Generated trajectories may exhibit unrealistic patterns (e.g., negative prices, explosive growth) that violate domain constraints.
        \end{enumerate}
        }
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{Mitigation}
        {\small Post-processing (clipping, reconciliation), ensemble with classical models, conformal calibration.}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Ethical Considerations}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Data Provenance}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Pre-training data may include \textbf{private or proprietary} time series (e.g., internal sales data scraped from APIs)
            \item Lack of transparency: most models do not fully disclose training data composition
            \item LOTSA (Moirai) is a positive example: open, curated, documented sources
        \end{itemize}
        }
    \end{block}
    \vspace{-2mm}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{alertblock}{Fairness and Bias}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item Pre-training skews toward \textbf{data-rich domains} (energy, retail)
                \item Under-represented: healthcare in developing countries, agriculture in Africa
                \item Biased training $\to$ biased forecasts $\to$ biased decisions
            \end{itemize}
            }
        \end{alertblock}

        \column{0.48\textwidth}
        \begin{alertblock}{Environmental and Regulatory}
            {\scriptsize
            \begin{itemize}\setlength{\itemsep}{0pt}
                \item \textbf{Carbon footprint}: pre-training Chronos-Large $\approx$ 100+ GPU-hours on A100s
                \item \textbf{EU AI Act}: high-risk AI systems require explainability --- foundation models may not qualify
                \item \textbf{Accountability}: who is responsible when a forecast-based decision fails?
            \end{itemize}
            }
        \end{alertblock}
    \end{columns}
    \end{cminipage}
\end{frame}

\begin{frame}{Future Directions}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{block}{Where Is the Field Heading?}
        {\small
        \begin{enumerate}\setlength{\itemsep}{2pt}
            \item \textbf{Multimodal TS models}: Combine time series with text (news, reports), images (satellite), and metadata. Example: forecast sales using both historical data and product descriptions.
            \item \textbf{Agentic forecasting}: LLM agents that \textit{select, run, and evaluate} forecasting models autonomously. The LLM orchestrates the pipeline rather than producing the forecast directly.
            \item \textbf{Continual pre-training}: Update foundation models with new data without full retraining. Critical for adapting to distribution shifts (post-COVID, post-pandemic).
            \item \textbf{Multivariate foundation models}: Move beyond univariate. Learn cross-variable dynamics, hierarchical structures, and spatial correlations.
            \item \textbf{Domain-specific foundation models}: FinanceFM, EnergyFM, HealthFM --- smaller, specialized, more accurate within their domain.
        \end{enumerate}
        }
    \end{block}
    \vspace{-2mm}
    \begin{exampleblock}{The Next 3--5 Years}
        {\small Foundation models will likely become the \textbf{default starting point} for forecasting, much as pre-trained word embeddings became standard in NLP. Classical methods will remain essential for interpretability, causal inference, and small-data regimes.}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}{A Balanced Perspective}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-2mm}
    \begin{alertblock}{When to Use Classical Models}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Interpretability required}: regulatory settings, scientific research, policy analysis
            \item \textbf{Causal inference}: structural models (VAR, SVAR) for ``what if'' analysis
            \item \textbf{Small data}: fewer than $\sim$50 observations, seasonal structure known a priori
            \item \textbf{Computational constraints}: edge devices, real-time systems, no GPU access
            \item \textbf{Explainability mandate}: EU AI Act, financial regulation (MiFID II, Basel III)
        \end{itemize}
        }
    \end{alertblock}
    \vspace{-2mm}
    \begin{exampleblock}{When to Use Foundation Models}
        {\small
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item \textbf{Rapid prototyping}: get a strong baseline in minutes, not days
            \item \textbf{Cross-domain}: same model for retail, energy, finance --- no domain expertise needed
            \item \textbf{Zero-shot / cold-start}: new product, new market, no historical data
            \item \textbf{Scale}: 10{,}000+ series to forecast --- foundation models amortize the cost
            \item \textbf{Probabilistic forecasts}: native uncertainty quantification (quantiles, prediction intervals)
        \end{itemize}
        }
    \end{exampleblock}
    \vspace{-2mm}
    \begin{block}{}
        {\small \textbf{Bottom line}: Classical and foundation models are \textbf{complementary}, not competing. The best practitioners use both.}
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Key Takeaways}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{What We Learned}
        \begin{itemize}\setlength{\itemsep}{2pt}
            \item \textbf{Transformers} replace recurrence with self-attention: parallel, scalable, long-range dependencies
            \item \textbf{Foundation models} are pre-trained on billions of diverse time points and deployed zero-shot
            \item \textbf{Tokenization} strategies (patching, quantization) bridge continuous TS and discrete Transformer inputs
            \item \textbf{Chronos} (T5, quantized), \textbf{TimesFM} (decoder, patched), \textbf{Lag-Llama} (LLaMA, lags), \textbf{Moirai} (BERT, any-variate) each have distinct design philosophies

        \end{itemize}
    \end{block}

    \vspace{0.1cm}

    \begin{alertblock}{Practical Recommendations}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Always benchmark against \textbf{ARIMA/ETS} as a classical baseline
            \item Use \textbf{foundation models} for rapid prototyping and when domain expertise is limited
            \item Classical and foundation models are \textbf{complementary} --- combine them via ensembles
        \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 1}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Question}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item What is the key advantage of self-attention over recurrence in sequence modeling?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Answer Choices}

        \textcolor{MainBlue}{\textbf{(A)}} Self-attention uses fewer parameters than RNNs\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} Self-attention processes all positions in parallel with $O(1)$ path length between any two positions\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Self-attention does not require positional encoding\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Self-attention is always faster than recurrence regardless of sequence length

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 1: Answer}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    \begin{center}
        \includegraphics[width=0.98\textwidth, height=0.58\textheight, keepaspectratio]{ch11_quiz1_attention_vs_rnn.pdf}
    \end{center}
    \vspace{-3mm}
    {\small
    \begin{exampleblock}{Answer: (B)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Self-attention connects every position to every other position directly (path length $O(1)$), enabling full parallelization. RNNs require $O(T)$ sequential steps, causing vanishing gradients and slow training.
    \end{itemize}
    \end{exampleblock}
    }
    \hfill\quantlet{TSA\_ch11\_quiz1\_attention\_vs\_rnn}{https://github.com/QuantLet/TSA/tree/main/TSA_ch11/TSA_ch11_quiz1_attention_vs_rnn}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 2}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Question}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item Which tokenization strategy does Chronos use to convert continuous time series into discrete tokens?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Answer Choices}

        \textcolor{MainBlue}{\textbf{(A)}} Patching: grouping consecutive time points into patches\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} Scaling and quantization into discrete bins using the normal CDF\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Direct linear embedding of raw values\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Byte-pair encoding (BPE) borrowed from NLP

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 2: Answer}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    \begin{center}
        \includegraphics[width=0.98\textwidth, height=0.58\textheight, keepaspectratio]{ch11_quiz2_tokenization.pdf}
    \end{center}
    \vspace{-3mm}
    {\small
    \begin{exampleblock}{Answer: (B)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Chronos normalizes values (mean-scaling), then maps to 4096 bins via $b_t = \lfloor (B-1) \cdot \Phi(\tilde{x}_t) \rfloor$. This allows reuse of the T5 vocabulary and cross-entropy loss, making forecasting a language modeling problem over quantized tokens.
    \end{itemize}
    \end{exampleblock}
    }
    \hfill\quantlet{TSA\_ch11\_quiz2\_tokenization}{https://github.com/QuantLet/TSA/tree/main/TSA_ch11/TSA_ch11_quiz2_tokenization}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 3}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Question}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item When should you prefer zero-shot inference over fine-tuning a foundation model?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Answer Choices}

        \textcolor{MainBlue}{\textbf{(A)}} When you have millions of labeled domain-specific observations\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} When you need the highest possible accuracy on a single dataset\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} When you have limited labeled data, need rapid deployment, or face cold-start problems\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} When computational resources are not a concern

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 3: Answer}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    \begin{center}
        \includegraphics[width=0.98\textwidth, height=0.58\textheight, keepaspectratio]{ch11_quiz3_zeroshot_decision.pdf}
    \end{center}
    \vspace{-3mm}
    {\small
    \begin{exampleblock}{Answer: (C)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Zero-shot excels with limited data, cold-start, and rapid prototyping. The model's pre-trained knowledge transfers temporal patterns from billions of diverse series. Fine-tuning is preferred when sufficient domain data is available and marginal accuracy gains justify the cost.
    \end{itemize}
    \end{exampleblock}

\begin{frame}{Question 4}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Question}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item What is ``patching'' in the context of Transformer-based time series models, and why does it help?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Answer Choices}

        \textcolor{MainBlue}{\textbf{(A)}} Patching means filling missing values in the time series\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} Patching groups $P$ consecutive points into one token, reducing attention complexity from $O(T^2)$ to $O((T/P)^2)$ while capturing local patterns\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Patching is a data augmentation technique that creates new training series\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Patching refers to updating model weights after deployment

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 4: Answer}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    \begin{center}
        \includegraphics[width=0.98\textwidth, height=0.58\textheight, keepaspectratio]{ch11_quiz4_patching.pdf}
    \end{center}
    \vspace{-3mm}
    {\small
    \begin{exampleblock}{Answer: (B)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Patching (PatchTST, 2023) groups $P$ consecutive points into a single token. This reduces the effective sequence length for attention, captures local temporal patterns within each patch, and enables the model to handle much longer context windows.
    \end{itemize}
    \end{exampleblock}
    }
    \hfill\quantlet{TSA\_ch11\_quiz4\_patching}{https://github.com/QuantLet/TSA/tree/main/TSA_ch11/TSA_ch11_quiz4_patching}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 5}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Question}
        \begin{itemize}\setlength{\itemsep}{0pt}
            \item You need to forecast a multivariate energy dataset with 50 correlated variables at hourly frequency. Which foundation model is the best fit?
        \end{itemize}
    \end{alertblock}

    \vspace{0.3cm}

    \begin{block}{Answer Choices}

        \textcolor{MainBlue}{\textbf{(A)}} Chronos --- because it has the most model sizes to choose from\\[3pt]

        \textcolor{MainBlue}{\textbf{(B)}} TimeGPT --- because it is the easiest to use via API\\[3pt]

        \textcolor{MainBlue}{\textbf{(C)}} Moirai --- because it is the only model designed to handle any number of variates and any frequency natively\\[3pt]

        \textcolor{MainBlue}{\textbf{(D)}} Lag-Llama --- because it uses lag features that capture temporal dependencies

    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}{Question 5: Answer}
    \begin{cminipage}{0.95\textwidth}
    \vspace{-0.2cm}
    \begin{center}
        \includegraphics[width=0.98\textwidth, height=0.58\textheight, keepaspectratio]{ch11_quiz5_model_selection.pdf}
    \end{center}
    \vspace{-3mm}
    {\small
    \begin{exampleblock}{Answer: (C)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Moirai is the only foundation model designed for \textbf{any-variate} forecasting. It uses variate IDs and flexible patching to handle 50 correlated variables simultaneously. Chronos and Lag-Llama are univariate; they would process each variable independently, ignoring cross-variable correlations.
    \end{itemize}
    \end{exampleblock}
    }
